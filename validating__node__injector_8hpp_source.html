62 // peer info getter
63 template <typename Injector>
64 sptr<network::OwnPeerInfo> get_peer_info(const Injector &injector,
65 uint16_t p2p_port) {
66 static auto initialized =
67 boost::optional<sptr<network::OwnPeerInfo>>(boost::none);
68 if (initialized) {
69 return initialized.value();
70 }
71
72 // get key storage
73 auto &keys = injector.template create<application::KeyStorage &>();
74 auto &&local_pair = keys.getP2PKeypair();
75 libp2p::crypto::PublicKey &public_key = local_pair.publicKey;
76 auto &key_marshaller =
77 injector.template create<libp2p::crypto::marshaller::KeyMarshaller &>();
78
79 libp2p::peer::PeerId peer_id =
80 libp2p::peer::PeerId::fromPublicKey(
81 key_marshaller.marshal(public_key).value())
82 .value();
83 spdlog::debug("Received peer id: {}", peer_id.toBase58());
84 std::string multiaddress_str =
85 "/ip4/0.0.0.0/tcp/" + std::to_string(p2p_port);
86 spdlog::debug("Received multiaddr: {}", multiaddress_str);
87 auto multiaddress = libp2p::multi::Multiaddress::create(multiaddress_str);
88 if (!multiaddress) {
89 base::raise(multiaddress.error()); // exception
90 }
91 std::vector<libp2p::multi::Multiaddress> addresses;
92 addresses.push_back(std::move(multiaddress.value()));
93
94 initialized = std::make_shared<network::OwnPeerInfo>(std::move(peer_id),
95 std::move(addresses));
96 return initialized.value();
97 }
