342void sgns::message_parser::deserialize_buffer (uint8_t const * buffer_a, size_t size_a)
343{
344 static sgns::network_constants network_constants;
345 status = parse_status::success;
346 auto error (false);
347 if (size_a <= max_safe_udp_message_size)
348 {
349 // Guaranteed to be deliverable
350 sgns::bufferstream stream (buffer_a, size_a);
351 sgns::message_header header (error, stream);
352 if (!error)
353 {
354 if (header.version_using < get_protocol_constants ().protocol_version_min (use_epoch_2_min_version))
355 {
356 status = parse_status::outdated_version;
357 }
358 else
359 {
360 switch (header.type)
361 {
362 case sgns::message_type::keepalive:
363 {
364 deserialize_keepalive (stream, header);
365 break;
366 }
367 case sgns::message_type::publish:
368 {
369 sgns::uint128_t digest;
370 if (!publish_filter.apply (buffer_a + header.size, size_a - header.size, &digest))
371 {
372 deserialize_publish (stream, header, digest);

378 break;
379 }

381 {
382 deserialize_confirm_req (stream, header);
383 break;
384 }
385 case sgns::message_type::confirm_ack:
386 {

388 break;
389 }
390 case sgns::message_type::node_id_handshake:
391 {
392 deserialize_node_id_handshake (stream, header);

394 }

930 * is a limit specifed. Additionally, do not allow
931 * the "count_present" flag with a value of 0, since
932 * that is a sentinel which we use to mean "all blocks"
933 * and that is the behavior of not having the flag set
934 * so it is wasteful to do this.
935 */

937

939 write (stream_a, start);
940 write (stream_a, end);
941
942 if (is_count_present ())
943 {
944 std::array<uint8_t, extended_parameters_size> count_buffer{ { 0 } };
945 decltype (count) count_little_endian;
946 static_assert (sizeof (count_little_endian) < (count_buffer.size () - 1), "count must fit within buffer");
947
948 count_little_endian = boost::endian::native_to_little (count);
949 memcpy (count_buffer.data () + 1, &count_little_endian, sizeof (count_little_endian));
950
951 write (stream_a, count_buffer);
952 }
