124 bool RouterLibp2p::processGossipMessage(const GossipMessage &msg) const {
125 using MsgType = GossipMessage::Type;
126
127 switch (msg.type) {
128 case MsgType::BLOCK_ANNOUNCE: {
129 auto msg_res = scale::decode<BlockAnnounce>(msg.data);
130 if (!msg_res) {
131 log_->error("error while decoding a block announce message: {}",
132 msg_res.error().message());
133 return false;
134 }
135 log_->info("Received block announce: block number {}",
136 msg_res.value().header.number);
137 production_observer_->onBlockAnnounce(msg_res.value());
138 return true;
139 }
140 case MsgType::VERIFICATION: {
141 auto vote_msg_res =
142 scale::decode<verification::finality::VoteMessage>(msg.data);
143 if (vote_msg_res) {
144 finality_observer_->onVoteMessage(vote_msg_res.value());
145 return true;
146 }
147
148 auto fin_msg_res = scale::decode<verification::finality::Fin>(msg.data);
149 if (fin_msg_res) {
150 finality_observer_->onFinalize(fin_msg_res.value());
151 return true;
152 }
153
154 log_->error("error while decoding a verification message");
155 return false;
156 }
157 case MsgType::TRANSACTIONS: {
158 auto txs_msg_res =
159 scale::decode<std::vector<primitives::Extrinsic>>(msg.data);
160
161 if (! txs_msg_res) {
162 log_->error("error while decoding a transactions message");
163 return false;
164 }
165
166 log_->info("Received tx announce: {} txs", txs_msg_res.value().size());
167
168 for (auto &extrinsic : txs_msg_res.value()) {
169 auto result = extrinsic_observer_->onTxMessage(extrinsic);
170 if (result) {
171 log_->debug(" Received tx {}", result.value());
172 } else {
173 log_->debug(" Rejected tx: {}", result.error().message());
174 }
175 }
176 return true;
177 }
178 case GossipMessage::Type::STATUS: {
179 log_->error("Status message processing is not implemented yet");
180 return false;
181 }
182 case GossipMessage::Type::BLOCK_REQUEST: {
183 log_->error("BlockRequest message processing is not implemented yet");
184 return false;
185 }
186 case GossipMessage::Type::UNKNOWN: {
187 log_->error("unknown message type is set");
188 return false;
189 }
190 }
191 }
