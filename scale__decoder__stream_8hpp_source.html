137 template <class T>
138 ScaleDecoderStream &operator>>(boost::optional<T> &v) {
139 using mutableT = std::remove_const_t<T>;
140
141 static_assert(std::is_default_constructible_v<mutableT>);
142
143 // optional bool is special case of optional values
144 // it is encoded as one byte instead of two
145 // as described in specification
146 if constexpr (std::is_same<mutableT, bool>::value) {
147 v = decodeOptionalBool();
148 return *this;
149 }
150 // detect if optional has value
151 bool has_value = false;
152 *this >> has_value;
153 if (!has_value) {
154 v.reset();
155 return *this;
156 }
157 // decode value
158 v.emplace();
159 return *this >> const_cast<mutableT &>(*v); // NOLINT

175 template <class T>
176 ScaleDecoderStream &operator>>(std::vector<T> &v) {
177 using mutableT = std::remove_const_t<T>;
178 using size_type = typename std::list<T>::size_type;
179
180 static_assert(std::is_default_constructible_v<mutableT>);
181
182 CompactInteger size{0u};
183 *this >> size;
184
185 if (size > std::numeric_limits<size_type>::max()) {
186 base::raise(DecodeError::TOO_MANY_ITEMS);
187 }
188 auto item_count = size.convert_to<size_type>();
189
190 std::vector<mutableT> vec;
191 vec.resize(item_count);
192
193 for (size_type i = 0u; i < item_count; ++i) {
194 *this >> vec[i];
195 }
196
197 v = std::move(vec);
