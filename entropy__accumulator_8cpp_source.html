27 outcome::result<std::vector<uint8_t>> EntropyAccumulator::getEntropy() const {
28 if (bits_.size() != total_bits_count_) {
29 return Bip39EntropyError::STORAGE_NOT_COMPLETE;
30 }
31
32 // convert data
33 size_t bytes_count = (total_bits_count_ - checksum_bits_count_) / 8;
34 std::vector<uint8_t> res;
35 res.reserve(bytes_count);
36 auto it = bits_.begin();
37 for (size_t i = 0; i < bytes_count; ++i) {
38 uint8_t byte = 0;
39 for (size_t j = 0; j < 8u; ++j) {
40 byte <<= 1u;
41 byte += *it++;
42 }
43
44 res.push_back(byte);
45 }
46
47 return res;
48 }

50 outcome::result<uint8_t> EntropyAccumulator::getChecksum() const {
51 if (bits_.size() != total_bits_count_) {
52 return Bip39EntropyError::STORAGE_NOT_COMPLETE;
53 }
54
55 uint8_t checksum = 0u;
56 auto it = bits_.rbegin();
57 for (auto i = 0u; i < checksum_bits_count_; ++i) {
58 checksum += (*it++ << i);
59 }
60
61 return checksum;

64 outcome::result<void> EntropyAccumulator::append(const EntropyToken &value) {
65 if (bits_.size() + value.size() > total_bits_count_) {
66 return Bip39EntropyError::STORAGE_IS_FULL;
67 }
68
69 for (size_t i = 0; i < value.size(); ++i) {
70 // bits order is little-endian, but we need big endian, reverse it
71 auto position = value.size() - i - 1;
72 uint8_t v = value.test(position) ? 1 : 0;
73 bits_.push_back(v);
74 }
75
76 return outcome::success();
77 }

79 outcome::result<uint8_t> EntropyAccumulator::calculateChecksum() const {
80 OUTCOME_TRY((auto &&, entropy), getEntropy());
81 auto hash = sha256(entropy);
82 return hash[0] >> static_cast<uint8_t>(8 - checksum_bits_count_);
