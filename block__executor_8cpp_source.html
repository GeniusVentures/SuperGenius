46 void BlockExecutor::processNextBlock(
47 const primitives::BlockHeader &header,
48 const std::function<void(const primitives::BlockHeader &)>
49 &new_block_handler) {
50 auto block_hash = hasher_->blake2b_256(scale::encode(header).value());
51
52 // insert block_header if it is missing
53 if (! block_tree_->getBlockHeader(block_hash)) {
54 new_block_handler(header);
55 logger_->info("Received block header. Number: {}, Hash: {}",
56 header.number,
57 block_hash.toHex());
58
59 auto [_, production_header] = getProductionDigests(header).value();
60
61 if (! block_tree_->getBlockHeader(header.parent_hash)) {
62 const auto &[last_number, last_hash] = block_tree_->getLastFinalized();
63 // we should request blocks between last finalized one and received
64 // block
65 requestBlocks(
66 last_hash, block_hash, production_header.authority_index, [] {});
67 } else {
68 requestBlocks(
69 header.parent_hash, block_hash, production_header.authority_index, [] {});
70 }
71 }
72 }

87 void BlockExecutor::requestBlocks(const primitives::BlockId &from,
88 const primitives::BlockHash &to,
89 primitives::AuthorityIndex authority_index,
90 std::function<void()> &&next) {
91 production_synchronizer_->request(
92 from,
93 to,
94 authority_index,
95 [self_wp{weak_from_this()},
96 next(std::move(next))](const std::vector<primitives::Block> &blocks) {
97 auto self = self_wp.lock();
98 if (! self) return;
99
100 if (blocks.empty()) {
101 self->logger_->warn("Received empty list of blocks");
102 } else {
103 auto front_block_hex =
104 self->hasher_
105 ->blake2b_256(scale::encode(blocks.front().header).value())
106 .toHex();
107 auto back_block_hex =
108 self->hasher_
109 ->blake2b_256(scale::encode(blocks.back().header).value())
110 .toHex();
111 self->logger_->info("Received blocks from: {}, to {}",
112 front_block_hex,
113 back_block_hex);
114 }
115 for (const auto &block : blocks) {
116 if (auto apply_res = self->applyBlock(block); ! apply_res) {
117 if (apply_res
118 == outcome::failure(
119 blockchain::BlockTreeError::BLOCK_EXISTS)) {
120 continue;
121 }
122 self->logger_->warn(
123 "Could not apply block during synchronizing slots.Error: {}",
124 apply_res.error().message());
125 break;
126 }
127 }
128 next();
129 });
130 }

132 outcome::result<void> BlockExecutor::applyBlock(
133 const primitives::Block &block) {
134 auto block_hash = hasher_->blake2b_256(scale::encode(block.header).value());
135
136 // check if block body already exists. If so, do not apply
137 if (block_tree_->getBlockBody(block_hash)) {
138 return blockchain::BlockTreeError::BLOCK_EXISTS;
139 }
140 logger_->info(
141 "Applying block number: {}, hash: {}",
142 block.header.number,
143 hasher_->blake2b_256(scale::encode(block.header).value()).toHex());
144
145 OUTCOME_TRY((auto &&, production_digests), getProductionDigests(block.header));
146
147 auto [seal, production_header] = production_digests;
148
149 auto epoch_index =
150 production_header.slot_number / genesis_configuration_->epoch_length;
151
152 // TODO (kamilsa): PRE-364 uncomment outcome try and remove dirty workaround
153 // below
154 // OUTCOME_TRY((auto &&, this_block_epoch_descriptor),
155 // epoch_storage_->getEpochDescriptor(epoch_index));
156 auto this_block_epoch_descriptor_res =
157 epoch_storage_->getEpochDescriptor(epoch_index);
158 if (! this_block_epoch_descriptor_res) { // take authorities and
159 // randomness
160 // from config
161 this_block_epoch_descriptor_res = NextEpochDescriptor{
162 /*.authorities =*/ genesis_configuration_->genesis_authorities,
163 /*.randomness =*/ genesis_configuration_->randomness};
164 }
165 auto this_block_epoch_descriptor = this_block_epoch_descriptor_res.value();
166
167 auto threshold = calculateThreshold(genesis_configuration_->leadership_rate,
168 this_block_epoch_descriptor.authorities,
169 production_header.authority_index);
170
171 // update authorities and randomnesss
172 auto next_epoch_digest_res = getNextEpochDigest(block.header);
173 if (next_epoch_digest_res) {
174 logger_->info("Got next epoch digest for epoch: {}", epoch_index);
175 epoch_storage_
176 ->addEpochDescriptor(epoch_index + 2, next_epoch_digest_res.value())
177 .value();
178 }
179
180 BOOST_OUTCOME_TRYV2(auto &&, block_validator_->validateHeader(
181 block.header,
182 this_block_epoch_descriptor.authorities[production_header.authority_index].id,
183 threshold,
184 this_block_epoch_descriptor.randomness));
185
186 auto block_without_seal_digest = block;
187
188 // block should be applied without last digest which contains the seal
189 block_without_seal_digest.header.digest.pop_back();
190 // apply block
191 BOOST_OUTCOME_TRYV2(auto &&, core_->execute_block(block_without_seal_digest));
192
193 // add block header if it does not exist
194 BOOST_OUTCOME_TRYV2(auto &&, block_tree_->addBlock(block));
195
196 // observe possible changes of authorities
197 for (auto &digest_item : block_without_seal_digest.header.digest) {
198 BOOST_OUTCOME_TRYV2(auto &&, visit_in_place(
199 digest_item,
200 [&](const primitives::Verification &verification_message)
201 -> outcome::result<void> {
202 return authority_update_observer_->onVerification(
203 verification_message.verification_engine_id,
204 primitives::BlockInfo{block.header.number, block_hash},
205 verification_message);
206 },
207 [](const auto &) { return outcome::success(); }));
208 }
209
210 // remove block's extrinsics from tx pool
211 for (const auto &extrinsic : block.body) {
212 auto res = tx_pool_->removeOne(hasher_->blake2b_256(extrinsic.data));
213 if (res.has_error()
214 && res
215 != outcome::failure(
216 transaction_pool::TransactionPoolError::TX_NOT_FOUND)) {
217 return res;
218 }
219 }
220
221 logger_->info("Imported block with number: {}, hash: {}",
222 block.header.number,
223 block_hash.toHex());
224 return outcome::success();
225 }
