33 void ValidatingNodeApplication::run() {

35

38 app_state_manager_->atLaunch([this] {
39 // execute listeners
40 io_context_->post([this] {
41 const auto &current_peer_info =
42 injector_.template create<network::OwnPeerInfo>();
43 auto &host = injector_.template create<libp2p::Host &>();
44 for (const auto &ma : current_peer_info.addresses) {
45 auto listen = host.listen(ma);
46 if (! listen) {
47 logger_->error("Cannot listen address {}. Error: {}",
