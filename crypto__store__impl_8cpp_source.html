17 KeyTypeId keyTypeFromBytes(gsl::span<const T> bytes) {
18 BOOST_ASSERT_MSG(bytes.size() == 4, "Wrong span size");
19
20 KeyTypeId res = static_cast<uint32_t>(bytes[3])
21 + (static_cast<uint32_t>(bytes[2]) << 8)
22 + (static_cast<uint32_t>(bytes[1]) << 16)
23 + (static_cast<uint32_t>(bytes[0]) << 24);
24
25 return res;
26 }

51 CryptoStoreImpl::parseKeyFileName(std::string_view file_name) const {
52 if (file_name.size() < 4 + store::PublicKey::size()) {
53 return CryptoStoreError::WRONG_KEYFILE_NAME;

56 auto key_type_str = file_name.substr(0, 4);

62 //---------- fix ----------------//

64 logger_->warn("key type <{}> is not officially supported", key_type_str);

66 auto public_key_hex = file_name.substr(4);

68 OUTCOME_TRY((auto &&, public_key), store::PublicKey::fromHex(public_key_hex));

136 outcome::result<ED25519Keypair> CryptoStoreImpl::generateEd25519Keypair(
137 KeyTypeId key_type, std::string_view mnemonic_phrase) {
138 OUTCOME_TRY((auto &&, mnemonic), bip39::Mnemonic::parse(mnemonic_phrase));
139 OUTCOME_TRY((auto &&, entropy), bip39_provider_->calculateEntropy(mnemonic.words));
140 OUTCOME_TRY((auto &&, seed), bip39_provider_->makeSeed(entropy, mnemonic.password));
141 if (seed.size() < ED25519Seed::size()) {
142 return CryptoStoreError::WRONG_SEED_SIZE;
143 }
144
145 OUTCOME_TRY((auto &&, ed_seed),
146 ED25519Seed::fromSpan(
147 gsl::make_span(seed).subspan(0, ED25519Seed::size())));
148 auto &&pair = ed25519_provider_->generateKeypair(ed_seed);
149
150 ed_keys_[key_type].emplace(pair.public_key, pair.private_key);
151
152 return pair;
153 }

212 outcome::result<ED25519Keypair> CryptoStoreImpl::findEd25519Keypair(
213 KeyTypeId key_type, const ED25519PublicKey &pk) const {
214 // try find in memory
215 if (ed_keys_.count(key_type) > 0) {
216 const auto &map = ed_keys_.at(key_type);
217 if (auto it = map.find(pk); it != map.end()) {
218 return ED25519Keypair{it->second, it->first};
219 }
220 }
221 // try find in filesystem
222 auto path = composeKeyPath(key_type, pk);
223 if (!boost::filesystem::exists(path)) {
224 return CryptoStoreError::KEY_NOT_FOUND;
225 }
226 OUTCOME_TRY((auto &&, content), loadFileContent(path));
227 OUTCOME_TRY((auto &&, seed), ED25519Seed::fromHex(content));
228
229 return ed25519_provider_->generateKeypair(seed);
230 }

252 CryptoStore::ED25519Keys CryptoStoreImpl::getEd25519PublicKeys(
253 KeyTypeId key_type) const {
254 std::set<ED25519PublicKey> keys;
255 // iterate over in-memory map
256 if (ed_keys_.find(key_type) != ed_keys_.end()) {
257 const auto &map = ed_keys_.at(key_type);
258 for (const auto &k : map) {
259 keys.emplace(k.first);
260 }
261 }
262
263 // iterate over fs
264 for (DirectoryIterator it(keys_directory_), end{}; it != end; ++it) {
265 if (!boost::filesystem::is_regular_file(*it)) {
266 continue;
267 }
268 auto info = parseKeyFileName(it->path().filename().string());
269 if (!info) {
270 continue;
271 }
272 auto &[id, pk] = info.value();
273 if (id == key_type && keys.count(pk) == 0) {
274 auto &&content = loadFileContent(it->path());
275 if (!content) {
276 logger_->error("failed to load keyfile {} : {}",
277 it->path().string(),
278 content.error().message());
279 continue;
280 }
281
282 auto &&seed = ED25519Seed::fromHex(content.value());
283 if (!seed) {
284 logger_->error("failed to load seed from keyfile {} : {}",
285 it->path().string(),
286 seed.error().message());
287 continue;
288 }
289 auto &&pair = ed25519_provider_->generateKeypair(seed.value());
290 if (pair.public_key == pk) {
291 keys.emplace(pk);
292 }
293 }
294 }
295
296 return ED25519Keys(keys.begin(), keys.end());
297 }
