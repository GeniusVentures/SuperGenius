82 void LoopbackStream::write(gsl::span<const uint8_t> in,
83 size_t bytes,
84 libp2p::basic::Writer::WriteCallbackFunc cb) {
85 if (is_reset_) {
86 return cb(Error::IS_RESET);
87 }
88 if (!is_writable_) {
89 return cb(Error::IS_CLOSED_FOR_WRITES);
90 }
91 if (bytes == 0 || in.empty() || static_cast<size_t>(in.size()) < bytes) {
92 return cb(Error::INVALID_ARGUMENT);
93 }
94
95 if (boost::asio::buffer_copy(buffer_.prepare(bytes),
96 boost::asio::const_buffer(in.data(), bytes))
97 != bytes) {
98 return cb(Error::INTERNAL_ERROR);
99 }
100 buffer_.commit(bytes);
101
102 cb(outcome::success(bytes));
103
104 if (data_notifyee_) {
105 data_notifyee_(buffer_.size());
106 }
107 }

115 void LoopbackStream::read(gsl::span<uint8_t> out,
116 size_t bytes,
117 libp2p::basic::Reader::ReadCallbackFunc cb,
118 bool some) {
119 if (is_reset_) {
120 return cb(Error::IS_RESET);
121 }
122 if (!is_readable_) {
123 return cb(Error::IS_CLOSED_FOR_READS);
124 }
125 if (bytes == 0 || out.empty() || static_cast<size_t>(out.size()) < bytes) {
126 return cb(Error::INVALID_ARGUMENT);
127 }
128
129 // this lambda checks, if there's enough data in our read buffer, and gives
130 // it to the caller, if so
131 auto read_lambda = [self{shared_from_this()},
132 cb{std::move(cb)},
133 out,
134 bytes,
135 some](outcome::result<size_t> res) mutable {
136 if (!res) {
137 self->data_notified_ = true;
138 cb(res);
139 return;
140 }
141 if (self->buffer_.size() >= (some ? 1 : bytes)) {
142 auto to_read = some ? std::min(self->buffer_.size(), bytes) : bytes;
143 if (boost::asio::buffer_copy(boost::asio::buffer(out.data(), to_read),
144 self->buffer_.data(),
145 to_read)
146 != to_read) {
147 return cb(Error::INTERNAL_ERROR);
148 }
149
150 self->buffer_.consume(to_read);
151 self->data_notified_ = true;
152 cb(to_read);
153 }
154 };
155
156 // return immediately, if there's enough data in the buffer
157 data_notified_ = false;
158 read_lambda(0);
159 if (data_notified_) {
160 return;
161 }
162
163 // subscribe to new data updates
164 data_notifyee_ = std::move(read_lambda);
165 };
