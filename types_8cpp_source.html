39 base::Buffer trieRoot(
40 const std::vector<std::pair<base::Buffer, base::Buffer>> &key_vals) {
41 auto trie = storage::trie::SuperGeniusTrieImpl();
42 auto codec = storage::trie::SuperGeniusCodec();
43
44 for (const auto &[key, val] : key_vals) {
45 auto res = trie.put(key, val);
46 BOOST_ASSERT_MSG(res.has_value(), "Insertion into trie failed");
47 }
48 auto root = trie.getRoot();

50 return base::Buffer{codec.hash256({0})};
51 }
52 auto encode_res = codec.encodeNode(*root);
53 BOOST_ASSERT_MSG(encode_res.has_value(), "Trie encoding failed");
