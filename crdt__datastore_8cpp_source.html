164 void CrdtDatastore::HandleNext()
165 {
166
167 if (broadcaster_ == nullptr)
168 {
169 // offline
170 return;
171 }
172
173 handleNextThreadRunning_ = true;
174
175 LogDebug("HandleNext thread started");
176 while (handleNextThreadRunning_)
177 {
178 std::this_thread::sleep_for(threadSleepTimeInMilliseconds_);
179
180 auto broadcasterNextResult = broadcaster_->Next();
181 if (broadcasterNextResult.has_failure())
182 {
183 if (broadcasterNextResult.error().value() != (int)Broadcaster::ErrorCode::ErrNoMoreBroadcast)
184 {
185 LogDebug("Failed to get next broadcaster (error code " +
186 std::to_string(broadcasterNextResult.error().value()) + ")");
187 }
188 continue;
189 }
190
191 auto decodeResult = DecodeBroadcast(broadcasterNextResult.value());
192 if (decodeResult.has_failure())
193 {
194 LogError("Broadcaster: Unable to decode broadcast (error code " +
195 std::to_string(broadcasterNextResult.error().value()) + ")");
196 continue;
197 }
198
199 // For each head, we process it.
200 for (const auto& bCastHeadCID : decodeResult.value())
201 {
202 auto handleBlockResult = HandleBlock(bCastHeadCID);
203 if (handleBlockResult.has_failure())
204 {
205 LogError("Broadcaster: Unable to handle block (error code " +
206 std::to_string(handleBlockResult.error().value()) + ")");
207 continue;
208 }
209 std::unique_lock lock(seenHeadsMutex_);
210 seenHeads_.push_back(bCastHeadCID);
211 }
212
213 // We should store trusted-peer signatures associated to
214 // each head in a timecache. When we broadcast, attach the
215 // signatures (along with our own) to the broadcast.
216 // Other peers can use the signatures to verify that the
217 // received CIDs have been issued by a trusted peer.
218 }
219
220 LogDebug("HandleNext thread finished");
221 }

386 outcome::result<void> CrdtDatastore::HandleBlock(const CID& aCid)
387 {
388 // Ignore already known blocks.
389 // This includes the case when the block is a current
390 // head.
391
392 if (this->dagSyncer_ == nullptr)
393 {
394 return outcome::failure(boost::system::error_code{});
395 }
396 auto dagSyncerResult = this->dagSyncer_->HasBlock(aCid);
397 if (dagSyncerResult.has_failure())
398 {
399 LOG_ERROR("HandleBlock: error checking for known block");
400 return outcome::failure(dagSyncerResult.error());
401 }
402
403 if (dagSyncerResult.value())
404 {
405 // cid is known. Skip walking tree
406 return outcome::success();
407 }
408
409 std::vector<CID> children;
410 children.push_back(aCid);
411 SendNewJobs(aCid, 0, children);
412
413 return outcome::success();
414 }

416 void CrdtDatastore::SendNewJobs(const CID& aRootCID, const uint64_t& aRootPriority, const std::vector<CID>& aChildren)
417 {
418 // sendNewJobs calls getDeltas with the given
419 // children and sends each response to the workers.
420
421 if (this->dagSyncer_ == nullptr || aChildren.empty())
422 {
423 return;
424 }
425
426 // @todo figure out how should dagSyncerTimeoutSec be used
427 std::chrono::seconds dagSyncerTimeoutSec = std::chrono::seconds(5 * 60); // 5 mins by default
428 if (this->options_ != nullptr)
429 {
430 dagSyncerTimeoutSec = std::chrono::seconds(options_->dagSyncerTimeoutSec);
431 }
432
433 uint64_t rootPriority = aRootPriority;
434 if (rootPriority == 0)
435 {
436 std::shared_lock lock(dagWorkerMutex_);
437 auto getNodeResult = dagSyncer_->getNode(aChildren[0]);
438 if (getNodeResult.has_failure())
439 {
440 return;
441 }
442 auto node = getNodeResult.value();
443 auto nodeBuffer = node->content();
444
445 auto delta = std::make_shared<Delta>();
446
447 if (!delta->ParseFromArray(nodeBuffer.data(), nodeBuffer.size()))
448 {
449 return;
450 }
451
452 if (rootPriority == 0)
453 {
454 rootPriority = delta->priority();
455 }
456 }
457
458 for (const auto& cid : aChildren)
459 {
460 //Fetch only root node with all children, but without children of their children
461 dagWorkerMutex_.lock_shared();
462 auto graphResult = dagSyncer_->fetchGraphOnDepth(cid, 1);
463 dagWorkerMutex_.unlock_shared();
464 if (graphResult.has_failure())
465 {
466 LOG_ERROR("SendNewJobs: error fetching graph for CID:" << cid.toString().value());
467 continue;
468 }
469
470 auto leaf = graphResult.value();
471 auto nodeBuffer = leaf->content();
472
473 // @todo Check if it is OK that the node has only content and doesn't have links
474 auto nodeResult = ipfs_lite::ipld::IPLDNodeImpl::createFromRawBytes(nodeBuffer);
475 if (nodeResult.has_failure())
476 {
477 continue;
478 }
479 auto node = nodeResult.value();
480
481 auto delta = std::make_shared<Delta>();
482 if (!delta->ParseFromArray(nodeBuffer.data(), nodeBuffer.size()))
483 {
484 continue;
485 }
486
487 DagJob dagJob;
488 dagJob.rootCid_ = aRootCID;
489 dagJob.rootPriority_ = rootPriority;
490 dagJob.delta_ = delta;
491 dagJob.node_ = node;
492 {
493 std::unique_lock lock(dagWorkerMutex_);
494 dagWorkerJobList.push(dagJob);
495 }
496 }

740 outcome::result<std::vector<CID>> CrdtDatastore::ProcessNode(const CID& aRoot, const uint64_t& aRootPrio,
741 const std::shared_ptr<Delta>& aDelta, const std::shared_ptr<Node>& aNode)
742 {
743 if (this->set_ == nullptr || this->heads_ == nullptr || this->dagSyncer_ == nullptr ||
744 aDelta == nullptr || aNode == nullptr)
745 {
746 return outcome::failure(boost::system::error_code{});
747 }
748
749 // merge the delta
750 auto current = aNode->getCID();
751 auto strCidResult = current.toString();
752 if (strCidResult.has_failure())
753 {
754 return outcome::failure(strCidResult.error());
755 }
756 HierarchicalKey hKey(strCidResult.value());
757
758 {
759 std::shared_lock lock(this->dagWorkerMutex_);
760 auto mergeResult = this->set_->Merge(aDelta, hKey.GetKey());
761 if (mergeResult.has_failure())
762 {
763 LOG_ERROR("ProcessNode: error merging delta from " << hKey.GetKey());
764 return outcome::failure(mergeResult.error());
765 }
766 }
767
768 auto priority = aDelta->priority();
769 if (priority % 10 == 0)
770 {
771 LOG_INFO("ProcessNode: merged delta from " << strCidResult.value() << " (priority: " << priority << ")");
772 }
773
774 std::vector<CID> children;
775 auto links = aNode->getLinks();
776 if (links.empty())
777 {
778 // we reached the bottom, we are a leaf.
779 auto addHeadResult = this->heads_->Add(aRoot, aRootPrio);
780 if (addHeadResult.has_failure())
781 {
782 LOG_ERROR("ProcessNode: error adding head " << aRoot.toString().value());
783 return outcome::failure(addHeadResult.error());
784 }
785 return outcome::success(children);
786 }
787
788 // walkToChildren
789 for (const auto& link : links)
790 {
791 auto child = link.get().getCID();
792 auto isHeadResult = this->heads_->IsHead(child);
793 if (isHeadResult.has_failure())
794 {
795 LOG_ERROR("ProcessNode: error checking if " << child.toString().value() << " is head");
796 return outcome::failure(isHeadResult.error());
797 }
798
799 if (isHeadResult.value() == true)
800 {
801 // reached one of the current heads.Replace it with
802 // the tip of this branch
803 auto replaceResult = this->heads_->Replace(child, aRoot, aRootPrio);
804 if (replaceResult.has_failure())
805 {
806 LOG_ERROR("ProcessNode: error replacing head " << child.toString().value() << " -> " << aRoot.toString().value());
807 return outcome::failure(replaceResult.error());
808 }
809
810 continue;
811 }
812
813 std::shared_lock lock(this->dagWorkerMutex_);
814 auto knowBlockResult = this->dagSyncer_->HasBlock(child);
815 if (knowBlockResult.has_failure())
816 {
817 LOG_ERROR("ProcessNode: error checking for known block " << child.toString().value());
818 return outcome::failure(knowBlockResult.error());
819 }
820
821 if (knowBlockResult.value() == true)
822 {
823 // we reached a non-head node in the known tree.
824 // This means our root block is a new head.
825 auto addHeadResult = this->heads_->Add(aRoot, aRootPrio);
826 if (addHeadResult.has_failure())
827 {
828 // Don't let this failure prevent us from processing the other links.
829 LOG_ERROR("ProcessNode: error adding head " << aRoot.toString().value());
830 }
831 continue;
832 }
833
834 children.push_back(child);
835 }
836
837 return children;
838 }

842 {
843 if (this->heads_ == nullptr)
844 {
845 return outcome::failure(boost::system::error_code{});
846 }
847
848 uint64_t height = 0;
849 std::vector<CID> heads;
850 auto getListResult = this->heads_->GetList(heads, height);
851 if (getListResult.has_failure())
852 {
853 return outcome::failure(getListResult.error());
854 }
855
856 height = height + 1; // This implies our minimum height is 1

858

860 if (putBlockResult.has_failure())
861 {
862 return outcome::failure(putBlockResult.error());
863 }
864
865 auto node = putBlockResult.value();
866
867 // Process new block. This makes that every operation applied
868 // to this store take effect (delta is merged) before
869 // returning. Since our block references current heads, children
870 // should be empty
871 LOG_INFO("AddDAGNode: Processing generated block " << node->getCID().toString().value());
872
873 auto processNodeResult = this->ProcessNode(node->getCID(), height, aDelta, node);
874 if (processNodeResult.has_failure())
875 {
876 LOG_ERROR("AddDAGNode: error processing new block");
877 return outcome::failure(processNodeResult.error());
878 }
879
880 if (!processNodeResult.value().empty())
881 {
882 LOG_ERROR("AddDAGNode: bug - created a block to unknown children");
883 }
884
885 return node->getCID();

984 outcome::result<void> CrdtDatastore::Sync(const HierarchicalKey& aKey)
985 {
986 // This is a quick write up of the internals from the time when
987 // I was thinking many underlying datastore entries are affected when
988 // an add operation happens:
989 //
990 // When a key is added:
991 // - a new delta is made
992 // - Delta is marshalled and a DAG-node is created with the bytes,
993 // pointing to previous heads. DAG-node is added to DAGService.
994 // - Heads are replaced with new CID.
995 // - New CID is broadcasted to everyone
996 // - The new CID is processed (up until now the delta had not
997 // taken effect). Implementation detail: it is processed before
998 // broadcast actually.
999 // - processNode() starts processing that branch from that CID
1000 // - it calls set.Merge()
1001 // - that calls putElems() and putTombs()
1002 // - that may make a batch for all the elems which is later committed
1003 // - each element has a datastore entry /setNamespace/elemsNamespace/<key>/<block_id>
1004 // - each tomb has a datastore entry /setNamespace/tombsNamespace/<key>/<block_id>
1005 // - each value has a datastore entry /setNamespace/keysNamespace/<key>/valueSuffix
1006 // - each value has an additional priority entry /setNamespace/keysNamespace/<key>/prioritySuffix
1007 // - the last two are only written if the added entry has more priority than any the existing
1008 // - For a value to not be lost, those entries should be fully synced.
1009 // - In order to check if a value is in the set:
1010 // - List all elements on /setNamespace/elemsNamespace/<key> (will return several block_ids)
1011 // - If we find an element which is not tombstoned, then value is in the set
1012 // - In order to retrieve an element's value:
1013 // - Check that it is in the set
1014 // - Read the value entry from the /setNamespace/keysNamespace/<key>/valueSuffix path
1015
1016 // Be safe and just sync everything in our namespace
1017 if (aKey.GetKey() == "/")
1018 {
1019 return this->Sync(this->namespaceKey_);
1020 }
1021
1022 // attempt to be intelligent and sync only all heads and the
1023 // set entries related to the given prefix.
1024 std::vector<HierarchicalKey> keysToSync;
1025 keysToSync.push_back(this->set_->ElemsPrefix(aKey.GetKey()));
1026 keysToSync.push_back(this->set_->TombsPrefix(aKey.GetKey()));
1027 keysToSync.push_back(this->set_->KeysKey(aKey.GetKey())); // covers values and priorities
1028
1029 keysToSync.push_back(this->heads_->GetNamespaceKey());
1030 return this->SyncDatastore(keysToSync);
1031 }
