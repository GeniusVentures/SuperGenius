30 inline uint8_t byteFromNibbles(uint8_t high, uint8_t low) {
31 // get low4 from nibbles to avoid check: if(low > 0xff) return error
32 return (high << 4u) | (low & 0xfu);
33 }

35 inline base::Buffer ushortToBytes(uint16_t b) {
36 base::Buffer out(2, 0);
37 out[1] = (b >> 8u) & 0xffu;
38 out[0] = b & 0xffu;
39 return out;
40 }

42 base::Buffer SuperGeniusCodec::nibblesToKey(const KeyNibbles &nibbles) {
43 Buffer res;
44 if (nibbles.size() % 2 == 0) {
45 res = Buffer(nibbles.size() / 2, 0);
46 for (size_t i = 0; i < nibbles.size(); i += 2) {
47 res[i / 2] = byteFromNibbles(nibbles[i], nibbles[i + 1]);
48 }
49 } else {
50 res = Buffer(nibbles.size() / 2 + 1, 0);
51 res[0] = nibbles[0];
52 for (size_t i = 2; i < nibbles.size(); i += 2) {
53 res[i / 2] = byteFromNibbles(nibbles[i - 1], nibbles[i]);
54 }
55 }
56 return res;
57 }

59 KeyNibbles SuperGeniusCodec::keyToNibbles(const base::Buffer &key) {
60 if (key.empty()) {
61 return {};
62 }
63 if (key.size() == 1 && key[0] == 0) {
64 return KeyNibbles{base::Buffer{0, 0}};
65 }
66
67 auto l = key.size() * 2;
68 KeyNibbles res(base::Buffer(l, 0));
69 for (size_t i = 0; i < key.size(); i++) {
70 res[2 * i] = key[i] >> 4u;
71 res[2 * i + 1] = key[i] & 0xfu;
72 }
73
74 return res;
75 }

115 outcome::result<base::Buffer> SuperGeniusCodec::encodeHeader(
116 const SuperGeniusNode &node) const {
117 if (node.key_nibbles.size() > 0xffffu) {
118 return Error::TOO_MANY_NIBBLES;
119 }
120
121 uint8_t head = 0;
122 // set bits 6..7
123 switch (static_cast<SuperGeniusNode::Type>(node.getType())) {
124 case SuperGeniusNode::Type::BranchEmptyValue:
125 case SuperGeniusNode::Type::BranchWithValue:
126 case SuperGeniusNode::Type::Leaf:
127 head = node.getType();
128 break;
129 default:
130 return Error::UNKNOWN_NODE_TYPE;
131 }
132 // the first two bits are a node type
133 head <<= 6u; // head_{6..7} * 64
134
135 // set bits 0..5, partial key length
136 if (node.key_nibbles.size() < 63u) {
137 head |= uint8_t(node.key_nibbles.size());
138 return Buffer{head}; // header contains 1 byte
139 }
140 // if partial key length is greater than 62, then the rest of the length is
141 // stored in consequent bytes
142 head += 63u;
143
144 size_t l = node.key_nibbles.size() - 63u;
145 Buffer out(1u +
146 l / 0xffu +
147 1u,
148 0xffu
149 );
150
151 // first byte is head
152 out[0] = head;
153 // last byte after while(l>255) l-=255;
154 out[out.size() - 1] = l % 0xffu;
155
156 return out;
157 }

159 outcome::result<base::Buffer> SuperGeniusCodec::encodeBranch(
160 const BranchNode &node) const {
161 // node header
162 OUTCOME_TRY((auto &&, encoding), encodeHeader(node));
163
164 // key
165 encoding += nibblesToKey(node.key_nibbles);
166
167 // children bitmap
168 encoding += ushortToBytes(node.childrenBitmap());
169
170 if (node.getTrieType() == SuperGeniusNode::Type::BranchWithValue) {
171 // scale encoded value
172 OUTCOME_TRY((auto &&, encNodeValue), scale::encode(node.value.get()));
173 encoding += Buffer(std::move(encNodeValue));
174 }
175
176 // encode each child
177 for (auto &child : node.children) {
178 if (child) {
179 if (child->isDummy()) {
180 auto merkle_value =
181 std::dynamic_pointer_cast<DummyNode>(child)->db_key;
182 OUTCOME_TRY((auto &&, scale_enc), scale::encode(std::move(merkle_value)));
183 encoding.put(scale_enc);
184 } else {
185 OUTCOME_TRY((auto &&, enc), encodeNode(*child));
186 OUTCOME_TRY((auto &&, scale_enc), scale::encode(merkleValue(enc)));
187 encoding.put(scale_enc);
188 }
189 }
190 }
191
192 return outcome::success(std::move(encoding));
193 }

196 const LeafNode &node) const {

198

205 encoding += Buffer(std::move(encNodeValue));
206
207 return outcome::success(std::move(encoding));

210 outcome::result<std::shared_ptr<Node>> SuperGeniusCodec::decodeNode(
211 const base::Buffer &encoded_data) const {
212 BufferStream stream{encoded_data};
213 // decode the header with the node type and the partial key length
214 OUTCOME_TRY((auto &&, header), decodeHeader(stream));
215 auto [type, pk_length] = header;
216 // decode the partial key
217 OUTCOME_TRY((auto &&, partial_key), decodePartialKey(pk_length, stream));
218 // decode the node subvalue (see Definition 28 of the supergenius
219 // specification)
220 switch (type) {
221 case SuperGeniusNode::Type::Leaf: {
222 OUTCOME_TRY((auto &&, value), scale::decode<Buffer>(stream.leftBytes()));
223 return std::make_shared<LeafNode>(partial_key, value);
224 }
225 case SuperGeniusNode::Type::BranchEmptyValue:
226 case SuperGeniusNode::Type::BranchWithValue: {
227 return decodeBranch(type, partial_key, stream);
228 }
229 default:
230 return Error::UNKNOWN_NODE_TYPE;
231 }
232 }

234 outcome::result<std::pair<SuperGeniusNode::Type, size_t>>
235 SuperGeniusCodec::decodeHeader(BufferStream &stream) const {
236 SuperGeniusNode::Type type;
237 size_t pk_length = 0;
238 if (!stream.hasMore(1)) {
239 return Error::INPUT_TOO_SMALL;
240 }
241 auto first = stream.next();
242 // decode type, which is in the first two bits
243 type = static_cast<SuperGeniusNode::Type>((first & 0xC0u) >> 6u);
244 first &= 0x3Fu;
245 // decode partial key length, which is stored in the last 6 bits and, if its
246 // greater than 62, in the following bytes
247 pk_length += first;
248
249 if (pk_length == 63) {
250 uint8_t read_length = 0;
251 do {
252 if (!stream.hasMore(1)) {
253 return Error::INPUT_TOO_SMALL;
254 }
255 read_length = stream.next();
256 pk_length += read_length;
257 } while (read_length == 0xFF);
258 }
259 return std::make_pair(type, pk_length);
260 }

262 outcome::result<KeyNibbles> SuperGeniusCodec::decodePartialKey(
263 size_t nibbles_num, BufferStream &stream) const {
264 // length in bytes is length in nibbles over two round up
265 auto byte_length = nibbles_num / 2 + nibbles_num % 2;
266 Buffer partial_key;
267 partial_key.reserve(byte_length);
268 while (byte_length-- != 0) {
269 if (!stream.hasMore(1)) {
270 return Error::INPUT_TOO_SMALL;
271 }
272 partial_key.putUint8(stream.next());
273 }
274 // array of nibbles is much more convenient than array of bytes, though it
275 // wastes some memory
276 KeyNibbles partial_key_nibbles = keyToNibbles(partial_key);
277 if (nibbles_num % 2 == 1) {
278 partial_key_nibbles = partial_key_nibbles.subbuffer(1);
279 }
280 return partial_key_nibbles;
281 }

283 outcome::result<std::shared_ptr<Node>> SuperGeniusCodec::decodeBranch(
284 SuperGeniusNode::Type type,
285 const KeyNibbles &partial_key,
286 BufferStream &stream) const {
287 constexpr uint8_t kChildrenBitmapSize = 2;
288
289 if (!stream.hasMore(kChildrenBitmapSize)) {
290 return Error::INPUT_TOO_SMALL;
291 }
292 auto node = std::make_shared<BranchNode>(partial_key);
293
294 uint16_t children_bitmap = stream.next();
295 children_bitmap += stream.next() << 8u;
296
297 scale::ScaleDecoderStream ss(stream.leftBytes());
298
299 // decode the branch value if needed
300 base::Buffer value;
301 if (type == SuperGeniusNode::Type::BranchWithValue) {
302 try {
303 ss >> value;
304 } catch (std::system_error &e) {
305 return outcome::failure(e.code());
306 }
307 node->value = value;
308 }
309
310 uint8_t i = 0;
311 while (children_bitmap != 0) {
312 // if there is a child
313 if ((children_bitmap & (1u << i)) != 0) {
314 // unset bit for this child
315 children_bitmap &= ~(1u << i);
316 // read the hash of the child and make a dummy node from it for this
317 // child in the processed branch
318 base::Buffer child_hash;
319 try {
320 ss >> child_hash;
321 } catch (std::system_error &e) {
322 return outcome::failure(e.code());
323 }
324 node->children.at(i) = std::make_shared<DummyNode>(child_hash);
325 }
326 i++;
327 }
328 return node;
329 }
