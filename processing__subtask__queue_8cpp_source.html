90bool ProcessingSubTaskQueue::RollbackOwnership()
91{
92 if (HasOwnership())
93 {
94 // Do nothing. The node is already the queue owner
95 return true;
96 }
97
98 // Find the current queue owner
99 auto ownerNodeId = m_queue->owner_node_id();
100 int ownerNodeIdx = -1;
101 for (int itemIdx = 0; itemIdx < (int)m_queue->items_size(); ++itemIdx)
102 {
103 if (m_queue->items(itemIdx).lock_node_id() == ownerNodeId)
104 {
105 ownerNodeIdx = itemIdx;
106 break;
107 }
108 }
109
110 if (ownerNodeIdx >= 0)
111 {
112 // Check if the local node is the previous queue owner
113 for (int idx = 1; idx < m_queue->items_size(); ++idx)
114 {
115 // Loop cyclically over queue items in backward direction starting from item[ownerNodeIdx - 1]
116 // and excluding the item[ownerNodeIdx]
117 int itemIdx = (m_queue->items_size() + ownerNodeIdx - idx) % m_queue->items_size();
118 if (!m_queue->items(itemIdx).lock_node_id().empty())
119 {
120 if (m_queue->items(itemIdx).lock_node_id() == m_localNodeId)
121 {
122 // The local node is the previous queue owner
123 ChangeOwnershipTo(m_localNodeId);
124 return true;
125 }
126 else
127 {
128 // Another node should take the ownership
129 return false;
130 }
131 }
132 }
133 }
134 else
135 {
136 // The queue owner didn't lock any subtask
137 // Find the last locked item
138 for (int idx = 1; idx <= m_queue->items_size(); ++idx)
139 {
140 int itemIdx = (m_queue->items_size() - idx);
141 if (!m_queue->items(itemIdx).lock_node_id().empty())
142 {
143 if (m_queue->items(itemIdx).lock_node_id() == m_localNodeId)
144 {
145 // The local node is the last locked item
146 ChangeOwnershipTo(m_localNodeId);
147 return true;
148 }
149 else
150 {
151 // Another node should take the ownership
152 return false;
153 }
154 }
155 }
156 }
157
158 // No locked items found
159 // Allow the local node to take the ownership
160 ChangeOwnershipTo(m_localNodeId);
161 return true;
162}
