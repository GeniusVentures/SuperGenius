89 runtime::WasmSize CryptoExtension::ext_ed25519_verify(
90 runtime::WasmPointer msg_data,
91 runtime::WasmSize msg_len,
92 runtime::WasmPointer sig_data,
93 runtime::WasmPointer pubkey_data) {
94 // for some reason, 0 and 5 are used in the reference implementation, so
95 // it's better to stick to them in ours, at least for now
96 static constexpr uint32_t kVerifySuccess = 0;
97 static constexpr uint32_t kVerifyFail = 5;
98
99 auto msg = memory_->loadN(msg_data, msg_len);
100 auto sig_bytes =
101 memory_->loadN(sig_data, ed25519_constants::SIGNATURE_SIZE).toVector();
102
103 auto signature_res = crypto::ED25519Signature::fromSpan(sig_bytes);
104 if (!signature_res) {
105 BOOST_UNREACHABLE_RETURN(kVerifyFail);
106 }
107 auto &&signature = signature_res.value();
108
109 auto pubkey_bytes =
110 memory_->loadN(pubkey_data, ed25519_constants::PUBKEY_SIZE).toVector();
111 auto pubkey_res = crypto::ED25519PublicKey::fromSpan(pubkey_bytes);
112 if (!pubkey_res) {
113 BOOST_UNREACHABLE_RETURN(kVerifyFail);
114 }
115 auto pubkey = pubkey_res.value();
116
117 auto result = ed25519_provider_->verify(signature, msg, pubkey);
118 auto is_succeeded = result && result.value();
119
120 return is_succeeded ? kVerifySuccess : kVerifyFail;
121 }

123 runtime::WasmSize CryptoExtension::ext_sr25519_verify(
124 runtime::WasmPointer msg_data,
125 runtime::WasmSize msg_len,
126 runtime::WasmPointer sig_data,
127 runtime::WasmPointer pubkey_data) {
128 // for some reason, 0 and 5 are used in the reference implementation, so
129 // it's better to stick to them in ours, at least for now
130 static constexpr uint32_t kVerifySuccess = 0;
131 static constexpr uint32_t kVerifyFail = 5;
132
133 auto msg = memory_->loadN(msg_data, msg_len);
134 auto signature_buffer =
135 memory_->loadN(sig_data, sr25519_constants::SIGNATURE_SIZE);
136
137 auto pubkey_buffer =
138 memory_->loadN(pubkey_data, sr25519_constants::PUBLIC_SIZE);
139 auto key_res = crypto::SR25519PublicKey::fromSpan(pubkey_buffer);
140 if (!key_res) {
141 BOOST_UNREACHABLE_RETURN(kVerifyFail);
142 }
143 auto &&key = key_res.value();
144
145 crypto::SR25519Signature signature{};
146 std::copy_n(signature_buffer.begin(),
147 sr25519_constants::SIGNATURE_SIZE,
148 signature.begin());
149
150 auto res = sr25519_provider_->verify(signature, msg, key);
151 bool is_succeeded = res && res.value();
152
153 return is_succeeded ? kVerifySuccess : kVerifyFail;
154 }

276 base::Blob<32> CryptoExtension::deriveSeed(std::string_view content) {
277 // first check if content is a hexified seed value
278 if (auto res = base::Blob<32>::fromHexWithPrefix(content); res) {
279 return res.value();
280 }
281
282 logger_->debug("failed to unhex seed, try parse mnemonic");
283
284 // now check if it is a bip39 mnemonic phrase with optional password
285 auto mnemonic = crypto::bip39::Mnemonic::parse(content);
286 if (!mnemonic) {
287 logger_->error("failed to parse mnemonic {}", mnemonic.error().message());
288 std::terminate();
289 }
290
291 auto &&entropy = bip39_provider_->calculateEntropy(mnemonic.value().words);
292 if (!entropy) {
293 logger_->error("failed to calculate entropy {}",
294 entropy.error().message());
295 std::terminate();
296 }
297
298 auto &&big_seed =
299 bip39_provider_->makeSeed(entropy.value(), mnemonic.value().password);
300 if (!big_seed) {
301 logger_->error("failed to generate seed {}", big_seed.error().message());
302 std::terminate();
303 }
304
305 auto big_span = gsl::span<uint8_t>(big_seed.value());
306 constexpr size_t size = base::Blob<32>::size();
307 // get first 32 bytes from big seed as ed25519 or sr25519 seed
308 auto seed = base::Blob<32>::fromSpan(big_span.subspan(0, size));
309 if (!seed) {
310 // impossible case bip39 seed is always 64 bytes long
311 BOOST_UNREACHABLE_RETURN({});
312 }
313 return seed.value();
314 }

319 runtime::WasmPointer CryptoExtension::ext_ed25519_generate_v1(
320 runtime::WasmSize key_type, runtime::WasmSpan seed) {
321 auto key_type_id = static_cast<crypto::KeyTypeId>(key_type);
322 if (!crypto::isSupportedKeyType(key_type_id)) {
323 auto kt = crypto::decodeKeyTypeId(key_type_id);
324 logger_->warn("key type '{}' is not officially supported", kt);
325 }
326
327 auto [seed_ptr, seed_len] = runtime::WasmResult(seed);
328 auto seed_buffer = memory_->loadN(seed_ptr, seed_len);
329 auto seed_res = scale::decode<boost::optional<std::string>>(seed_buffer);
330 if (!seed_res) {
331 logger_->error("failed to decode seed");
332 std::terminate();
333 }
334
335 crypto::ED25519Keypair kp{};
336 boost::optional<std::string> bip39_seed = seed_res.value();
337 if (bip39_seed.has_value()) {
338 auto ed_seed = deriveSeed(*bip39_seed);
339 kp = ed25519_provider_->generateKeypair(ed_seed);
340 } else {
341 auto key_pair = ed25519_provider_->generateKeypair();
342 if (!key_pair) {
343 logger_->error("failed to generate ed25519 key pair: {}",
344 key_pair.error().message());
345 std::terminate();
346 }
347 kp = key_pair.value();
348 }
349
350 runtime::WasmSpan ps = memory_->storeBuffer(kp.public_key);
351
352 return runtime::WasmResult(ps).address;
353 }

358 runtime::WasmSpan CryptoExtension::ext_ed25519_sign_v1(
359 runtime::WasmSize key_type,
360 runtime::WasmPointer key,
361 runtime::WasmSpan msg) {
362 using ResultType = boost::optional<crypto::ED25519Signature>;
363
364 auto key_type_id = static_cast<crypto::KeyTypeId>(key_type);
365 if (!crypto::isSupportedKeyType(key_type_id)) {
366 logger_->warn("key type '{}' is not supported",
367 decodeKeyTypeId(key_type_id));
368 }
369
370 auto public_buffer = memory_->loadN(key, crypto::ED25519PublicKey::size());
371 auto [msg_data, msg_len] = runtime::WasmResult(msg);
372 auto msg_buffer = memory_->loadN(msg_data, msg_len);
373 auto pk = crypto::ED25519PublicKey::fromSpan(public_buffer);
374 if (!pk) {
375 BOOST_UNREACHABLE_RETURN({});
376 }
377 auto key_pair = crypto_store_->findEd25519Keypair(key_type_id, pk.value());
378 if (!key_pair) {
379 logger_->error("failed to find required key");
380 auto error_result = scale::encode(ResultType(boost::none)).value();
381 return memory_->storeBuffer(error_result);
382 }
383
384 auto sign = ed25519_provider_->sign(key_pair.value(), msg_buffer);
385 if (!sign) {
386 logger_->error("failed to sign message, error = {}",
387 sign.error().message());
388 std::terminate();
389 }
390
391 auto buffer = scale::encode(ResultType(sign.value())).value();
392 return memory_->storeBuffer(buffer);

536 runtime::WasmSpan CryptoExtension::ext_crypto_secp256k1_ecdsa_recover_v1(
537 runtime::WasmPointer sig, runtime::WasmPointer msg) {
538 using ResultType = boost::variant<ecdsa::PublicKey, EcdsaVerifyError>;
539
540 constexpr auto signature_size = RSVSignature::size();
541 constexpr auto message_size = MessageHash::size();
542
543 auto sig_buffer = memory_->loadN(sig, signature_size);
544 auto msg_buffer = memory_->loadN(msg, message_size);
545
546 auto signature = RSVSignature::fromSpan(sig_buffer).value();
547 auto message = MessageHash::fromSpan(msg_buffer).value();
548
549 auto public_key =
550 secp256k1_provider_->recoverPublickeyUncompressed(signature, message);
551 if (!public_key) {
552 logger_->error("failed to recover uncompressed secp256k1 public key: {}",
553 public_key.error().message());
554
555 auto error_code =
556 convertFailureToError<UncompressedPublicKey>(public_key.as_failure());
557 auto error_result =
558 scale::encode(static_cast<ResultType>(error_code)).value();
559
560 return memory_->storeBuffer(error_result);
561 }
562
563 // according to substrate implementation
564 // returned key shouldn't include the 0x04 prefix
565 // specification says, that it should have 64 bytes, not 65 as with prefix
566 // On success it contains the 64-byte recovered public key or an error type
567 auto truncated_span = gsl::span<uint8_t>(public_key.value()).subspan(1, 64);
568 auto truncated_public_key =
569 ecdsa::PublicKey::fromSpan(truncated_span).value();
570 auto buffer = scale::encode(ResultType(truncated_public_key)).value();
571 return memory_->storeBuffer(buffer);
572 }
