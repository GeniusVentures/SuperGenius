36 outcome::result<void> TransactionPoolImpl::submitOne(
37 const std::shared_ptr<Transaction> &tx) {
38 if (auto [_, ok] = imported_txs_.emplace(tx->hash, tx); !ok) {
39 return TransactionPoolError::TX_ALREADY_IMPORTED;
40 }
41
42 auto processResult = processTransaction(tx);
43 if (processResult.has_error()
44 && processResult.error() == TransactionPoolError::POOL_IS_FULL) {
45 imported_txs_.erase(tx->hash);
46 } else {
47 logger_->debug("Extrinsic {} with hash {} was added to the pool",
48 tx->ext.data.toHex(),
49 tx->hash.toHex());
50 }
51
52 return processResult;
53 }

141 void TransactionPoolImpl::processPostponedTransactions() {
142 // Move to local for avoid endless cycle at possible coming back tx
143 auto postponed_txs = std::move(postponed_txs_);
144 while (!postponed_txs.empty()) {
145 auto tx = postponed_txs.front().lock();
146 postponed_txs.pop_front();
147
148 auto result = processTransaction(tx);
149 if (result.has_error()
150 && result.error() == TransactionPoolError::POOL_IS_FULL) {
151 postponed_txs_.insert(postponed_txs_.end(),
152 std::make_move_iterator(postponed_txs.begin()),
153 std::make_move_iterator(postponed_txs.end()));
154 return;
155 }
156 }
157 }

226 void TransactionPoolImpl::commitRequiredTags(
227 const std::shared_ptr<Transaction> &tx) {
228 for (auto &tag : tx->requires) {
229 auto range = tx_waits_tag_.equal_range(tag);
230 for (auto i = range.first; i != range.second;) {
231 auto ci = i++;
232 if (ci->second.lock() == tx) {
233 auto node = tx_waits_tag_.extract(ci);
234 tx_depends_on_tag_.emplace(std::move(node.key()),

293 void TransactionPoolImpl::unprovideTag(const Transaction::Tag &tag) {
294 if (tx_provides_tag_.find(tag) == tx_provides_tag_.end()) {
295 for (auto it = tx_depends_on_tag_.find(tag);
296 it != tx_depends_on_tag_.end();
297 it = tx_depends_on_tag_.find(tag)) {
298 if (auto tx = it->second.lock()) {
299 unsetReady(tx);
300 }
301 tx_depends_on_tag_.erase(it);
