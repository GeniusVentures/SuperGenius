186outcome::result<std::string> GlobalDB::KeyToString(const Buffer& key) const
187{
188 // @todo cache the prefix and suffix
189 auto keysPrefix = m_crdtDatastore->GetKeysPrefix();
190 if (!keysPrefix.has_value())
191 {
192 return outcome::failure(boost::system::error_code{});
193 }
194 auto valueSuffix = m_crdtDatastore->GetValueSuffix();
195 if (!valueSuffix.has_value())
196 {
197 return outcome::failure(boost::system::error_code{});
198 }
199
200 auto sKey = std::string(key.toString());
201
202 size_t prefixPos = (keysPrefix.value().size() != 0) ? sKey.find(keysPrefix.value(), 0) : 0;
203 if (prefixPos != 0)
204 {
205 return outcome::failure(boost::system::error_code{});
206 }
207
208 size_t keyPos = keysPrefix.value().size();
209 auto suffixPos = (valueSuffix.value().size() != 0) ? sKey.rfind(valueSuffix.value(), std::string::npos) : sKey.size();
210 if ((suffixPos == std::string::npos) || (suffixPos < keyPos))
211 {
212 return outcome::failure(boost::system::error_code{});
213 }
214
215 return sKey.substr(keyPos, suffixPos - keyPos);
216}
