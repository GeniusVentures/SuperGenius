12 VoteTracker::PushResult VoteTrackerImpl::push(const VotingMessage &vote,
13 size_t weight) {
14 auto vote_it = messages_.find(vote.id);
15 if (vote_it == messages_.end()) {
16 messages_[vote.id] = {vote};
17 total_weight_ += weight;
18 return PushResult::SUCCESS;
19 }
20 auto &equivotes = vote_it->second;
21 bool isDuplicate = visit_in_place(
22 equivotes,
23 [&vote](const VotingMessage &voting_message) {
24 return voting_message.block_hash() == vote.block_hash();
25 },
26 [&vote](const EquivocatoryVotingMessage &equivocatory_vote) {
27 return equivocatory_vote.first.block_hash() == vote.block_hash()
28 || equivocatory_vote.second.block_hash() == vote.block_hash();
29 });
30 if (! isDuplicate) {
31 return visit_in_place(
32 equivotes,
33 // if there is only single vote for that id, make it equivocatory vote
34 [&](const VotingMessage &voting_message) {
35 EquivocatoryVotingMessage v;
36 v.first = boost::get<VotingMessage>(equivotes);
37 v.second = vote;
38
39 messages_[vote.id] = v;
40 total_weight_ += weight;
41 return PushResult::EQUIVOCATED;
42 },
43 // otherwise return duplicated
44 [&](const EquivocatoryVotingMessage &) {
45 return PushResult::DUPLICATED;
46 });
47 }
48 return PushResult::DUPLICATED;
49 }
