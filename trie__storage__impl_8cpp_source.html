62 outcome::result<std::unique_ptr<PersistentTrieBatch>>
63 TrieStorageImpl::getPersistentBatch() {
64 logger_->debug("Initialize persistent trie batch with root: {}",
65 root_hash_.toHex());
66 auto trie_res = serializer_->retrieveTrie(Buffer{root_hash_});
67 if (trie_res.has_error()) {
68 logger_->error("Batch initialization failed, invalid root: {}",
69 root_hash_.toHex());
70 return trie_res.error();
71 }
72 return std::make_unique<PersistentTrieBatchImpl>(
73 codec_,
74 serializer_,
75 changes_,
76 std::move(trie_res.value()),
77 [this](auto const &new_root) {
78 root_hash_ = new_root;
79 logger_->debug("Update state root: {}", root_hash_);
80 });
81 }
