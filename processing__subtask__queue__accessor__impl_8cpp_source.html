89{
90 std::lock_guard<std::mutex> guard(m_mutexResults);
91 auto queue = m_subTaskQueueManager->GetQueueSnapshot();
92
93 std::set<std::string> subTaskIds;
94 for (size_t itemIdx = 0; itemIdx < (size_t)queue->subtasks().items_size(); ++itemIdx)

97 }
98
