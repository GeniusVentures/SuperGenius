49 base::Buffer numberToIndexKey(primitives::BlockNumber n) {

66 outcome::result<primitives::BlockNumber> lookupKeyToNumber(
67 const base::Buffer &key) {
68 if (key.size() < 4) {

70 }
71 return (uint64_t(key[0]) << 24u) | (uint64_t(key[1]) << 16u)
72 | (uint64_t(key[2]) << 8u) | uint64_t(key[3]);
