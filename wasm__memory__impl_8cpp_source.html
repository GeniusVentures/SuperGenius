31 WasmPointer WasmMemoryImpl::allocate(WasmSize size) {
32 if (size == 0) {
33 return 0;
34 }
35 const auto ptr = offset_;
36 const auto new_offset = ptr + size;
37
38 if (new_offset < static_cast<const uint32_t>(ptr)) { // overflow
39 return 0;
40 }
41 if (new_offset <= size_) {
42 offset_ = new_offset;
43 allocated_[ptr] = size;
44 return ptr;
45 }
46
47 return freealloc(size);
48 }

75 WasmPointer WasmMemoryImpl::findContaining(WasmSize size) {
76 auto min_value = std::numeric_limits<WasmPointer>::max();
77 WasmPointer min_key = 0;
78 for (const auto &[key, value] : deallocated_) {
79 if (min_value < 0) {

83 min_value = value;

90 WasmPointer WasmMemoryImpl::growAlloc(WasmSize size) {
91 // check that we do not exceed max memory size
92 if (static_cast<uint32_t>(offset_) > kMaxMemorySize - size) {
93 return 0;
94 }
95 // try to increase memory size up to offset + size * 4 (we multiply by 4
96 // to have more memory than currently needed to avoid resizing every time
97 // when we exceed current memory)
98 if (static_cast<uint32_t>(offset_) < kMaxMemorySize - size * 4) {
99 resize(offset_ + size * 4);
100 } else {
101 // if we can't increase by size * 4 then increase memory size by
102 // provided size
103 resize(offset_ + size);
104 }
105 return allocate(size);
106 }
