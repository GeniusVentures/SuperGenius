32bool ProcessingSubTaskQueueManager::CreateQueue(
33 std::list<SGProcessing::SubTask>& subTasks)
34{
35 auto timestamp = std::chrono::system_clock::now();
36
37 bool hasChunksDuplicates = false;
38
39 auto queue = std::make_shared<SGProcessing::SubTaskQueue>();
40 auto queueSubTasks = queue->mutable_subtasks();
41 auto processingQueue = queue->mutable_processing_queue();
42 for (auto itSubTask = subTasks.begin(); itSubTask != subTasks.end(); ++itSubTask)
43 {
44 // Move subtask to heap
45 auto subTask = std::make_unique<SGProcessing::SubTask>(std::move(*itSubTask));
46 queueSubTasks->mutable_items()->AddAllocated(subTask.release());
47 processingQueue->add_items();
48 }
49
50 std::unique_lock<std::mutex> guard(m_queueMutex);
51 m_queue = std::move(queue);
52
53 m_processedSubTaskIds = {};
54 // Map subtask IDs to subtask indices
55 std::vector<int> unprocessedSubTaskIndices;
56 for (int subTaskIdx = 0; subTaskIdx < m_queue->subtasks().items_size(); ++subTaskIdx)
57 {
58 const auto& subTaskId = m_queue->subtasks().items(subTaskIdx).subtaskid();
59 if (m_processedSubTaskIds.find(subTaskId) == m_processedSubTaskIds.end())
60 {
61 unprocessedSubTaskIndices.push_back(subTaskIdx);
62 }
63 }
64
65 m_processingQueue.CreateQueue(processingQueue, unprocessedSubTaskIndices);
66
67 m_logger->debug("QUEUE_CREATED");
68 LogQueue();
69
70 PublishSubTaskQueue();
71
72 if (m_subTaskQueueAssignmentEventSink)
73 {
74 std::vector<std::string> subTaskIds;
75 for (int subTaskIdx = 0; subTaskIdx < m_queue->subtasks().items_size(); ++subTaskIdx)
76 {
77 subTaskIds.push_back(m_queue->subtasks().items(subTaskIdx).subtaskid());
78 }
79
80 guard.unlock();
81 m_subTaskQueueAssignmentEventSink(subTaskIds);
82 }

114void ProcessingSubTaskQueueManager::ProcessPendingSubTaskGrabbing()
115{
116 // The method has to be called in scoped lock of queue mutex
117 m_dltGrabSubTaskTimeout.expires_at(boost::posix_time::pos_infin);
118 while (!m_onSubTaskGrabbedCallbacks.empty())
119 {
120 size_t itemIdx;
121 if (m_processingQueue.GrabItem(itemIdx))
122 {
123 LogQueue();
124 PublishSubTaskQueue();
125
126 m_onSubTaskGrabbedCallbacks.front()({ m_queue->subtasks().items(itemIdx) });
127 m_onSubTaskGrabbedCallbacks.pop_front();
128 }
129 else
130 {
131 // No available subtasks found
132 auto unlocked = m_processingQueue.UnlockExpiredItems(m_processingTimeout);
133 if (!unlocked)
134 {
135 break;
136 }
137 }
138 }
139
140 if (!m_onSubTaskGrabbedCallbacks.empty())
141 {
142 if (m_processedSubTaskIds.size() <(size_t)m_queue->processing_queue().items_size())
143 {
144 // Wait for subtasks are processed
145 auto timestamp = std::chrono::system_clock::now();
146 auto lastLockTimestamp = m_processingQueue.GetLastLockTimestamp();
147 auto lastExpirationTime = lastLockTimestamp + m_processingTimeout;
148
149 std::chrono::milliseconds grabSubTaskTimeout =
150 (lastExpirationTime > timestamp)
151 ? std::chrono::duration_cast<std::chrono::milliseconds>(lastExpirationTime - timestamp)
152 : std::chrono::milliseconds(1);
153
154 m_logger->debug("GRAB_TIMEOUT {}ms", grabSubTaskTimeout.count());
155
156 m_dltGrabSubTaskTimeout.expires_from_now(
157 boost::posix_time::milliseconds(grabSubTaskTimeout.count()));
158
159 m_dltGrabSubTaskTimeout.async_wait(std::bind(
160 &ProcessingSubTaskQueueManager::HandleGrabSubTaskTimeout, this, std::placeholders::_1));
161 }
162 else
163 {
164 while (!m_onSubTaskGrabbedCallbacks.empty())
165 {
166 // Let the requester know that there are no available subtasks
167 m_onSubTaskGrabbedCallbacks.front()({});
168 // Reset the callback
169 m_onSubTaskGrabbedCallbacks.pop_front();
170 }
171 }
172 }
