126 template <typename R, typename... Args>
127 outcome::result<R> executeMaybeAt(
128 std::string_view name,
129 const boost::optional<base::Hash256> &state_root,
130 CallPersistency persistency,
131 Args &&... args) {
132 logger_->debug("Executing export function: {}", name);
133 if (state_root.has_value()) {
134 logger_->debug("Resetting state to: {}", state_root.value().toHex());
135 }
136
137 auto environment = createRuntimeEnvironment(persistency, state_root);
138 auto &&[module, memory, opt_batch] = environment;
139
140 runtime::WasmPointer ptr = 0u;
141 runtime::WasmSize len = 0u;
142
143 if constexpr (sizeof...(args) > 0) {
144 OUTCOME_TRY((auto &&, buffer), scale::encode(std::forward<Args>(args)...));
145 len = buffer.size();
146 ptr = memory->allocate(len);
147 memory->storeBuffer(ptr, base::Buffer(std::move(buffer)));
148 }
149
150 wasm::LiteralList ll{wasm::Literal(ptr), wasm::Literal(len)};
151
152 wasm::Name wasm_name = std::string(name);
153
154 OUTCOME_TRY((auto &&, res), executor_.call(*module, wasm_name, ll));
155 memory->reset();
156 if constexpr (!std::is_same_v<void, R>) {
157 WasmResult r(res.geti64());
158 auto buffer = memory->loadN(r.address, r.length);
159 // TODO (yuraz) PRE-98: after check for memory overflow is done,
160 // refactor it
161 return scale::decode<R>(std::move(buffer));
162 }
163
164 if(opt_batch) {
165 BOOST_OUTCOME_TRYV2(auto &&, opt_batch.value()->writeBack());
166 }
167 return outcome::success();
168 }
