45 outcome::result<std::shared_ptr<KeyValueBlockStorage>>
46 KeyValueBlockStorage::create(
47 base::Buffer state_root,
48 const std::shared_ptr<storage::BufferStorage> &storage,
49 const std::shared_ptr<crypto::Hasher> &hasher,
50 const BlockHandler &on_finalized_block_found) {
51 auto block_storage = std::make_shared<KeyValueBlockStorage>(
52 KeyValueBlockStorage(storage, hasher));
53
54 auto last_finalized_block_hash_res =
55 block_storage->getLastFinalizedBlockHash();
56
57 if (last_finalized_block_hash_res.has_value()) {
58 return loadExisting(storage, hasher, on_finalized_block_found);
59 }
60
61 if (last_finalized_block_hash_res
62 == outcome::failure(Error::FINALIZED_BLOCK_NOT_FOUND)) {
63 return createWithGenesis(
64 std::move(state_root), storage, hasher, on_finalized_block_found);
65 }
66
67 return last_finalized_block_hash_res.error();
68 }

70 outcome::result<std::shared_ptr<KeyValueBlockStorage>>
71 KeyValueBlockStorage::loadExisting(
72 const std::shared_ptr<storage::BufferStorage> &storage,
73 std::shared_ptr<crypto::Hasher> hasher,
74 const BlockHandler &on_finalized_block_found) {
75 auto block_storage = std::make_shared<KeyValueBlockStorage>(
76 KeyValueBlockStorage(storage, std::move(hasher)));
77
78 OUTCOME_TRY((auto &&, last_finalized_block_hash),
79 block_storage->getLastFinalizedBlockHash());
80
81 OUTCOME_TRY((auto &&, block_header),
82 block_storage->getBlockHeader(last_finalized_block_hash));
83
84 primitives::Block finalized_block;
85 finalized_block.header = block_header;
86
87 on_finalized_block_found(finalized_block);
88
89 return std::move(block_storage);
90 }

92 outcome::result<std::shared_ptr<KeyValueBlockStorage>>
93 KeyValueBlockStorage::createWithGenesis(
94 base::Buffer state_root,
95 const std::shared_ptr<storage::BufferStorage> &storage,
96 std::shared_ptr<crypto::Hasher> hasher,
97 const BlockHandler &on_genesis_created) {
98 auto block_storage = std::make_shared<KeyValueBlockStorage>(
99 KeyValueBlockStorage(storage, std::move(hasher)));
100
101 BOOST_OUTCOME_TRYV2(auto &&, block_storage->ensureGenesisNotExists());
102
103 // state root type is Hash256, however for consistency with spec root hash
104 // returns buffer. So we need this conversion
105 OUTCOME_TRY((auto &&, state_root_blob),
106 base::Hash256::fromSpan(state_root.toVector()));
107
108 auto extrinsics_root_buf = trieRoot({});
109 // same reason for conversion as few lines above
110 OUTCOME_TRY((auto &&, extrinsics_root),
111 base::Hash256::fromSpan(extrinsics_root_buf.toVector()));
112
113 // genesis block initialization
114 primitives::Block genesis_block;
115 genesis_block.header.number = 0;
116 genesis_block.header.extrinsics_root = extrinsics_root;
117 genesis_block.header.state_root = state_root_blob;
118 // the rest of the fields have default value
119
120 OUTCOME_TRY((auto &&, genesis_block_hash), block_storage->putBlock(genesis_block));
121 BOOST_OUTCOME_TRYV2(auto &&, storage->put(storage::kGenesisBlockHashLookupKey,
122 Buffer{genesis_block_hash}));
123 BOOST_OUTCOME_TRYV2(auto &&, block_storage->setLastFinalizedBlockHash(genesis_block_hash));
124
125 on_genesis_created(genesis_block);
126 return std::move(block_storage);
127 }

176 outcome::result<void> KeyValueBlockStorage::putBlockData(
177 primitives::BlockNumber block_number,
178 const primitives::BlockData &block_data) {
179 primitives::BlockData to_insert;
180
181 // if block data does not exist, put a new one. Otherwise get the old one
182 // and merge with the new one. During the merge new block data fields have
183 // higher priority over the old ones (old ones should be rewritten)
184 auto existing_block_data_res = getBlockData(block_data.hash);
185 if (! existing_block_data_res) {
186 to_insert = block_data;
187 } else {
188 auto existing_data = existing_block_data_res.value();
189
190 // add all the fields from the new block_data
191 to_insert.header =
192 block_data.header ? block_data.header : existing_data.header;
193 to_insert.body = block_data.body ? block_data.body : existing_data.body;
194 to_insert.justification = block_data.justification
195 ? block_data.justification
196 : existing_data.justification;
197 to_insert.message_queue = block_data.message_queue
198 ? block_data.message_queue
199 : existing_data.message_queue;
200 to_insert.receipt =
201 block_data.receipt ? block_data.receipt : existing_data.receipt;
202 }
203
204 OUTCOME_TRY((auto &&, encoded_block_data), scale::encode(to_insert));
205 BOOST_OUTCOME_TRYV2(auto &&, putWithPrefix(*storage_,
206 Prefix::BLOCK_DATA,
207 block_number,
208 block_data.hash,
209 Buffer{encoded_block_data}));
210 return outcome::success();
211 }

213 outcome::result<primitives::BlockHash> KeyValueBlockStorage::putBlock(
214 const primitives::Block &block) {
215 // TODO(xDimon): Need to implement mechanism for wipe out orphan blocks
216 // (in side-chains whom rejected by finalization)
217 // for avoid leaks of storage space
218 auto block_hash = hasher_->blake2b_256(scale::encode(block.header).value());
219 auto block_in_storage_res =
220 getWithPrefix(*storage_, Prefix::HEADER, block_hash);
221 if (block_in_storage_res.has_value()) {
222 return Error::BLOCK_EXISTS;
223 }
224 if (block_in_storage_res
225 != outcome::failure(blockchain::Error::BLOCK_NOT_FOUND)) {
226 return block_in_storage_res.error();
227 }
228
229 // insert our block's parts into the database-
230 BOOST_OUTCOME_TRYV2(auto &&, putBlockHeader(block.header));
231
232 primitives::BlockData block_data;
233 block_data.hash = block_hash;
234 block_data.header = block.header;
235 block_data.body = block.body;
236
237 BOOST_OUTCOME_TRYV2(auto &&, putBlockData(block.header.number, block_data));
238 logger_->info("Added block. Number: {}. Hash: {}. State root: {}",
239 block.header.number,
240 block_hash.toHex(),
241 block.header.state_root.toHex());
242 return block_hash;
243 }

259 outcome::result<void> KeyValueBlockStorage::removeBlock(
260 const primitives::BlockHash &hash,
261 const primitives::BlockNumber &number) {
262 auto block_lookup_key = numberAndHashToLookupKey(number, hash);
263 auto header_lookup_key = prependPrefix(block_lookup_key, Prefix::HEADER);
264 if (auto rm_res = storage_->remove(header_lookup_key); !rm_res) {
265 logger_->error("could not remove header from the storage: {}",
266 rm_res.error().message());
267 return rm_res;
268 }
269
270 auto body_lookup_key = prependPrefix(block_lookup_key, Prefix::BLOCK_DATA);
271 if (auto rm_res = storage_->remove(body_lookup_key); !rm_res) {
272 logger_->error("could not remove body from the storage: {}",
273 rm_res.error().message());
274 return rm_res;
275 }
276 return outcome::success();
277 }
