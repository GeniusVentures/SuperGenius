170 void StorageExtension::ext_blake2_256_enumerated_trie_root(
171 runtime::WasmPointer values_data,
172 runtime::WasmPointer lengths_data,
173 runtime::WasmSize values_num,
174 runtime::WasmPointer result) {
175 std::vector<uint32_t> lengths(values_num);
176 for (size_t i = 0; i < values_num; i++) {
177 lengths.at(i) = memory_->load32u(lengths_data + i * 4);
178 }
179 std::forward_list<Buffer> values(values_num);
180 uint32_t offset = 0;
181 auto curr_value = values.begin();
182 for (size_t i = 0; i < values_num; i++, curr_value++) {
183 *curr_value = memory_->loadN(values_data + offset, lengths.at(i));
184 offset += lengths.at(i);
185 }
186 auto ordered_hash =
187 storage::trie::calculateOrderedTrieHash(values.begin(), values.end());
188 if (ordered_hash.has_value()) {
189 memory_->storeBuffer(result, ordered_hash.value());
190 } else {
191 logger_->error(
192 "ext_blake2_256_enumerated_trie_root resulted with an error: {}",
193 ordered_hash.error().message());
194 }
195 }

383 runtime::WasmPointer StorageExtension::ext_trie_blake2_256_root_version_1(
384 runtime::WasmSpan values_data) {
385 auto [ptr, size] = runtime::WasmResult(values_data);
386 const auto &buffer = memory_->loadN(ptr, size);
387 const auto &pairs = scale::decode<KeyValueCollection>(buffer);
388 if (!pairs) {
389 logger_->error("failed to decode pairs: {}", pairs.error().message());
390 std::terminate();
391 }
392
393 auto &&pv = pairs.value();
394 storage::trie::SuperGeniusCodec codec;
395 if (pv.empty()) {
396 static const auto empty_root = base::Buffer{}.put(codec.hash256({0}));
397 auto res = memory_->storeBuffer(empty_root);
398 return runtime::WasmResult(res).address;
399 }
400 storage::trie::SuperGeniusTrieImpl trie;
401 for (auto &&p : pv) {
402 auto &&key = p.first;
403 auto &&value = p.second;
404 // already scale-encoded
405 trie.put(key, value);
406 }
407 const auto &enc = codec.encodeNode(*trie.getRoot());
408 if (!enc) {
409 logger_->error("failed to encode trie root: {}", enc.error().message());
410 std::terminate();
411 }
412 const auto &hash = codec.hash256(enc.value());
413
414 std::cout << "res hash = " << hash.toHex() << std::endl;
415
416 auto res = memory_->storeBuffer(hash);
417 return runtime::WasmResult(res).address;
418 }

445 boost::optional<base::Buffer> StorageExtension::calcStorageChangesRoot(
446 base::Hash256 parent_hash) const {
447 auto batch = storage_provider_->tryGetPersistentBatch().value();
448 boost::optional<storage::changes_trie::ChangesTrieConfig> trie_config;
449 auto config_bytes_res = batch->get(CHANGES_CONFIG_KEY);
450 if (config_bytes_res.has_error()) {
451 if (config_bytes_res.error() != storage::trie::TrieError::NO_VALUE) {
452 logger_->error("ext_storage_changes_root resulted with an error: {}",
453 config_bytes_res.error().message());
454 return boost::none;
455 }
456 logger_->debug("ext_storage_changes_root: no changes trie config found");
457 trie_config = boost::none;
458 } else {
459 auto config_res = scale::decode<storage::changes_trie::ChangesTrieConfig>(
460 config_bytes_res.value());
461 if (config_res.has_error()) {
462 logger_->error("ext_storage_changes_root resulted with an error: {}",
463 config_res.error().message());
464 return boost::none;
465 }
466 trie_config = config_res.value();
467 }
468
469 // if no config found in the storage, then disable tracking blocks changes
470 if (! trie_config.has_value()) {
471 trie_config = storage::changes_trie::ChangesTrieConfig{
472 /*.digest_interval = */0, /*.digest_levels = */0};
473 }
474
475 logger_->debug(
476 "ext_storage_changes_root constructing changes trie with parent_hash: "
477 "{}",
478 parent_hash.toHex());
479 auto trie_hash_res = changes_tracker_->constructChangesTrie(
480 parent_hash, trie_config.value());
481 if (trie_hash_res.has_error()) {
482 logger_->error("ext_storage_changes_root resulted with an error: {}",
483 trie_hash_res.error().message());
484 return boost::none;
485 }
486 base::Buffer result_buf(trie_hash_res.value());
487 logger_->debug("ext_storage_changes_root with parent_hash {} result is: {}",
488 parent_hash.toHex(),
489 result_buf.toHex());
490 return result_buf;
491 }
