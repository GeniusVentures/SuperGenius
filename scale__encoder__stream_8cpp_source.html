53 void encodeCompactInteger(const CompactInteger &value,
54 ScaleEncoderStream &out) {
55 // cannot encode negative numbers
56 // there is no description how to encode compact negative numbers
57 if (value < 0) {
58 base::raise(EncodeError::NEGATIVE_COMPACT_INTEGER);
59 }
60
61 if (value < compact::EncodingCategoryLimits::kMinUint16) {
62 encodeFirstCategory(value.convert_to<uint8_t>(), out);
63 return;
64 }
65
66 if (value < compact::EncodingCategoryLimits::kMinUint32) {
67 encodeSecondCategory(value.convert_to<uint16_t>(), out);
68 return;
69 }
70
71 if (value < compact::EncodingCategoryLimits::kMinBigInteger) {
72 encodeThirdCategory(value.convert_to<uint32_t>(), out);
73 return;
74 }
75
76 // number of bytes required to represent value
77 size_t bigIntLength = countBytes(value);
78
79 // number of bytes to scale-encode value
80 // 1 byte is reserved for header
81 size_t requiredLength = 1 + bigIntLength;
82
83 if (bigIntLength > 67) {
84 base::raise(EncodeError::COMPACT_INTEGER_TOO_BIG);
85 }
86
87 ByteArray result;
88 result.reserve(requiredLength);
89
90 /* The value stored in 6 major bits of header is used
91 * to encode number of bytes for storing big integer.
92 * Value formed by 6 bits varies from 0 to 63 == 2^6 - 1,
93 * However big integer byte count starts from 4,
94 * so to store this number we should decrease this value by 4.
95 * And the range of bytes number for storing big integer
96 * becomes 4 .. 67. To form resulting header we need to move
97 * those bits representing bytes count to the left by 2 positions
98 * by means of multiplying by 4.
99 * Minor 2 bits store encoding option, in our case it is 0b11 == 3
100 * We just add 3 to the result of operations above
101 */
102 uint8_t header = (bigIntLength - 4) * 4 + 3;
103
104 result.push_back(header);
105
106 CompactInteger v{value};
107 for (size_t i = 0; i < bigIntLength; ++i) {
108 result.push_back(static_cast<uint8_t>(
109 v & 0xFF)); // push back least significant byte
110 v >>= 8;
111 }
112
113 for (const uint8_t c : result) {
114 out << c;
115 }
116 }
