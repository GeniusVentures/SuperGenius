17 outcome::result<std::vector<uint8_t>> Bip39ProviderImpl::calculateEntropy(
18 const std::vector<std::string> &word_list) {
19 // make entropy accumulator
20 OUTCOME_TRY((auto &&, entropy_accumulator),
21 bip39::EntropyAccumulator::create(word_list.size()));
22
23 // accumulate entropy
24 for (const auto &w : word_list) {
25 OUTCOME_TRY((auto &&, entropy_token), dictionary_.findValue(w));
26 BOOST_OUTCOME_TRYV2(auto &&, entropy_accumulator.append(entropy_token));
27 }
28
29 OUTCOME_TRY((auto &&, mnemonic_checksum), entropy_accumulator.getChecksum());
30 OUTCOME_TRY((auto &&, calculated_checksum), entropy_accumulator.calculateChecksum());
31
32 if (mnemonic_checksum != calculated_checksum) {
33 return bip39::MnemonicError::INVALID_MNEMONIC;
34 }
35
36 // finally get entropy
37 return entropy_accumulator.getEntropy();
38 }
