45 std::shared_ptr<network::SyncProtocolClient>
46 ProductionSynchronizerImpl::selectNextClient(
47 std::unordered_set<std::shared_ptr<network::SyncProtocolClient>>
48 &polled_clients) const {
49 // we want to ask each client until we get the blocks we lack, but the
50 // sync_clients_ set can change between the requests, so we need to keep
51 // track of the clients we already asked
52 std::shared_ptr<network::SyncProtocolClient> next_client;
53 for (const auto &client : sync_clients_->clients) {
54 if (polled_clients.find(client) == polled_clients.end()) {
55 next_client = client;
56 polled_clients.insert(client);
57 break;
58 }
59 }
60
61 if (!next_client) {
62 // we asked all clients we could, so start over
63 polled_clients.clear();
64 next_client = *sync_clients_->clients.begin();
65 polled_clients.insert(next_client);
66 }
67 return next_client;
68 }

75 boost::optional<std::vector<primitives::Block>> getBlocks(
76 const network::BlocksResponse &response) {
77 // now we need to check if every block data from response contains header;
78 // if any of them does not contain block header, we should proceed to the
79 // next client
80 std::vector<primitives::Block> blocks;
81 for (const auto &block_data : response.blocks) {
82 primitives::Block block;
83 if (!block_data.header) {
84 // that's bad, we can't insert a block, which does not have at
85 // least a header
86 return boost::none;
87 }
88 block.header = *block_data.header;
89
90 if (block_data.body) {
91 block.body = *block_data.body;
92 }
93
94 blocks.push_back(block);
95 }
96 return blocks;
97 }

99 void ProductionSynchronizerImpl::pollClients(
100 network::BlocksRequest request,
101 primitives::AuthorityIndex authority_index,
102 const BlocksHandler &requested_blocks_handler) const {
103 auto next_client = sync_clients_->clients[authority_index];
104
105 next_client->requestBlocks(
106 request,
107 [self_wp{weak_from_this()},
108 request{std::move(request)},
109 requested_blocks_handler{requested_blocks_handler}](
110 auto &&response_res) mutable {
111 if (auto self = self_wp.lock()) {
112 // if response exists then get blocks and send them to handle
113 if (response_res && ! response_res.value().blocks.empty()) {
114 auto blocks_opt = getBlocks(response_res.value());
115 if (blocks_opt) {
116 return requested_blocks_handler(blocks_opt.value());
117 }
118 } else if (! response_res) {
119 self->logger_->error("Could not sync. Error: {}",
120 response_res.error().message());
121 } else {
122 self->logger_->error("Could not sync. Empty response");
123 }
124 }
125 });
126 }
