9 outcome::result<std::unique_ptr<ChangesTrie>> ChangesTrie::buildFromChanges(
10 const primitives::BlockNumber &parent_number,
11 const std::shared_ptr<storage::trie::SuperGeniusTrieFactory> &trie_factory,
12 std::shared_ptr<storage::trie::Codec> codec,
13 const ExtrinsicsChanges &extinsics_changes,
14 const ChangesTrieConfig &config) {
15 BOOST_ASSERT(trie_factory != nullptr);
16 auto changes_storage = trie_factory->createEmpty();
17
18 for (auto &change : extinsics_changes) {
19 auto &key = change.first;
20 auto &changers = change.second;
21 auto current_number = parent_number + 1;
22 KeyIndexVariant keyIndex{ExtrinsicsChangesKey{{current_number, key}}};
23 OUTCOME_TRY((auto &&, key_enc), scale::encode(keyIndex));
24 OUTCOME_TRY((auto &&, value), scale::encode(changers));
25 base::Buffer value_buf {std::move(value)};
26 BOOST_OUTCOME_TRYV2(auto &&, changes_storage->put(base::Buffer{std::move(key_enc)},
27 std::move(value_buf)));
28 }
29
30 return std::unique_ptr<ChangesTrie>{
31 new ChangesTrie{std::move(changes_storage), std::move(codec)}};
32 }

43 base::Hash256 ChangesTrie::getHash() const {
44 if (changes_trie_ == nullptr) {
45 return {};
46 }
47
48 auto root = changes_trie_->getRoot();
49 if (root == nullptr) {
50 logger_->warn("Get root of empty changes trie");
51 return codec_->hash256({0});
52 }
53 auto enc_res = codec_->encodeNode(*root);
54 if (enc_res.has_error()) {
55 logger_->error("Encoding Changes trie failed"
56 + enc_res.error().message());
57 return {};
58 }
59 auto hash_bytes = codec_->hash256(enc_res.value());
60 return hash_bytes;
61 }
