23 void SyncingNodeApplication::run() {
24 logger_->info("Start as {} with PID {}", typeid(*this).name(), getpid());
25
26 app_state_manager_->atLaunch([this] {
27 // execute listeners
28 io_context_->post([this] {
29 const auto &current_peer_info =
30 injector_.template create<network::OwnPeerInfo>();
31 auto &host = injector_.template create<libp2p::Host &>();
32 for (const auto &ma : current_peer_info.addresses) {
33 auto listen = host.listen(ma);
34 if (! listen) {
35 logger_->error("Cannot listen address {}. Error: {}",
36 ma.getStringAddress(),
37 listen.error().message());
38 std::exit(1);
39 }
40 }
41 for (const auto &boot_node : config_storage_->getBootNodes().peers) {
42 host.newStream(
43 boot_node,
44 network::kGossipProtocol,
45 [this, boot_node](const auto &stream_res) {
46 if (! stream_res) {
47 this->logger_->error(
48 "Could not establish connection with {}. Error: {}",
49 boot_node.id.toBase58(),
50 stream_res.error().message());
51 return;
52 }
53 this->router_->handleGossipProtocol(stream_res.value());
54 });
55 }
56 this->router_->init();
57 });
58 return true;
59 });
60
61 app_state_manager_->atLaunch([ctx{io_context_}] {
62 std::thread asio_runner([ctx{ctx}] { ctx->run(); });
63 asio_runner.detach();
64 return true;
65 });
66
67 app_state_manager_->atShutdown([ctx{io_context_}] { ctx->stop(); });
68
69 app_state_manager_->run();
70 }
