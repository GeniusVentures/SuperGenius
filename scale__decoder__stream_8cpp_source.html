12 CompactInteger decodeCompactInteger(ScaleDecoderStream &stream) {
13 auto first_byte = stream.nextByte();
14
15 const uint8_t flag = (first_byte)&0b00000011u;
16
17 size_t number = 0u;
18
19 switch (flag) {
20 case 0b00u: {
21 number = static_cast<size_t>(first_byte >> 2u);
22 break;
23 }
24
25 case 0b01u: {
26 auto second_byte = stream.nextByte();
27
28 number = (static_cast<size_t>((first_byte)&0b11111100u)
29 + static_cast<size_t>(second_byte) * 256u)
30 >> 2u;
31 break;
32 }
33
34 case 0b10u: {
35 number = first_byte;
36 size_t multiplier = 256u;
37 if (!stream.hasMore(3u)) {
38 // not enough data to decode integer
39 base::raise(DecodeError::NOT_ENOUGH_DATA);
40 }
41
42 for (auto i = 0u; i < 3u; ++i) {
43 // we assured that there are 3 more bytes,
44 // no need to make checks in a loop
45 number += (stream.nextByte()) * multiplier;
46 multiplier = multiplier << 8u;
47 }
48 number = number >> 2u;
49 break;
50 }
51
52 case 0b11: {
53 auto bytes_count = ((first_byte) >> 2u) + 4u;
54 if (!stream.hasMore(bytes_count)) {
55 // not enough data to decode integer
56 base::raise(DecodeError::NOT_ENOUGH_DATA);
57 }
58
59 CompactInteger multiplier{1u};
60 CompactInteger value = 0;
61 // we assured that there are m more bytes,
62 // no need to make checks in a loop
63 for (auto i = 0u; i < bytes_count; ++i) {
64 value += (stream.nextByte()) * multiplier;
65 multiplier *= 256u;
66 }
67
68 return value; // special case
69 }
70
71 default:
72 UNREACHABLE
73 }
74
75 return CompactInteger{number};
76 }
