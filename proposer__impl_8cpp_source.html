26 auto inherent_xts_res =

34
35 auto log_push_error = [this](const primitives::Extrinsic &xt,
36 std::string_view message) {
37 logger_->warn("Extrinsic {} was not added to the block. Reason: {}",
38 xt.data.toHex(),
39 message);
40 };
41
42 for (const auto &xt : inherent_xts) {
43 logger_->debug("Adding inherent extrinsic: {}", xt.data.toHex());
44 auto inserted_res = block_builder->pushExtrinsic(xt);
45 if (! inserted_res) {
46 log_push_error(xt, inserted_res.error().message());
47 return inserted_res.error();
48 }
49 }
50
51 const auto &ready_txs = transaction_pool_->getReadyTransactions();
52
53 for (const auto &[hash, tx] : ready_txs) {
54 logger_->debug("Adding extrinsic: {}", tx->ext.data.toHex());
55 auto inserted_res = block_builder->pushExtrinsic(tx->ext);
56 if (! inserted_res) {
57 log_push_error(tx->ext, inserted_res.error().message());
58 return inserted_res.error();
59 }
60 }
61
62 auto block = block_builder->bake();
63
64 for (const auto &[hash, tx] : ready_txs) {
65 auto removed_res = transaction_pool_->removeOne(hash);
66 if (! removed_res) {
67 logger_->error(
68 "Can't remove extrinsic (hash={}) after adding to the block. "
69 "Reason: {}",
70 hash.toHex(),
71 removed_res.error().message());
72 }
73 }
74
75 return block;
