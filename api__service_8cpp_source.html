52 bool ApiService::prepare() {
53 for (const auto &listener : listeners_) {
54 auto on_new_session = [wp = weak_from_this()](
55 const sptr<Session> &session) mutable {
56 auto self = wp.lock();
57 if (!self) {
58 return;
59 }
60
61 if (SessionType::kWs == session->type()) {
62 auto subscribed_session =
63 self->storeSessionWithId(session->id(), session);
64 subscribed_session->setCallback([wp](SessionPtr &session,
65 const auto &key,
66 const auto &data,
67 const auto &block) {
68 if (auto self = wp.lock()) {
69 jsonrpc::Value::Array out_data;
70 out_data.emplace_back(api::makeValue(key));
71 out_data.emplace_back(api::makeValue(data));
72
75
76 jsonrpc::Value::Struct result;
77 result["changes"] = std::move(out_data);
78 result["block"] = api::makeValue(block);
79
80 jsonrpc::Value::Struct params;
81 params["result"] = std::move(result);
82 params["subscription"] = 0;
83
84 self->server_->processJsonData(params,
85 [&](const std::string &response) {
86 session->respond(response);
87 });
88 }
89 });
90 }
91
92 session->connectOnRequest(
93 [wp](std::string_view request,
94 std::shared_ptr<Session> session) mutable {
95 auto self = wp.lock();
96 if (! self) return;
97
98 auto thread_session_auto_release = [](void *) {
99 threaded_info.releaseSessionId();
100 };
101
102 threaded_info.storeSessionId(session->id());
103 std::unique_ptr<void, decltype(thread_session_auto_release)>
104
105 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
106 thread_session_keeper(reinterpret_cast<void *>(0xff),
107 std::move(thread_session_auto_release));
108
109 // process new request
110 self->server_->processData(
111 std::string(request),
112 [session = std::move(session)](
113 const std::string &response) mutable {
114 // process response
115 session->respond(response);
116 });
117 });
118
119 session->connectOnCloseHandler(
120 [wp](Session::SessionId id, SessionType /*type*/) {
121 if (auto self = wp.lock()) self->removeSessionById(id);
122 });
123 };
124
125 listener->setHandlerForNewSession(std::move(on_new_session));
126 }
127 return true;

167 outcome::result<uint32_t> ApiService::subscribeSessionToKeys(
168 const std::vector<base::Buffer> &keys) {
169 if (auto session_id = threaded_info.fetchSessionId(); session_id) {
170 if (auto session = findSessionById(*session_id)) {
171 const auto id = session->generateSubscriptionSetId();
172 for (auto &key : keys) {
174 session->subscribe(id, key);
175 }
176 return static_cast<uint32_t>(id);
177 }
178 throw jsonrpc::InternalErrorFault(
179 "Internal error. No session was stored for subscription.");
180 }
181 throw jsonrpc::InternalErrorFault(
182 "Internal error. No session was bound to subscription.");
183 }

185 outcome::result<void> ApiService::unsubscribeSessionFromIds(

187 if (auto session_id = threaded_info.fetchSessionId(); session_id) {
188 if (auto session = findSessionById(*session_id)) {

191 }
192 throw jsonrpc::InternalErrorFault(
193 "Internal error. No session was stored for subscription.");
194 }
195 throw jsonrpc::InternalErrorFault(
196 "Internal error. No session was binded to subscription.");
