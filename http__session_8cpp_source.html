65 void HttpSession::asyncWrite(Message &&message) {
66 // we need to put message into a shared ptr for async operations
67 using message_type = decltype(message);
68 auto m = std::make_shared<std::decay_t<message_type>>(
69 std::forward<message_type>(message));
70
71 // write response
72 boost::beast::http::async_write(
73 stream_, *m, [self = shared_from_this(), m](auto ec, auto size) {
74 self->onWrite(ec, size, m->need_eof());
