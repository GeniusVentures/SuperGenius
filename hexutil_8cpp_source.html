27 std::string int_to_hex(uint64_t n, size_t fixed_width) noexcept {
28 std::stringstream result;
29 result.width(fixed_width);
30 result.fill('0');
31 result << std::hex << std::uppercase << n;
32 auto str = result.str();
33 if (str.length() % 2 != 0) {
34 str.push_back('\0');
35 for (int64_t i = str.length() - 2; i >= 0; --i) {
36 str[i + 1] = str[i];
37 }
38 str[0] = '0';
39 }
40 return str;
41 }

55 outcome::result<std::vector<uint8_t>> unhex(std::string_view hex) {
