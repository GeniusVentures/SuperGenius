57/* generally called after SHA3_KECCAK_SPONGE_WORDS-ctx->capacityWords words
58 * are XORed into the state s
59 */
60static void keccakf(uint64_t s[25]) {
61 int i, j, round; // NOLINT
62 uint64_t t, bc[5]; // NOLINT
63#define KECCAK_ROUNDS 24
64
65 for (round = 0; round < KECCAK_ROUNDS; round++) {
66 /* Theta */
67 for (i = 0; i < 5; i++)
68 bc[i] = s[i] ^ s[i + 5] ^ s[i + 10] ^ s[i + 15] ^ s[i + 20];
69
70 for (i = 0; i < 5; i++) {
71 t = bc[(i + 4) % 5] ^ SHA3_ROTL64(bc[(i + 1) % 5], 1);
72 for (j = 0; j < 25; j += 5) s[j + i] ^= t;
73 }
74
75 /* Rho Pi */
76 t = s[1];
77 for (i = 0; i < 24; i++) {
78 j = keccakf_piln[i]; // NOLINT
79 bc[0] = s[j];
80 s[j] = SHA3_ROTL64(t, keccakf_rotc[i]);
81 t = bc[0];
82 }
83
84 /* Chi */
85 for (j = 0; j < 25; j += 5) {
86 for (i = 0; i < 5; i++) bc[i] = s[j + i];
87 for (i = 0; i < 5; i++) s[j + i] ^= (~bc[(i + 1) % 5]) & bc[(i + 2) % 5];
88 }
89
90 /* Iota */
91 s[0] ^= keccakf_rndc[round];
92 }
93}

98sha3_return_t sha3_Init(void *priv, unsigned bitSize) {
99 sha3_context *ctx = (sha3_context *)priv;
100 if (bitSize != 256 && bitSize != 384 && bitSize != 512)
101 return SHA3_RETURN_BAD_PARAMS;
102 memset(ctx, 0, sizeof(*ctx));
103 ctx->capacityWords = 2 * bitSize / (8 * sizeof(uint64_t));
104 return SHA3_RETURN_OK;
105}

126void sha3_Update(void *priv, void const *bufIn, size_t len) {
127 sha3_context *ctx = (sha3_context *)priv;
128
129 /* 0...7 -- how much is needed to have a word */
130 unsigned old_tail = (8 - ctx->byteIndex) & 7;
131
132 size_t words;
133 unsigned tail;
134 size_t i;
135
136 const uint8_t *buf = bufIn;
137
138 SHA3_TRACE_BUF("called to update with:", buf, len);
139
140 SHA3_ASSERT(ctx->byteIndex < 8);
141 SHA3_ASSERT(ctx->wordIndex < sizeof(ctx->s) / sizeof(ctx->s[0]));
142
143 if (len < old_tail) { /* have no complete word or haven't started
144 * the word yet */
145 SHA3_TRACE("because %d<%d, store it and return",
146 (unsigned)len,
147 (unsigned)old_tail);
148 /* endian-independent code follows: */
149 while (len--)
150 ctx->saved |= (uint64_t)(*(buf++)) << ((ctx->byteIndex++) * 8);
151 SHA3_ASSERT(ctx->byteIndex < 8);
152 return;
153 }
154
155 if (old_tail) { /* will have one word to process */
156 SHA3_TRACE("completing one word with %d bytes", (unsigned)old_tail);
157 /* endian-independent code follows: */
158 len -= old_tail;
159 while (old_tail--)
160 ctx->saved |= (uint64_t)(*(buf++)) << ((ctx->byteIndex++) * 8);
161
162 /* now ready to add saved to the sponge */
163 ctx->s[ctx->wordIndex] ^= ctx->saved;
164 SHA3_ASSERT(ctx->byteIndex == 8);
165 ctx->byteIndex = 0;
166 ctx->saved = 0;
167 if (++ctx->wordIndex
168 == (SHA3_KECCAK_SPONGE_WORDS - SHA3_CW(ctx->capacityWords))) {
169 keccakf(ctx->s);
170 ctx->wordIndex = 0;
171 }
172 }
173
174 /* now work in full words directly from input */
175
176 SHA3_ASSERT(ctx->byteIndex == 0);
177
178 words = len / sizeof(uint64_t);
179 tail = len - words * sizeof(uint64_t);
180
181 SHA3_TRACE("have %d full words to process", (unsigned)words);
182
183 for (i = 0; i < words; i++, buf += sizeof(uint64_t)) {
184 const uint64_t t =
185 (uint64_t)(buf[0]) | ((uint64_t)(buf[1]) << 8 * 1)
186 | ((uint64_t)(buf[2]) << 8 * 2) | ((uint64_t)(buf[3]) << 8 * 3)
187 | ((uint64_t)(buf[4]) << 8 * 4) | ((uint64_t)(buf[5]) << 8 * 5)
188 | ((uint64_t)(buf[6]) << 8 * 6) | ((uint64_t)(buf[7]) << 8 * 7);
189#if defined(__x86_64__) || defined(__i386__)
190 SHA3_ASSERT(memcmp(&t, buf, 8) == 0);
191#endif
192 ctx->s[ctx->wordIndex] ^= t;
193 if (++ctx->wordIndex
194 == (SHA3_KECCAK_SPONGE_WORDS - SHA3_CW(ctx->capacityWords))) {
195 keccakf(ctx->s);
196 ctx->wordIndex = 0;
197 }
198 }
199
200 SHA3_TRACE("have %d bytes left to process, save them", (unsigned)tail);
201
202 /* finally, save the partial word */
203 SHA3_ASSERT(ctx->byteIndex == 0 && tail < 8);
204 while (tail--) {
205 SHA3_TRACE("Store byte %02x '%c'", *buf, *buf);
206 ctx->saved |= (uint64_t)(*(buf++)) << ((ctx->byteIndex++) * 8);
207 }
208 SHA3_ASSERT(ctx->byteIndex < 8);
209 SHA3_TRACE("Have saved=0x%016" PRIx64 " at the end", ctx->saved);
210}

212/* This is simply the 'update' with the padding block.
213 * The padding block is 0x01 || 0x00* || 0x80. First 0x01 and last 0x80
214 * bytes are always present, but they can be the same byte.
215 */
216void const *sha3_Finalize(void *priv) {
217 sha3_context *ctx = (sha3_context *)priv;
218
219 SHA3_TRACE("called with %d bytes in the buffer", ctx->byteIndex);
220
221 /* Append 2-bit suffix 01, per SHA-3 spec. Instead of 1 for padding we
222 * use 1<<2 below. The 0x02 below corresponds to the suffix 01.
223 * Overall, we feed 0, then 1, and finally 1 to start padding. Without
224 * M || 01, we would simply use 1 to start padding. */
225
226 uint64_t t;
227
228 if (ctx->capacityWords & SHA3_USE_KECCAK_FLAG) {
229 /* Keccak version */
230 t = (uint64_t)(((uint64_t)1) << (ctx->byteIndex * 8)); // NOLINT
231 } else {
232 /* SHA3 version */
233 t = (uint64_t)(((uint64_t)(0x02 | (1 << 2))) // NOLINT
234 << ((ctx->byteIndex) * 8)); // NOLINT
235 }
236
237 ctx->s[ctx->wordIndex] ^= ctx->saved ^ t;
238
239 ctx->s[SHA3_KECCAK_SPONGE_WORDS - SHA3_CW(ctx->capacityWords) - 1] ^=
240 SHA3_CONST(0x8000000000000000UL);
241 keccakf(ctx->s);
242
243 /* Return first bytes of the ctx->s. This conversion is not needed for
244 * little-endian platforms e.g. wrap with #if !defined(__BYTE_ORDER__)
245 * || !defined(__ORDER_LITTLE_ENDIAN__) ||
246 * __BYTE_ORDER__!=__ORDER_LITTLE_ENDIAN__
247 * ... the conversion below ...
248 * #endif */
249 {
250 unsigned i;
251 for (i = 0; i < SHA3_KECCAK_SPONGE_WORDS; i++) { // NOLINT
252 const unsigned t1 = (uint32_t)ctx->s[i];
253 const unsigned t2 = (uint32_t)((ctx->s[i] >> 16) >> 16);
254 ctx->sb[i * 8 + 0] = (uint8_t)(t1);
255 ctx->sb[i * 8 + 1] = (uint8_t)(t1 >> 8);
256 ctx->sb[i * 8 + 2] = (uint8_t)(t1 >> 16);
257 ctx->sb[i * 8 + 3] = (uint8_t)(t1 >> 24);
258 ctx->sb[i * 8 + 4] = (uint8_t)(t2);
259 ctx->sb[i * 8 + 5] = (uint8_t)(t2 >> 8);
260 ctx->sb[i * 8 + 6] = (uint8_t)(t2 >> 16);
261 ctx->sb[i * 8 + 7] = (uint8_t)(t2 >> 24);
262 }
263 }
264
265 SHA3_TRACE_BUF("Hash: (first 32 bytes)", ctx->sb, 256 / 8);
266
267 return (ctx->sb);
268}

270sha3_return_t sha3_HashBuffer(unsigned bitSize,
271 enum SHA3_FLAGS flags,
272 const void *in,
273 unsigned inBytes,
274 void *out,
275 unsigned outBytes) {
276 sha3_return_t err;
277 sha3_context c;
278
279 err = sha3_Init(&c, bitSize);
280 if (err != SHA3_RETURN_OK) return err;
281 if (sha3_SetFlags(&c, flags) != flags) {
282 return SHA3_RETURN_BAD_PARAMS;
283 }
284 sha3_Update(&c, in, inBytes);
285 const void *h = sha3_Finalize(&c);
286
287 if (outBytes > bitSize / 8) outBytes = bitSize / 8;
288 memcpy(out, h, outBytes);
289 return SHA3_RETURN_OK;
290}
