104 outcome::result<void> TopperTrieBatchImpl::writeBack() {
105 if (auto p = parent_.lock(); p != nullptr) {
106 auto it = cache_.begin();
107 for (auto &prefix : cleared_prefixes_) {
108 BOOST_OUTCOME_TRYV2(auto &&, p->clearPrefix(prefix));
109 }
110 for (; it != cache_.end(); it++) {
111 if (it->second.has_value()) {
112 BOOST_OUTCOME_TRYV2(auto &&, p->put(it->first, it->second.value()));
113 } else {
114 BOOST_OUTCOME_TRYV2(auto &&, p->remove(it->first));
115 }
116 }
117 return outcome::success();
118 }
119 return Error::PARENT_EXPIRED;

122 bool TopperTrieBatchImpl::wasClearedByPrefix(const Buffer &key) const {
123 for (auto prefix : cleared_prefixes_) {
124 auto key_end = key.begin();
125 std::advance(key_end, std::min(key.size(), prefix.size()) - 1);
126 auto is_cleared = std::equal(key.begin(), key_end, prefix.begin());
127 if (is_cleared) return true;
128 }
129 return false;
