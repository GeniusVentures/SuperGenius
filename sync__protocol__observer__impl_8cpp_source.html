31 outcome::result<network::BlocksResponse>
32 SyncProtocolObserverImpl::onBlocksRequest(
33 const BlocksRequest &request) const {
34 if (!requested_ids_.emplace(request.id).second) {
35 return Error::DUPLICATE_REQUEST_ID;
36 }
37
38 BlocksResponse response{request.id};
39
40 // firstly, check if we have both "from" & "to" blocks (if set)
41 auto from_hash_res = blocks_headers_->getHashById(request.from);
42 if (!from_hash_res) {
43 log_->warn("cannot find a requested block with id {}", request.from);
44 requested_ids_.erase(request.id);
45 return response;
46 }
47
48 // secondly, retrieve hashes of blocks the other peer is interested in
49 auto chain_hash_res =
50 retrieveRequestedHashes(request, from_hash_res.value());
51 if (!chain_hash_res) {
52 log_->warn("cannot retrieve a chain of blocks: {}",
53 chain_hash_res.error().message());
54 requested_ids_.erase(request.id);
55 return response;
56 }
57
58 // thirdly, fill the resulting response with data, which we were asked for
59 fillBlocksResponse(request, response, chain_hash_res.value());
60 if (! response.blocks.empty()) {
61 log_->debug("Return response: {}", response.blocks[0].hash.toHex());
62 }
63
64 requested_ids_.erase(request.id);
65 return response;
66 }

68 blockchain::BlockTree::BlockHashVecRes
69 SyncProtocolObserverImpl::retrieveRequestedHashes(
70 const BlocksRequest &request,
71 const primitives::BlockHash &from_hash) const {
72 auto ascending_direction =
73 request.direction == network::Direction::ASCENDING;
74 blockchain::BlockTree::BlockHashVecRes chain_hash_res{{}};
75 if (!request.to) {
76 // if there's no "stop" block, get as many as possible
77 chain_hash_res = block_tree_->getChainByBlock(
78 from_hash, ascending_direction, maxRequestBlocks);
79 } else {
80 // else, both blocks are specified
81 OUTCOME_TRY((auto &&, chain_hash),
82 block_tree_->getChainByBlocks(from_hash, *request.to));
83 if (ascending_direction) {
84 std::reverse(chain_hash.begin(), chain_hash.end());
85 }
86 chain_hash_res = chain_hash;
87 }
88 return chain_hash_res;
89 }

91 void SyncProtocolObserverImpl::fillBlocksResponse(
92 const BlocksRequest &request,
93 BlocksResponse &response,
94 const std::vector<primitives::BlockHash> &hash_chain) const {
95 auto header_needed =
96 request.attributeIsSet(network::BlockAttributesBits::HEADER);
97 auto body_needed =
98 request.attributeIsSet(network::BlockAttributesBits::BODY);
99 auto justification_needed =
100 request.attributeIsSet(network::BlockAttributesBits::JUSTIFICATION);
101
102 for (const auto &hash : hash_chain) {
103 auto &new_block =
104 response.blocks.emplace_back(primitives::BlockData{hash});
105
106 if (header_needed) {
107 auto header_res = blocks_headers_->getBlockHeader(hash);
108 if (header_res) {
109 new_block.header = std::move(header_res.value());
110 }
111 }
112 if (body_needed) {
113 auto body_res = block_tree_->getBlockBody(hash);
114 if (body_res) {
115 new_block.body = std::move(body_res.value());
116 }
117 }
118 if (justification_needed) {
119 auto justification_res = block_tree_->getBlockJustification(hash);
120 if (justification_res) {
121 new_block.justification = std::move(justification_res.value());
122 }
123 }
124 }
125 }
