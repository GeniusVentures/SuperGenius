66 void WsListenerImpl::acceptOnce() {
67 new_session_ = std::make_shared<SessionImpl>(
68 *context_, session_config_, next_session_id_.fetch_add(1ull));
69 auto on_accept = [wp = weak_from_this()](boost::system::error_code ec) {
70 if (auto self = wp.lock()) {
71 if (! ec) {
72 if (self->on_new_session_) {
73 (*self->on_new_session_)(self->new_session_);
74 }
75 self->new_session_->start();
76 }
77
78 if (self->acceptor_->is_open()) {
79 // continue to accept until acceptor is ready
80 self->acceptOnce();
81 }
82 }
83 };
84
85 acceptor_->async_accept(new_session_->socket(), std::move(on_accept));
