70 bool AppConfigurationImpl::load_u16(const rapidjson::Value &val,
71 char const *name,
72 uint16_t &target) {
73 auto m = val.FindMember(name);
74 if (val.MemberEnd() != m && m->value.IsInt()) {
75 const auto v = m->value.GetInt();
76 const auto in_range = (v & ~std::numeric_limits<uint16_t>::max()) == 0;
77
78 if (in_range) {
79 target = static_cast<uint16_t>(v);
80 return true;
81 }
82 }
83 return false;

116 void AppConfigurationImpl::validate_config(
117 AppConfiguration::LoadScheme scheme) {
118 if (genesis_path_.empty()) {
119 logger_->error("Node configuration must contain 'genesis' option.");
120 exit(EXIT_FAILURE);
121 }
122
123 if (rocksdb_path_.empty()) {
124 logger_->error("Node configuration must contain 'rocksdb_path' option.");
125 exit(EXIT_FAILURE);
126 }
127
128 if (p2p_port_ == 0) {
129 logger_->error("p2p port is 0.");
130 exit(EXIT_FAILURE);
131 }
132
133 if (rpc_ws_port_ == 0) {
134 logger_->error("RPC ws port is 0.");
135 exit(EXIT_FAILURE);
136 }
137
138 if (rpc_http_port_ == 0) {
139 logger_->error("RPC http port is 0.");
140 exit(EXIT_FAILURE);
141 }
142
143 const auto need_keystore =
144 (AppConfiguration::LoadScheme::kBlockProducing == scheme)
145 || (AppConfiguration::LoadScheme::kValidating == scheme);
146
147 if (need_keystore && keystore_path_.empty()) {
148 logger_->error("Node configuration must contain 'keystore_path' option.");
149 exit(EXIT_FAILURE);
150 }

153 void AppConfigurationImpl::read_config_from_file(
154 const std::string &filepath) {
155 assert(!filepath.empty());
156
157 auto file = open_file(filepath);
158 if (!file) {
159 logger_->error("Configuration file path is invalid: {}", filepath);
160 return;
161 }
162
163 using FileReadStream = rapidjson::FileReadStream;
164 using Document = rapidjson::Document;
165
166 std::array<char, 1024> buffer_size{};
167 FileReadStream input_stream(
168 file.get(), buffer_size.data(), buffer_size.size());
169
170 Document document;
171 document.ParseStream(input_stream);
172 if (document.HasParseError()) {
173 logger_->error("Configuration file {} parse failed, with error {}",
174 filepath,
175 document.GetParseError());
176 return;
177 }
178
179 for (auto &handler : handlers) {
180 auto it = document.FindMember(handler.segment_name);
181 if (document.MemberEnd() != it) {
182 (this->*handler.handler)(it->value);
183 }
184 }
185 }

187 boost::asio::ip::tcp::endpoint AppConfigurationImpl::get_endpoint_from(
188 std::string const &host, uint16_t port) {
189 boost::asio::ip::tcp::endpoint endpoint;
190 boost::system::error_code err;
191
192 endpoint.address(boost::asio::ip::address::from_string(host, err));
193 if (err.failed()) {
194 logger_->error("RPC address '{}' is invalid", host);
195 exit(EXIT_FAILURE);
196 }
197
198 endpoint.port(port);
199 return endpoint;
