50 bool FinalityImpl::prepare() {
51 // Lambda which is executed when voting round is completed.
52 environment_->doOnCompleted(
53 [wp = weak_from_this()](
54 outcome::result<CompletedRound> completed_round_res) {
55 if (auto self = wp.lock()) {
56 self->onCompletedRound(std::move(completed_round_res));
57 }
58 });
59 return true;

62 bool FinalityImpl::start() {
63 // Obtain last completed round
64 auto last_round_res = getLastCompletedRound();
65 if (! last_round_res.has_value()) {
66 // No saved data
67 if (last_round_res
68 != outcome::failure(storage::DatabaseError::NOT_FOUND)) {
69 logger_->critical(
70 "Can't retrieve last round data: {}. Stopping finality execution",
71 last_round_res.error().message());
72 return false;
73 }
74 }
75
76 auto &[last_round_number, last_round_state] = last_round_res.value();
77
78 logger_->debug("Finality is starting with round #{}", last_round_number + 1);
79
80 current_round_ = makeInitialRound(last_round_number, last_round_state);
81 if (current_round_ == nullptr) {
82 logger_->critical(
83 "Next round hasn't been made. Stopping finality execution");
84 return false;
85 }
86
87 // Planning play round
88 boost::asio::post(*io_context_, [wp = current_round_->weak_from_this()] {
89 if (auto round = wp.lock()) {
90 round->play();
91 }
92 });
93
94 return true;
95 }

99 std::shared_ptr<VotingRound> FinalityImpl::makeInitialRound(
100 RoundNumber previous_round_number,
101 std::shared_ptr<const RoundState> previous_round_state) {
102 if (previous_round_state == nullptr) {
103 logger_->critical(
104 "Can't make initial round: previous round state is null");
105 return {};
106 }
107
108 // Obtain finality voters
109 auto voters_res = getVoters();
110 if (! voters_res.has_value()) {
111 logger_->critical("Can't retrieve voters: {}. Stopping finality execution",
112 voters_res.error().message());
113 return {};
114 }
115 const auto &voters = voters_res.value();
116 if (voters->empty()) {
117 logger_->critical("Voters are empty. Stopping finality execution");
118 return {};
119 }
120
121 const auto new_round_number = previous_round_number + 1;
122
123 auto vote_graph = std::make_shared<VoteGraphImpl>(
124 previous_round_state->finalized.value_or(
125 previous_round_state->last_finalized_block),
126 environment_);
127
128 FinalityConfig config{
129 /*.voters = */voters,
130 /*.round_number =*/ new_round_number,
131 /*.duration = */std::chrono::milliseconds(
132 333), // Note: Duration value was gotten from substrate:
133 // https://github.com/paritytech/substrate/blob/efbac7be80c6e8988a25339061078d3e300f132d/bin/node-template/node/src/service.rs#L166
134 /*.peer_id = */keypair_.public_key};
135
136 auto vote_crypto_provider = std::make_shared<VoteCryptoProviderImpl>(
137 keypair_, crypto_provider_, new_round_number, voters);
138
139 auto new_round = std::make_shared<VotingRoundImpl>(
140 shared_from_this(),
141 std::move(config),
142 environment_,
143 std::move(vote_crypto_provider),
144 std::make_shared<VoteTrackerImpl>(), // Prevote tracker
145 std::make_shared<VoteTrackerImpl>(), // Precommit tracker
146 std::move(vote_graph),
147 clock_,
148 io_context_,
149 std::move(previous_round_state));
150
151 return new_round;
152 }

203 std::shared_ptr<VotingRound> FinalityImpl::selectRound(
204 RoundNumber round_number) {
205 std::shared_ptr<VotingRound> target_round;
206 if (current_round_ && current_round_->roundNumber() == round_number) {
207 target_round = current_round_;
208 } else if (previous_round_
209 && previous_round_->roundNumber() == round_number) {
210 target_round = previous_round_;
211 }

229 outcome::result<CompletedRound> FinalityImpl::getLastCompletedRound() const {
230 auto last_round_encoded_res = storage_->get(storage::kSetStateKey);
231
232 // Saved data exists
233 if (last_round_encoded_res.has_value()) {
234 return scale::decode<CompletedRound>(last_round_encoded_res.value());
235 }
236
237 // Fail at retrieve data
238 if (last_round_encoded_res
239 != outcome::failure(storage::DatabaseError::NOT_FOUND)) {
240 logger_->critical(
241 "Can't retrieve last round data: {}. Stopping finality execution",
242 last_round_encoded_res.error().message());
243 return last_round_encoded_res.as_failure();
244 }
245
246 // No saved data - make from genesis
247 auto genesis_hash_res = storage_->get(storage::kGenesisBlockHashLookupKey);
248 if (! genesis_hash_res.has_value()) {
249 logger_->critical(
250 "Can't retrieve genesis block hash: {}. Stopping finality execution",
251 genesis_hash_res.error().message());
252 return genesis_hash_res.as_failure();
253 }
254
255 primitives::BlockHash genesis_hash;
256 std::copy(genesis_hash_res.value().begin(),
257 genesis_hash_res.value().end(),
258 genesis_hash.begin());
259
260 auto round_state = std::make_shared<const RoundState>(RoundState{
261 /*.last_finalized_block =*/ primitives::BlockInfo(1, genesis_hash),
262 /*.best_prevote_candidate =*/ Prevote(1, genesis_hash),
263 /*.best_final_candidate =*/ primitives::BlockInfo(1, genesis_hash),
264 /*.finalized =*/ primitives::BlockInfo(1, genesis_hash)});
265
266 CompletedRound zero_round{/*.round_number = */0, /*.state = */round_state};
267
268 return std::move(zero_round);
269 }

271 void FinalityImpl::executeNextRound() {
272 logger_->debug("Execute next round #{} -> #{}",
273 current_round_->roundNumber(),
274 current_round_->roundNumber() + 1);
275 previous_round_.swap(current_round_);
276 previous_round_->end();
277 current_round_ = makeNextRound(previous_round_);
278 current_round_->play();
279 }

281 void FinalityImpl::onVoteMessage(const VoteMessage &msg) {
282 std::shared_ptr<VotingRound> target_round = selectRound(msg.round_number);
283 if (! target_round) return;
284
285 // get block info
286 auto blockInfo = visit_in_place(msg.vote.message, [](const auto &vote) {
287 return BlockInfo(vote.block_number, vote.block_hash);
288 });
289
290 // get authorities
291 const auto &weighted_authorities_res =
292 finality_api_->authorities(primitives::BlockId(blockInfo.block_number));
293 if (!weighted_authorities_res.has_value()) {
294 logger_->error("Can't get authorities");
295 return;
296 };
297 auto &weighted_authorities = weighted_authorities_res.value();
298
299 // find signer in authorities
300 auto weighted_authority_it =
301 std::find_if(weighted_authorities.begin(),
302 weighted_authorities.end(),
303 [&id = msg.vote.id](const auto &weighted_authority) {
304 return weighted_authority.id.id == id;
305 });
306
307 if (weighted_authority_it == weighted_authorities.end()) {
308 logger_->warn("Vote signed by unknown validator");
309 return;
310 };
311
312 visit_in_place(
313 msg.vote.message,
314 [&target_round, &msg](const PrimaryPropose &primary_propose) {
315 target_round->onPrimaryPropose(msg.vote);
316 },
317 [&target_round, &msg](const Prevote &prevote) {
318 target_round->onPrevote(msg.vote);
319 },
320 [&target_round, &msg](const Precommit &precommit) {
321 target_round->onPrecommit(msg.vote);
322 });
323 }

334 void FinalityImpl::onCompletedRound(
335 outcome::result<CompletedRound> completed_round_res) {
336 round_id++;
337
338 if (! completed_round_res) {
339 logger_->debug("Finality round was not finalized: {}",
340 completed_round_res.error().message());
341 } else {
342 const auto &completed_round = completed_round_res.value();
343
344 logger_->debug("Event OnCompleted for round #{}",
345 completed_round.round_number);
346
347 if (auto put_res = storage_->put(
348 storage::kSetStateKey,
349 base::Buffer(scale::encode(completed_round).value()));
350 ! put_res) {
351 logger_->error("New round state was not added to the storage");
352 return;
353 }
354
355 BOOST_ASSERT(storage_->get(storage::kSetStateKey));
356 }
357
358 boost::asio::post(*io_context_, [wp = weak_from_this()] {
359 if (auto finality = wp.lock()) {
360 finality->executeNextRound();
361 }
362 });
363 }
