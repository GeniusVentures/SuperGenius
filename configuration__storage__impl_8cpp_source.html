47 outcome::result<void> ConfigurationStorageImpl::loadGenesis(
48 const boost::property_tree::ptree &tree) {
49 OUTCOME_TRY((auto &&, genesis_tree), ensure(tree.get_child_optional("genesis")));
50 OUTCOME_TRY((auto &&, genesis_raw_tree),
51 ensure(genesis_tree.get_child_optional("raw")));
52 boost::property_tree::ptree top_tree;
53 // v0.7 format
54 if(auto top_tree_opt = genesis_raw_tree.get_child_optional("top"); top_tree_opt.has_value()) {
55 top_tree = top_tree_opt.value();
56 } else {
57 // Try to fall back to v0.6
58 top_tree = genesis_raw_tree.begin()->second;
59 }
60
61 for (const auto &[key, value] : top_tree) {
62 // get rid of leading 0x for key and value and unhex
63 OUTCOME_TRY((auto &&, key_processed), base::unhexWith0x(key));
64 OUTCOME_TRY((auto &&, value_processed), base::unhexWith0x(value.data()));
65 genesis_.emplace_back(key_processed, value_processed);
66 }
67 // ignore child storages as they are not yet implemented
68
69 return outcome::success();
70 }

72 outcome::result<void> ConfigurationStorageImpl::loadBootNodes(
73 const boost::property_tree::ptree &tree) {
74 OUTCOME_TRY((auto &&, boot_nodes), ensure(tree.get_child_optional("bootNodes")));
75 for (auto &v : boot_nodes) {
76 OUTCOME_TRY((auto &&, multiaddr),
77 libp2p::multi::Multiaddress::create(v.second.data()));
78 OUTCOME_TRY((auto &&, peer_id_base58), ensure(multiaddr.getPeerId()));
79
80 OUTCOME_TRY((auto &&, peer_id), libp2p::peer::PeerId::fromBase58(peer_id_base58));
81 libp2p::peer::PeerInfo info{/*.id =*/ std::move(peer_id),
82 /*.addresses =*/ {std::move(multiaddr)}};
83 boot_nodes_.peers.push_back(info);
84 }
85 return outcome::success();
