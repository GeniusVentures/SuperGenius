51 outcome::result<void> StorageChangesTrackerImpl::onPut(
52 const base::Buffer &key,
53 const base::Buffer &value,
54 bool is_new_entry) {
55 auto change_it = extrinsics_changes_.find(key);
56 OUTCOME_TRY((auto &&, idx_bytes), get_extrinsic_index_());
57 OUTCOME_TRY((auto &&, idx), scale::decode<primitives::ExtrinsicIndex>(idx_bytes));
58
59 // if key was already changed in the same block, just add extrinsic to
60 // the changers list
61 if (change_it != extrinsics_changes_.end()) {
62 change_it->second.push_back(idx);
63 } else {
64 extrinsics_changes_.insert(std::make_pair(key, std::vector{idx}));
65 if (is_new_entry) {
66 new_entries_.insert(key);
67 }
68 }
69 subscription_engine_->notify(key, value, parent_hash_);
70 return outcome::success();
71 }

73 outcome::result<void> StorageChangesTrackerImpl::onRemove(
74 const base::Buffer &key) {
75 auto change_it = extrinsics_changes_.find(key);
76 OUTCOME_TRY((auto &&, idx_bytes), get_extrinsic_index_());
77 OUTCOME_TRY((auto &&, idx), scale::decode<primitives::ExtrinsicIndex>(idx_bytes));
78
79 // if key was already changed in the same block, just add extrinsic to
80 // the changers list
81 if (change_it != extrinsics_changes_.end()) {
82 // if new entry, i. e. it doesn't exist in the persistent storage, then
83 // don't track it, because it's just temporary
84 if (auto i = new_entries_.find(key); i != new_entries_.end()) {
85 extrinsics_changes_.erase(change_it);
86 new_entries_.erase(i);
87 } else {
88 change_it->second.push_back(idx);
89 }
90
91 } else {
92 extrinsics_changes_.insert(std::make_pair(key, std::vector{idx}));
93 }
94 return outcome::success();
95 }
