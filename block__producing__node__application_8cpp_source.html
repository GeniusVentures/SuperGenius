26 void BlockProducingNodeApplication::run() {

28
29 production_->setExecutionStrategy(Production::ExecutionStrategy::SYNC_FIRST);
30
31 app_state_manager_->atLaunch([this] {
32 // execute listeners
33 io_context_->post([this] {
34 const auto &current_peer_info =
35 injector_.template create<network::OwnPeerInfo>();
36 auto &host = injector_.template create<libp2p::Host &>();
37 for (const /*auto*/ libp2p::multi::Multiaddress &ma : current_peer_info.addresses) {
38 /*auto*/outcome::result<void> listen = host.listen(ma);
39 if (! listen) {
40 logger_->error("Cannot listen address {}. Error: {}",
41 ma.getStringAddress(),
42 listen.error().message());
43 std::exit(1);
44 }
45 }
46 this->router_->init();
47 });
48 return true;
