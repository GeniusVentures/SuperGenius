31 outcome::result<base::Hash256> AuthorApiImpl::submitExtrinsic(
32 const primitives::Extrinsic &extrinsic) {
33 OUTCOME_TRY((auto &&, res), api_->validate_transaction(extrinsic));
34
35 return visit_in_place(
36 res,
37 [&](const primitives::TransactionValidityError &e) {
38 return visit_in_place(
39 e,
40 // return either invalid or unknown validity error
41 [](const auto &validity_error)
42 -> outcome::result<base::Hash256> {
43 return validity_error;
44 });
45 },
46 [&](const primitives::ValidTransaction &v)
47 -> outcome::result<base::Hash256> {
48 // compose Transaction object
49 base::Hash256 hash = hasher_->blake2b_256(extrinsic.data);
50 size_t length = extrinsic.data.size();
51
52 primitives::Transaction transaction{extrinsic,
53 length,
54 hash,
55 v.priority,
56 v.longevity,
57 v.requires,
58 v.provides,
59 v.propagate};
60
61 // send to pool
62 BOOST_OUTCOME_TRYV2(auto &&, pool_->submitOne(std::move(transaction)));
63
64 if (v.propagate) {
65 network::TransactionAnnounce announce;
66 announce.extrinsics.push_back(extrinsic);
67 gossiper_->transactionAnnounce(announce);
68 }
69
70 return hash;
71 });
72 }
