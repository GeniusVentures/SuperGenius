77 outcome::result<RuntimeEnvironment> RuntimeManager::createRuntimeEnvironment(
78 const base::Buffer &state_code) {
79 if (state_code.empty()) {
80 return Error::EMPTY_STATE_CODE;
81 }
82
83 auto hash = hasher_->twox_256(state_code);
84
85 std::shared_ptr<WasmModule> module;
86
87 // Trying retrieve pre-prepared module
88 {
89 std::lock_guard lockGuard(modules_mutex_);
90 auto it = modules_.find(hash);
91 if (it != modules_.end()) {
92 module = it->second;
93 }
94 }
95
96 if (external_interface_ == nullptr) {
97 external_interface_ = std::make_shared<RuntimeExternalInterface>(
98 extension_factory_, storage_provider_);
99 }
100
101 if (!module) {
102 // Prepare new module
103 OUTCOME_TRY((auto &&,
104 new_module),
105 module_factory_->createModule(state_code, external_interface_));
106
107 // Trying to safe emplace new module, and use existed one
108 // if it already emplaced in another thread
109 std::lock_guard lockGuard(modules_mutex_);
110 module = modules_.emplace(hash, std::move(new_module)).first->second;
111 }
112
113 return RuntimeEnvironment::create(
114 external_interface_, module, state_code);
115 }
