61 void WsSession::onRun() {
62 // Set suggested timeout settings for the websocket
63 stream_.set_option(boost::beast::websocket::stream_base::timeout::suggested(
64 boost::beast::role_type::server));
65
66 // Set a decorator to change the Server of the handshake
67 stream_.set_option(boost::beast::websocket::stream_base::decorator(
68 [](boost::beast::websocket::response_type &res) {
69 res.set(boost::beast::http::field::server,
70 std::string(BOOST_BEAST_VERSION_STRING)
71 + " websocket-server-async");
72 }));
73 // Accept the websocket handshake
74 stream_.async_accept(boost::beast::bind_front_handler(&WsSession::onAccept,
75 shared_from_this()));
