50 outcome::result<void> SuperGeniusTrieCursor::seek(const base::Buffer &key) {
51 if (trie_.getRoot() == nullptr) {
52 current_ = nullptr;
53 return Error::NULL_ROOT;
54 }
55 visited_root_ = true; // root is always visited first
56 OUTCOME_TRY((auto &&, last_child_path), constructLastVisitedChildPath(key));
57 auto nibbles = SuperGeniusCodec::keyToNibbles(key);
58 OUTCOME_TRY((auto &&, node), trie_.getNode(trie_.getRoot(), nibbles));
59
60 bool node_has_value = node != nullptr && node->value.has_value();
61 if (node_has_value) {
62 current_ = node;
63 } else {
64 current_ = nullptr;
65 return Error::NOT_FOUND;
66 }
67 last_visited_child_ = std::move(last_child_path);
68 return outcome::success();
69 }

71 outcome::result<void> SuperGeniusTrieCursor::seekToLast() {
72 NodePtr current = trie_.getRoot();
73 if (current == nullptr) {
74 current_ = nullptr;
75 return Error::NULL_ROOT;
76 }
77 visited_root_ = true; // root is always visited first
78 last_visited_child_.clear();
79 // find the rightmost leaf
80 while (current->getTrieType() != NodeType::Leaf) {
81 auto type = current->getTrieType();
82 if (type == NodeType::BranchEmptyValue
83 || type == NodeType::BranchWithValue) {
84 auto branch = std::dynamic_pointer_cast<BranchNode>(current);
85 // find the rightmost child
86 for (int8_t i = branch->kMaxChildren - 1; i >= 0; i--) {
87 if (branch->children.at(i) != nullptr) {
88 OUTCOME_TRY((auto &&, c), trie_.retrieveChild(branch, i));
89 last_visited_child_.emplace_back(branch, i);
90 current = c;
91 }
92 }
93
94 } else {
95 BOOST_ASSERT_MSG(
96 false,
97 "must not reach here, as with using retrieveChild there should "
98 "appear no other trie node type than leaf and both branch types");
99 }
100 }
101 current_ = current;
102 return outcome::success();
103 }

109 outcome::result<void> SuperGeniusTrieCursor::next() {
110 if (!visited_root_) {
111 current_ = trie_.getRoot();
112 }
113 if (current_ == nullptr) {
114 return trie_.getRoot() == nullptr ? Error::NULL_ROOT
115 : Error::INVALID_CURSOR_POSITION;
116 }
117 if (!visited_root_ && trie_.getRoot()->value.has_value()) {
118 visited_root_ = true;
119 return outcome::success();
120 }
121 visited_root_ = true;
122 do {
123 if (current_->getTrieType() == NodeType::Leaf) {
124 if (last_visited_child_.empty()) {
125 current_ = nullptr;
126 return outcome::success();
127 }
128 // assert last_visited_child_.back() == current.parent
129 auto p = last_visited_child_.back().parent; // self.current.parent
130 while (!hasNextChild(p, last_visited_child_.back().child_idx)) {
131 last_visited_child_.pop_back();
132 if (last_visited_child_.empty()) {
133 current_ = nullptr;
134 return outcome::success();
135 }
136 p = last_visited_child_.back().parent; // p.parent
137 }
138 auto i = getNextChildIdx(p, last_visited_child_.back().child_idx);
139 OUTCOME_TRY((auto &&, c), trie_.retrieveChild(p, i));
140 current_ = c;
141 updateLastVisitedChild(p, i);
142
143 } else if (current_->getTrieType() == NodeType::BranchEmptyValue
144 || current_->getTrieType() == NodeType::BranchWithValue) {
145 auto p = std::dynamic_pointer_cast<BranchNode>(current_);
146 if (last_visited_child_.empty()
147 || last_visited_child_.back().parent != p) {
148 last_visited_child_.emplace_back(p, -1);
149 }
150 while (!hasNextChild(p, last_visited_child_.back().child_idx)) {
151 last_visited_child_.pop_back();
152 if (last_visited_child_.empty()) {
153 current_ = nullptr;
154 return outcome::success();
155 }
156 p = last_visited_child_.back().parent; // p.parent
157 }
158 auto i = getNextChildIdx(p, last_visited_child_.back().child_idx);
159 OUTCOME_TRY((auto &&, c), trie_.retrieveChild(p, i));
160 current_ = c;
161 updateLastVisitedChild(p, i);
162 }
163 } while (! current_->value.has_value());
164 return outcome::success();
165 }

172 base::Buffer SuperGeniusTrieCursor::collectKey() const {
173 KeyNibbles key_nibbles;
174 // std::back_insert_iterator<Buffer> temp(key_nibbles);
175
176 for (auto &node_idx : last_visited_child_) {
177 auto &node = node_idx.parent;
178 auto idx = node_idx.child_idx;
179 std::copy(node->key_nibbles.begin(),
180 node->key_nibbles.end(),
181 std::back_inserter<Buffer>(key_nibbles));
182 key_nibbles.putUint8(idx);
183 }
184 key_nibbles.put(current_->key_nibbles);
185 return codec_.nibblesToKey(key_nibbles);
186 }
