65 void GossiperBroadcast::broadcast(GossipMessage &&msg) {
66 auto msg_send_lambda = [msg, this](auto stream) {
67 auto read_writer =
68 std::make_shared<ScaleMessageReadWriter>(std::move(stream));
69 read_writer->write(msg, [this](auto &&res) {
70 if (! res) {
71 logger_->error("Could not broadcast, reason: {}",
72 res.error().message());
73 }
74 });
75 };
76
77 // iterate over the existing streams and send them the msg. If stream is
78 // closed it is removed
79 auto stream_it = syncing_streams_.begin();
80 while (stream_it != syncing_streams_.end()) {
81 auto stream = *stream_it;
82 if (stream && !stream->isClosed()) {
83 msg_send_lambda(stream);
84 stream_it++;
85 } else {
86 // remove this stream
87 stream_it = syncing_streams_.erase(stream_it);
88 }
89 }
90 for (const auto &[info, stream] : streams_) {
91 if (stream && !stream->isClosed()) {
92 msg_send_lambda(stream);
93 continue;
94 }
95 // if stream does not exist or expired, open a new one
96 host_.newStream(info,
97 kGossipProtocol,
98 [self{shared_from_this()}, info = info, msg_send_lambda](
99 auto &&stream_res) mutable {
100 if (!stream_res) {
101 // we will try to open the stream again, when
102 // another gossip message arrives later
103 self->logger_->error(
104 "Could not send message to {} Error: {}",
105 info.id.toBase58(),
106 stream_res.error().message());
107 return;
108 }
109
110 // save the stream and send the message
111 self->streams_[info] = stream_res.value();
112 msg_send_lambda(std::move(stream_res.value()));
113 });
114 }
115 }
