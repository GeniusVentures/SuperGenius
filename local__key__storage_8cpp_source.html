47 outcome::result<void> LocalKeyStorage::loadSR25519Keys(
48 const boost::property_tree::ptree &tree) {
49 OUTCOME_TRY((auto &&, sr_tree), ensure(tree.get_child_optional("sr25519keypair")));
50
51 OUTCOME_TRY((auto &&, sr_pubkey_str),
52 ensure(sr_tree.get_optional<std::string>("public")));

54 ensure(sr_tree.get_optional<std::string>("private")));
55
56 // get rid of 0x from beginning

95 outcome::result<void> LocalKeyStorage::loadP2PKeys(

97 auto p2p_tree_opt = tree.get_child_optional("p2p_keypair");
98 if (! p2p_tree_opt) return ConfigReaderError::MISSING_ENTRY;
99 const auto &p2p_tree = p2p_tree_opt.value();
100
101 OUTCOME_TRY((auto &&, p2p_type),
102 ensure(p2p_tree.get_optional<std::string>("p2p_type")));
103
104 using KeyType = libp2p::crypto::Key::Type;
105 if (p2p_type == "ed25519") {
106 p2p_keypair_.publicKey.type = KeyType::Ed25519;
107 p2p_keypair_.privateKey.type = KeyType::Ed25519;
108 } else if (p2p_type == "rsa") {
109 p2p_keypair_.publicKey.type = KeyType::RSA;
110 p2p_keypair_.privateKey.type = KeyType::RSA;
111 } else if (p2p_type == "secp256k1") {
112 p2p_keypair_.publicKey.type = KeyType::Secp256k1;
113 p2p_keypair_.privateKey.type = KeyType::Secp256k1;
114 } else if (p2p_type == "ecdsa") {
115 p2p_keypair_.publicKey.type = KeyType::ECDSA;
116 p2p_keypair_.privateKey.type = KeyType::ECDSA;
117 } else {
118 p2p_keypair_.publicKey.type = KeyType::UNSPECIFIED;
119 p2p_keypair_.privateKey.type = KeyType::UNSPECIFIED;
120 }
121
122 auto p2p_keypair_tree_opt = p2p_tree.get_child_optional("keypair");
123 if (! p2p_keypair_tree_opt) return ConfigReaderError::MISSING_ENTRY;
124 const auto &p2p_keypair_tree = p2p_keypair_tree_opt.value();
125
126 OUTCOME_TRY((auto &&, p2p_public_key_str),
127 ensure(p2p_keypair_tree.get_optional<std::string>("public")));

129 ensure(p2p_keypair_tree.get_optional<std::string>("private")));

133 OUTCOME_TRY((auto &&, p2p_private_key), base::unhexWith0x(p2p_private_key_str));
134
135 p2p_keypair_.publicKey.data = p2p_public_key;
136 p2p_keypair_.privateKey.data = p2p_private_key;
