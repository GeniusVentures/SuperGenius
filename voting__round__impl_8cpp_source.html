142 void VotingRoundImpl::play() {
143 if (stage_ != Stage::INIT) {
144 return;
145 }
146 stage_ = Stage::START;
147
148 logger_->debug("Start round #{}", round_number_);
149
150 // Current local time (Tstart)
151 start_time_ = clock_->now();
152
153 // Derive-Primary
154 // see ctor
155
156 if (isPrimary_) {
157 logger_->debug("Node is primary proposer at round #{}", round_number_);
158
159 // Broadcast Fin-message with previous round best final candidate
160 // (or last finalized otherwise)
161 // spec: Broadcast(M vr �1;Fin (Best-Final-Candidate(r-1)))
162
163 // if Best-Final-Candidate greater than Last-Finalized-Block
164 // spec: if Best-Final-Candidate(r � 1) > Last-Finalized-Block
165 if (previous_round_state_->best_final_candidate.block_number
166 > previous_round_state_->last_finalized_block.block_number) {
167 doProposal();
168 }
169 }
170
171 startPrevoteStage();

174 void VotingRoundImpl::startPrevoteStage() {
175 BOOST_ASSERT(stage_ == Stage::START);
176 stage_ = Stage::START_PREVOTE;
177
178 logger_->debug("Start prevote stage of round #{}", round_number_);
179
180 // Continue to reveive messages until T>=Tstart + 2 * Duration or round is
181 // completable
182 // spec: Receive-Messages(until Time >t r ; v + 2 ␂ T or r is
183 // completable)
184
185 if (completable()) {
186 logger_->debug("Round #{} is already completable", round_number_);
187 stage_ = Stage::PREVOTE_RUNS;
188 endPrevoteStage();
189 return;
190 }
191
192 timer_.cancel();
193 timer_.expires_at(start_time_ + duration_ * 2);
194 timer_.async_wait([this](const auto &...) {
195 if (stage_ == Stage::PREVOTE_RUNS) {
196 logger_->debug("Time of prevote stage of round #{} is out",
197 round_number_);
198 endPrevoteStage();
199 }
200 });
201
202 on_complete_handler_ = [this] {
203 if (stage_ == Stage::PREVOTE_RUNS) {
204 logger_->debug("Round #{} became completable", round_number_);
205 endPrevoteStage();
206 }
207 };
208
209 stage_ = Stage::PREVOTE_RUNS;
210 }

284 void VotingRoundImpl::startWaitingStage() {
285 BOOST_ASSERT(stage_ == Stage::END_PRECOMMIT);
286 stage_ = Stage::START_WAITING;
287
288 logger_->debug("Start final stage of round #{}", round_number_);
289
290 // Continue to receive messages until current round is completable and
291 // previous one is finalisable and last filanized better than best filan
292 // candidate of prefious round
293
294 // spec: Receive-Messages(until r is completable and Finalizable(r � 1) and
295 // Last-Finalized-Block>Best-Final-Candidate(r-1))
296
297 bool isReadyToEnd =
298 completable_ && previous_round_state_->finalized
299 && previous_round_state_->finalized.get().block_number
300 >= previous_round_state_->best_final_candidate.block_number;
301
302 if (isReadyToEnd) {
303 logger_->debug("Conditions for final stage of round #{} already met",
304 round_number_);
305 stage_ = Stage::WAITING_RUNS;
306 endWaitingStage();
307 return;
308 }
309
310 auto handle_end_condition = [this] {
311 bool isReadyToEnd =
312 completable_ && previous_round_state_->finalized
313 && previous_round_state_->finalized.get().block_number
314 >= previous_round_state_->best_final_candidate.block_number;
315
316 if (isReadyToEnd) {
317 logger_->debug("Conditions for final stage of round #{} are met",
318 round_number_);
319 endWaitingStage();
320 }
321 };
322
323 on_complete_handler_ = handle_end_condition;
324
325 stage_ = Stage::WAITING_RUNS;
326 }

407 void VotingRoundImpl::doPrecommit() {
408 auto last_round_estimate = previous_round_state_->best_final_candidate;
409
410 // We should precommit if current state contains prevote and it is
411 // either equal to the last round estimate or is descendant of it
412 bool should_precommit =
413 current_round_state_->best_prevote_candidate.block_hash
414 == last_round_estimate.block_hash
415 || env_->isEqualOrDescendOf(
416 last_round_estimate.block_hash,
417 current_round_state_->best_prevote_candidate.block_hash);
418
419 if (should_precommit) {
420 logger_->debug("Casting precommit for round {}", round_number_);
421 auto precommit = constructPrecommit(*previous_round_state_);
422 if (precommit) {
423 logger_->debug("Precommit block #{} hash={} in round #{}",
424 precommit.value().block_number(),
425 precommit.value().block_hash(),
426 round_number_);
427 auto precommitted = env_->onPrecommitted(
428 round_number_, voter_set_->id(), precommit.value());
429 if (! precommitted) {
430 logger_->error("Precommit was not sent: {}",
431 precommitted.error().message());
432 }
433 return;
434 }
435 logger_->debug("No precommit in round #{}", round_number_);
436
437 // Current implementation of constructPrecommit() isn't return any error
438 BOOST_ASSERT_MSG(false,
439 "Unreachable code; Current implementation of "
440 "constructPrecommit() should not return any error");

489 bool VotingRoundImpl::validate(
490 const BlockInfo &vote, const FinalityJustification &justification) const {
491 size_t total_weight = 0;
492
493 std::unordered_map<Id, BlockHash> validators;
494 std::unordered_set<Id> equivocators;
495
496 for (const auto &signed_precommit : justification.items) {
497 // Skip known equivocators
498 if (auto index = voter_set_->voterIndex(signed_precommit.id);
499 index.has_value()) {
500 if (precommit_equivocators_.at(index.value())) {
501 continue;
502 }
503 }
504
505 // Verify signatures
506 if (! vote_crypto_provider_->verifyPrecommit(signed_precommit)) {
507 logger_->error(
508 "Received invalid signed precommit during the round {} from the "
509 "peer {}",
510 round_number_,
511 signed_precommit.id.toHex());
512 return false;
513 }
514
515 // check that every signed precommit corresponds to the vote (i.e.
516 // signed_precommits are descendants of the vote). If so add weight of
517 // that voter to the total weight
518
519 if (auto [it, success] = validators.emplace(
520 signed_precommit.id, signed_precommit.block_hash());
521 success) {
522 // New vote
523 if (env_->getAncestry(vote.block_hash, signed_precommit.block_hash())) {
524 total_weight += voter_set_->voterWeight(signed_precommit.id).value();
525 }
526
527 } else if (equivocators.emplace(signed_precommit.id).second) {
528 // Detected equivocation
529 if (env_->getAncestry(vote.block_hash, it->second)) {
530 total_weight -= voter_set_->voterWeight(signed_precommit.id).value();
531 }
532
533 } else {
534 // Detected duplicate of equivotation
535 logger_->error(
536 "Received third precommit of caught equivocator during the round "
537 "{} from the peer {}",
538 round_number_,
539 signed_precommit.id.toHex());
540 return false;
541 }
542 }
543
544 return total_weight >= threshold_;
545 }

547 bool VotingRoundImpl::tryFinalize() {
548 if (! completable()) {
549 return false;
550 }
551 if (! previous_round_state_) {
552 logger_->error("Last round state is empty during finalization");
553 return false;

557 logger_->debug("Did not notify. Reason: {}",

559 // Round is completable but we cannot notify others. Finish the round

641 void VotingRoundImpl::onSignedPrevote(const SignedMessage &vote) {
642 BOOST_ASSERT(vote.is<Prevote>());
643 auto weight = voter_set_->voterWeight(vote.id);
644 if (! weight) {
645 return;
646 }
647 switch (prevotes_->push(vote, weight.value())) {
648 case VoteTracker::PushResult::SUCCESS: {
649 const auto &voters = voter_set_->voters();
650
651 // prepare VoteWeight which contains index of who has voted and what
652 // kind of vote it was
653 VoteWeight voteWeight{voters.size()};
654 auto index = voter_set_->voterIndex(vote.id);
655 if (! index) {
656 logger_->warn("Voter {} is not known: {}", vote.id.toHex());
657 return;
658 }
659
660 voteWeight.setPrevote(index.value(),
661 voter_set_->voterWeight(vote.id).value());
662
663 if (auto inserted = graph_->insert(vote.message, voteWeight);
664 ! inserted) {
665 logger_->warn("Vote {} was not inserted with error: {}",
666 vote.block_hash().toHex(),
667 inserted.error().message());
668 }
669 break;
670 }
671 case VoteTracker::PushResult::DUPLICATED: {
672 break;
673 }
674 case VoteTracker::PushResult::EQUIVOCATED: {
675 auto index = voter_set_->voterIndex(vote.id);
676 if (! index) {
677 logger_->warn("Voter {} is not known: {}", vote.id.toHex());
678 return;
679 }
680 prevote_equivocators_[index.value()] = true;
681 break;
682 }
683 }

765 outcome::result<SignedMessage> VotingRoundImpl::constructPrevote(
766 const RoundState &last_round_state) const {
767 const auto &last_round_estimate = last_round_state.best_final_candidate;
768
769 // find the block that we should take descendent from
770 auto find_descendent_of =
771 primary_vote_.map(convertToBlockInfo)
772 .map([&](const BlockInfo &primary) {
773 // if we have primary_vote then:
774 // if last prevote is the same as primary vote, then return it
775 // else if primary vote is better than last prevote return last
776 // round estimate
777 auto last_prevote_g = last_round_state.best_prevote_candidate;
778
779 if (std::tie(primary.block_number, primary.block_hash)
780 == std::tie(last_prevote_g.block_number,
781 last_prevote_g.block_hash)) {
782 return primary;
783 }
784 if (primary.block_number >= last_prevote_g.block_number) {
785 return last_round_estimate;
786 }
787
788 // from this point onwards, the number of the
789 // primary-broadcasted block is less than the last
790 // prevote-GHOST's number. if the primary block is in the
791 // ancestry of p-G we vote for the best chain containing it.
792 if (auto ancestry =
793 env_->getAncestry(last_round_estimate.block_hash,
794 last_prevote_g.block_hash);
795 ancestry) {
796 auto to_sub = primary.block_number + 1;
797
798 primitives::BlockNumber offset = 0;
799 if (last_prevote_g.block_number >= to_sub) {
800 offset = last_prevote_g.block_number - to_sub;
801 }
802
803 if (ancestry.value().at(offset) == primary.block_hash) {
804 return primary;
805 }
806 }
807 return last_round_estimate;
808 })
809 .value_or_eval(
810 [last_round_estimate] { return last_round_estimate; });
811
812 auto rbest_chain = env_->bestChainContaining(find_descendent_of.block_hash);
813
814 if (! rbest_chain) {
815 logger_->error(
816 "Could not cast prevote: previously known block {} has disappeared",
817 find_descendent_of.block_hash.toHex());
818 return outcome::failure(boost::system::error_code());
819 }
820 const auto &best_chain = rbest_chain.value();
821
822 return vote_crypto_provider_->signPrevote(
823 {best_chain.block_number, best_chain.block_hash});
824 }

836 void VotingRoundImpl::update() {
837 // No supermajority in prevotes
838 if (prevotes_->getTotalWeight() < threshold_) {
839 return;
840 }
841
842 const auto &prevote_ghost = current_round_state_->best_prevote_candidate;
843
844 // anything new finalized? finalized blocks are those which have both
845 // 2/3+ precommit weight.
846 auto current_precommits = precommits_->getTotalWeight();
847
848 if (current_precommits >= threshold_) {
849 current_round_state_->finalized = graph_->findAncestor(
850 BlockInfo{
851 prevote_ghost.block_number,
852 prevote_ghost.block_hash,
853 },
854 [this](const VoteWeight &vote_weight) {
855 return vote_weight
856 .totalWeight(prevote_equivocators_,
857 precommit_equivocators_,
858 voter_set_)
859 .precommit
860 >= threshold_;
861 },
862 VoteWeight::precommitComparator);
863 }
864
865 // figuring out whether a block can still be committed for is
866 // not straightforward because we have to account for all possible future
867 // equivocations and thus cannot discount weight from validators who
868 // have already voted.
869 auto tolerated_equivocations = voter_set_->totalWeight() - threshold_;
870
871 // find how many more equivocations we could still get.
872 //
873 // it is only important to consider the voters whose votes
874 // we have already seen, because we are assuming any votes we
875 // haven't seen will target this block.
876
877 // get total weight of all equivocators
878 using namespace boost::adaptors; // NOLINT
879 auto current_equivocations = boost::accumulate(
880 voter_set_->voters() | indexed(0)
881 | transformed([this](const auto &element) {
882 return precommit_equivocators_[element.index()]
883 ? voter_set_->voterWeight(element.index()).value()
884 : 0UL;
885 }),
886 0UL);
887
888 auto additional_equiv = tolerated_equivocations - current_equivocations;
889 auto possible_to_precommit = [&](const VoteWeight &weight) {
890 // total precommits for this block, including equivocations.
891 auto precommited_for =
892 weight
893 .totalWeight(
894 prevote_equivocators_, precommit_equivocators_, voter_set_)
895 .precommit;
896
897 // equivocations we could still get are out of those who
898 // have already voted, but not on this block.
899 auto possible_equivocations = std::min<size_t>(
900 current_precommits - precommited_for, additional_equiv);
901
902 auto remaining_commit_votes =
903 voter_set_->totalWeight() - precommits_->getTotalWeight();
904
905 // all the votes already applied on this block,
906 // assuming all remaining actors commit to this block,
907 // and that we get further equivocations
908 auto full_possible_weight =
909 precommited_for + remaining_commit_votes + possible_equivocations;
910 return full_possible_weight >= threshold_;
911 };
912
913 // until we have threshold precommits, any new block could get
914 // supermajority precommits because there are at least f + 1 precommits
915 // remaining and then f equivocations.
916 //
917 // once it's at least that level, we only need to consider blocks
918 // already referenced in the graph, because no new leaf nodes
919 // could ever have enough precommits.
920 //
921 // the round-estimate is the highest block in the chain with head
922 // `prevote_ghost` that could have supermajority-commits.
923 if (precommits_->getTotalWeight() >= threshold_) {
924 auto block = graph_->findAncestor(
925 BlockInfo{prevote_ghost.block_number, prevote_ghost.block_hash},
926 possible_to_precommit,
927 VoteWeight::precommitComparator);
928 if (block.has_value()) {
929 current_round_state_->best_final_candidate = block.value();
930 }
931 } else {
932 current_round_state_->best_final_candidate =
933 BlockInfo{prevote_ghost.block_number, prevote_ghost.block_hash};
934 return;
935 }
936
937 const auto &block = current_round_state_->best_final_candidate;
938 completable_ =
939 block.block_hash != prevote_ghost.block_hash
940 || graph_
941 ->findGhost(block,
942 possible_to_precommit,
943 VoteWeight::precommitComparator)
944 .map([&prevote_ghost](const BlockInfo &block_info) {
945 return std::tie(block_info.block_hash, block_info.block_number)
946 == std::tie(prevote_ghost.block_hash,
947 prevote_ghost.block_number);
948 })
949 .value_or(true);
950 }

952 boost::optional<FinalityJustification> VotingRoundImpl::finalizingPrecommits(
953 const BlockInfo &estimate) const {
954 const auto &precommits = precommits_->getMessages();
955 FinalityJustification justification = std::accumulate(
956 precommits.begin(),
957 precommits.end(),
958 FinalityJustification{},
959 [this](FinalityJustification &j, const auto &precommit_variant) {
960 visit_in_place(
961 precommit_variant,
962 [&j, this](const SignedMessage &voting_message) {
963 if (voting_message.is<Precommit>()
964 && env_->isEqualOrDescendOf(
965 current_round_state_->finalized->block_hash,
966 voting_message.block_hash())) {
967 j.items.push_back(voting_message);
968 }
969 },
970 [&](const VoteTracker::EquivocatoryVotingMessage
971 &equivocatory_voting_message) {
972 j.items.push_back(equivocatory_voting_message.first);
973 j.items.push_back(equivocatory_voting_message.second);
974 });
975 return j;
976 });
977 return justification;
978 }
