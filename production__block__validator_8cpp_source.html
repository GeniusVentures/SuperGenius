53 outcome::result<void> ProductionBlockValidator::validateBlock(
54 const primitives::Block &block,
55 const primitives::AuthorityId &authority_id,
56 const Threshold &threshold,
57 const Randomness &randomness) const {
58 BOOST_OUTCOME_TRYV2(auto &&,
59 validateHeader(block.header, authority_id, threshold, randomness));
60
61 // all transactions in the block must be valid
62 if (!verifyTransactions(block.body)) {
63 return ValidationError::INVALID_TRANSACTIONS;
64 }
65
66 // there must exist a chain with the block in our storage, which is
67 // specified as a parent of the block we are validating; BlockTree takes
68 // care of this check and returns a specific error, if it fails
69 return outcome::success();
70 }

72 outcome::result<void> ProductionBlockValidator::validateHeader(
73 const primitives::BlockHeader &header,
74 const primitives::AuthorityId &authority_id,
75 const Threshold &threshold,
76 const Randomness &randomness) const {
77 log_->debug("Validates block signed by authority: {}",
78 authority_id.id.toHex());
79
80 // get PRODUCTION-specific digests, which must be inside of this block
81 OUTCOME_TRY((auto &&, production_digests), getProductionDigests(header));
82 auto [seal, production_header] = production_digests;
83
84 // signature in seal of the header must be valid
85 if (!verifySignature(header, production_header, seal, authority_id.id)) {
86 return ValidationError::INVALID_SIGNATURE;
87 }
88
89 // VRF must prove that the peer is the leader of the slot
90 if (!verifyVRF(production_header, authority_id.id, threshold, randomness)) {
91 return ValidationError::INVALID_VRF;
92 }
93
94 // peer must not send two blocks in one slot
95 if (!verifyProducer(production_header)) {
96 return ValidationError::TWO_BLOCKS_IN_SLOT;
97 }
98 return outcome::success();
99 }

101 bool ProductionBlockValidator::verifySignature(
102 const primitives::BlockHeader &header,
103 const ProductionBlockHeader &production_header,
104 const Seal &seal,
105 const primitives::SessionKey &public_key) const {
106 // firstly, take hash of the block's header without Seal, which is the last
107 // digest
108 auto unsealed_header = header;
109 unsealed_header.digest.pop_back();
110
111 auto unsealed_header_encoded = scale::encode(unsealed_header).value();
112
113 auto block_hash = hasher_->blake2b_256(unsealed_header_encoded);
114
115 // secondly, use verify function to check the signature
116 auto res =
117 sr25519_provider_->verify(seal.signature, block_hash, public_key);
118 return res && res.value();
119 }

121 bool ProductionBlockValidator::verifyVRF(const ProductionBlockHeader &production_header,
122 const primitives::SessionKey &public_key,
123 const Threshold &threshold,
124 const Randomness &randomness) const {
125 // verify VRF output
126 auto randomness_with_slot =
127 Buffer{}
128 .put(randomness)
129 .put(base::uint64_t_to_bytes(production_header.slot_number));
130 auto verify_res = vrf_provider_->verify(
131 randomness_with_slot, production_header.vrf_output, public_key, threshold);
132 if (! verify_res.is_valid) {
133 log_->error("VRF proof in block is not valid");
134 return false;
135 }
136
137 // verify threshold
138 if (! verify_res.is_less) {
139 log_->error("VRF value is not less than the threshold");
140 return false;
141 }
142
143 return true;
144 }

146 bool ProductionBlockValidator::verifyProducer(
147 const ProductionBlockHeader &production_header) const {
148 auto peer = production_header.authority_index;
149
150 auto slot_it = blocks_producers_.find(production_header.slot_number);
151 if (slot_it == blocks_producers_.end()) {
152 // this peer is the first producer in this slot
153 blocks_producers_.insert({production_header.slot_number, {peer}});
154 return true;
155 }
156
157 auto &slot = slot_it->second;
158 auto peer_in_slot = slot.find(peer);
159 if (peer_in_slot != slot.end()) {
160 log_->info("authority {} has already produced a block in the slot {}",
161 peer,
162 production_header.slot_number);
163 return false;
164 }
165
166 // OK
167 slot.insert(peer);
168 return true;
169 }

171 bool ProductionBlockValidator::verifyTransactions(
172 const primitives::BlockBody &block_body) const {
173 return std::all_of(
174 block_body.cbegin(), block_body.cend(), [this](const auto &ext) {
175 auto validation_res = tx_queue_->validate_transaction(ext);
176 if (!validation_res) {
177 log_->info("extrinsic validation failed: {}",
178 validation_res.error());
179 return false;
180 }
181 return visit_in_place(
182 validation_res.value(),
183 [](const primitives::ValidTransaction &) { return true; },
184 [](const auto &) { return false; });
185 });
186 }
