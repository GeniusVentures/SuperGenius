39 std::shared_ptr<BlockTreeImpl::TreeNode> BlockTreeImpl::TreeNode::getByHash(
40 const primitives::BlockHash &hash) {
41 // standard BFS
42 std::queue<std::shared_ptr<TreeNode>> nodes_to_scan;
43 nodes_to_scan.push(shared_from_this());
44 while (!nodes_to_scan.empty()) {
45 const auto &node = nodes_to_scan.front();
46 if (node->block_hash == hash) {
47 return node;
48 }
49 for (const auto &child : node->children) {
50 nodes_to_scan.push(child);
51 }
52 nodes_to_scan.pop();
53 }
54 return nullptr;
55 }

71 BlockTreeImpl::TreeMeta::TreeMeta(TreeNode &subtree_root_node)
72 : deepest_leaf{subtree_root_node}, last_finalized{subtree_root_node} {
73 std::function<void(std::shared_ptr<TreeNode>)> handle =
74 [&](std::shared_ptr<TreeNode> node) {
75 // avoid of deep recurse
76 while (node->children.size() == 1) {
77 node = node->children.front();
78 }
79
80 // is leaf
81 if (node->children.empty()) {
82 leaves.emplace(node->block_hash);
83
84 if (node->depth > deepest_leaf.get().depth) {
85 deepest_leaf = *node;
86 }
87 } else {
88 // follow descendants recursively
89 for (const auto &child : node->children) {
90 handle(child);
91 }
92 }
93 };
94
95 handle(subtree_root_node.shared_from_this());
96 }

199 outcome::result<void> BlockTreeImpl::finalize(
200 const primitives::BlockHash &block,
201 const primitives::Justification &justification) {
202 auto node = tree_->getByHash(block);
203 if (!node) {
204 return BlockTreeError::NO_SUCH_BLOCK;
205 }
206 if (storage_->getJustification(block)) {
207 // block was already finalized, fine
208 return outcome::success();
209 }
210
211 // insert justification into the database
212 BOOST_OUTCOME_TRYV2(auto &&, storage_->putJustification(justification, block, node->depth));
213
214 // update our local meta
215 node->finalized = true;
216
217 BOOST_OUTCOME_TRYV2(auto &&, prune(node));
218
219 tree_ = node;
220
221 tree_meta_ = std::make_shared<TreeMeta>(*tree_);
222
223 tree_->parent.reset();
224
225 BOOST_OUTCOME_TRYV2(auto &&, storage_->setLastFinalizedBlockHash(node->block_hash));
226
227 log_->info(
228 "Finalized block number {} with hash {}", node->depth, block.toHex());
229 return outcome::success();
230 }

252 BlockTreeImpl::BlockHashVecRes BlockTreeImpl::getChainByBlock(
253 const primitives::BlockHash &block, bool ascending, uint64_t maximum) {
254 auto block_number_res = header_repo_->getNumberByHash(block);
255 if (!block_number_res) {
256 log_->error("cannot retrieve block with hash {}: {}",
257 block,
258 block_number_res.error().message());
259 return BlockTreeError::NO_SUCH_BLOCK;
260 }
261 auto start_block_number = block_number_res.value();
262
263 primitives::BlockNumber finish_block_number; // NOLINT
264 if (ascending) {
265 if (start_block_number < maximum) {
266 // we want to finish at the root
267 finish_block_number = 0;
268 } else {
269 // some non-root block
270 finish_block_number = start_block_number - maximum;
271 }
272 } else {
273 auto finish_block_number_candidate = start_block_number + maximum;
274 auto current_depth = tree_meta_->deepest_leaf.get().depth;
275 if (finish_block_number_candidate <= current_depth) {
276 finish_block_number = finish_block_number_candidate;
277 } else {
278 finish_block_number = current_depth;
279 }
280 }
281
282 auto finish_block_hash = header_repo_->getHashByNumber(finish_block_number);
283 if (!finish_block_hash) {
284 log_->error("cannot retrieve block with number {}: {}",
285 finish_block_number,
286 finish_block_hash.error().message());
287 return BlockTreeError::NO_SUCH_BLOCK;
288 }
289
290 if (!ascending) {
291 return getChainByBlocks(block, finish_block_hash.value());
292 }
293
294 // the function returns the blocks in the chronological order, but we want a
295 // reverted one in this case
296 OUTCOME_TRY((auto &&, chain), getChainByBlocks(finish_block_hash.value(), block));
297 std::reverse(chain.begin(), chain.end());
298 return std::move(chain);
299 }

301 BlockTreeImpl::BlockHashVecRes BlockTreeImpl::getChainByBlocks(
302 const primitives::BlockHash &top_block,
303 const primitives::BlockHash &bottom_block) {
304 static constexpr std::string_view kNotAncestorError =
305 "impossible to get chain by blocks: most probably, block {} is "
306 "not an ancestor of {}";
307 std::vector<primitives::BlockHash> result;
308
309 auto top_block_node_ptr = tree_->getByHash(top_block);
310 auto bottom_block_node_ptr = tree_->getByHash(bottom_block);
311
312 // if both nodes are in our light tree, we can use this representation only
313 if (top_block_node_ptr && bottom_block_node_ptr) {
314 auto current_node = bottom_block_node_ptr;
315 while (current_node != top_block_node_ptr) {
316 result.push_back(current_node->block_hash);
317 if (auto parent = current_node->parent; !parent.expired()) {
318 current_node = parent.lock();
319 } else {
320 log_->warn(kNotAncestorError.data(), top_block, bottom_block);
321 return BlockTreeError::INCORRECT_ARGS;
322 }
323 }
324 result.push_back(top_block_node_ptr->block_hash);
325 std::reverse(result.begin(), result.end());
326 return result;
327 }
328
329 // else, we need to use a database
330 auto current_hash = bottom_block;
331 while (current_hash != top_block) {
332 result.push_back(current_hash);
333 auto current_header_res = header_repo_->getBlockHeader(current_hash);
334 if (!current_header_res) {
335 log_->error(kNotAncestorError.data(), top_block, bottom_block);
336 return BlockTreeError::INCORRECT_ARGS;
337 }
338 current_hash = current_header_res.value().parent_hash;
339 }
340 result.push_back(current_hash);
341 std::reverse(result.begin(), result.end());
342 return result;
343 }

345 bool BlockTreeImpl::hasDirectChain(const primitives::BlockHash &ancestor,
346 const primitives::BlockHash &descendant) {
347 auto ancestor_node_ptr = tree_->getByHash(ancestor);
348 auto descendant_node_ptr = tree_->getByHash(descendant);
349
350 // if both nodes are in our light tree, we can use this representation only
351 if (ancestor_node_ptr && descendant_node_ptr) {
352 auto current_node = descendant_node_ptr;
353 while (current_node != ancestor_node_ptr) {
354 if (auto parent = current_node->parent; !parent.expired()) {
355 current_node = parent.lock();
356 } else {
357 return false;
358 }
359 }
360 return true;
361 }
362
363 // else, we need to use a database
364 auto current_hash = descendant;
365 while (current_hash != ancestor) {
366 auto current_header_res = header_repo_->getBlockHeader(current_hash);
367 if (!current_header_res) {
368 return false;
369 }
370 current_hash = current_header_res.value().parent_hash;
371 }
372 return true;
373 }

385 outcome::result<primitives::BlockInfo> BlockTreeImpl::getBestContaining(
386 const primitives::BlockHash &target_hash,
387 const boost::optional<primitives::BlockNumber> &max_number) const {
388 OUTCOME_TRY((auto &&, target_header), header_repo_->getBlockHeader(target_hash));
389 if (max_number.has_value() && target_header.number > max_number.value()) {
390 return Error::TARGET_IS_PAST_MAX;
391 }
392 OUTCOME_TRY((auto &&, canon_hash),
393 header_repo_->getHashByNumber(target_header.number));
394 // if a max number is given we try to fetch the block at the
395 // given depth, if it doesn't exist or `max_number` is not
396 // provided, we continue to search from all leaves below.
397 if (canon_hash == target_hash) {
398 if (max_number.has_value()) {
399 auto header = header_repo_->getBlockHeader(max_number.value());
400 if (header) {
401 OUTCOME_TRY((auto &&, hash),
402 header_repo_->getHashByNumber(header.value().number));
403 return primitives::BlockInfo{header.value().number, hash};
404 }
405 }
406 } else {
407 OUTCOME_TRY((auto &&, last_finalized),
408 header_repo_->getNumberByHash(getLastFinalized().block_hash));
409 if (last_finalized >= target_header.number) {
410 return Error::BLOCK_ON_DEAD_END;
411 }
412 }
413 for (auto &leaf_hash : getLeavesSorted()) {
414 auto current_hash = leaf_hash;
415 auto best_hash = current_hash;
416 if (max_number.has_value()) {
417 OUTCOME_TRY((auto &&, hash), walkBackUntilLess(current_hash, max_number.value()));
418 best_hash = hash;
419 current_hash = hash;
420 }
421 OUTCOME_TRY((auto &&, best_header), header_repo_->getBlockHeader(best_hash));
422 while (true) {
423 OUTCOME_TRY((auto &&, current_header), header_repo_->getBlockHeader(current_hash));
424 if (current_hash == target_hash) {
425 return primitives::BlockInfo{best_header.number, best_hash};
426 }
427 if (current_header.number < target_header.number) {
428 break;
429 }
430 current_hash = current_header.parent_hash;
431 }
432 }
433
434 log_->warn(
435 "Block {:?} exists in chain but not found when following all \
436 leaves backwards. Max block number = {:?}",
437 target_hash,
438 max_number.has_value() ? max_number.value() : -1);
439 return Error::BLOCK_NOT_FOUND;
440 }

473 std::vector<primitives::BlockHash> BlockTreeImpl::getLeavesSorted() const {
474 std::vector<primitives::BlockInfo> leaf_depths;
475 auto leaves = getLeaves();
476 leaf_depths.reserve(leaves.size());
477 for (auto &leaf : leaves) {
478 auto leaf_node = tree_->getByHash(leaf);
479 leaf_depths.emplace_back(
480 primitives::BlockInfo{leaf_node->depth, leaf_node->block_hash});
481 }
482 std::sort(leaf_depths.begin(),
483 leaf_depths.end(),
484 [](auto const &p1, auto const &p2) {
485 return p1.block_number > p2.block_number;
486 });
487 std::vector<primitives::BlockHash> leaf_hashes;
488 leaf_hashes.reserve(leaf_depths.size());
489 std::transform(leaf_depths.begin(),
490 leaf_depths.end(),
491 std::back_inserter(leaf_hashes),
492 [](auto &p) { return p.block_hash; });
493 return leaf_hashes;
494 }

509 outcome::result<void> BlockTreeImpl::prune(
510 const std::shared_ptr<TreeNode> &lastFinalizedNode) {
511 std::vector<std::pair<primitives::BlockHash, primitives::BlockNumber>>
512 to_remove;
513
514 auto current_node = lastFinalizedNode;
515
516 for (;;) {
517 auto parent_node = current_node->parent.lock();
518 if (!parent_node || parent_node->finalized) {
519 break;
520 }
521
522 auto main_chain_node = current_node;
523 current_node = parent_node;
524
525 // collect hashes for removing (except main chain block)
526 for (const auto &child : current_node->children) {
527 if (child->block_hash != main_chain_node->block_hash) {
528 collectDescendants(child, to_remove);
529 to_remove.emplace_back(child->block_hash, child->depth);
530 }
531 }
532
533 // remove (in memory) all child, except main chain block
534 current_node->children = {main_chain_node};
535 }
536
537 std::vector<primitives::Extrinsic> extrinsics;
538
539 // remove from storage
540 for (const auto &[hash, number] : to_remove) {
541 auto block_body_res = storage_->getBlockBody(hash);
542 if (block_body_res.has_value()) {
543 extrinsics.reserve(extrinsics.size() + block_body_res.value().size());
544 for (auto &&extrinsic : block_body_res.value()) {
545 extrinsics.emplace_back(std::move(extrinsic));
546 }
547 }
548
549 BOOST_OUTCOME_TRYV2(auto &&, storage_->removeBlock(hash, number));
550 }
551
552 // trying to return back extrinsics to transaction pool
553 for (auto &&extrinsic : extrinsics) {
554 auto result = extrinsic_observer_->onTxMessage(extrinsic);
555 if (result) {
556 log_->debug("Reapplied tx {}", result.value());
557 } else {
558 log_->debug("Skipped tx: {}", result.error().message());
559 }
560 }
561
562 return outcome::success();
563 }

565 void BlockTreeImpl::collectDescendants(
566 std::shared_ptr<TreeNode> node,
567 std::vector<std::pair<primitives::BlockHash, primitives::BlockNumber>>
568 &container) {
569 // avoid deep recursion
570 while (node->children.size() == 1) {
571 container.emplace_back(node->block_hash, node->depth);
572 node = node->children.front();
573 }
574
575 // collect descendants' hashes recursively
576 for (const auto &child : node->children) {
577 collectDescendants(child, container);
578 container.emplace_back(child->block_hash, child->depth);
579 }
580 }
