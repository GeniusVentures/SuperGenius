78 return false;

101 epoch.randomness = epoch_digest.randomness;
102 epoch.epoch_duration = genesis_configuration_->epoch_length;

170 void ProductionImpl::onBlockAnnounce(const network::BlockAnnounce &announce) {
171 switch (current_state_) {
172 case ProductionState::WAIT_BLOCK:
173 // TODO(kamilsa): PRE-366 validate block. Now it is problematic as we
174 // need t know VRF threshold for validation. To calculate that we need
175 // to have weights of the authorities and to get it we need to have
176 // the latest state of a blockchain
177
178 // synchronize missing blocks with their bodies
179 log_->info("Catching up to block number: {}", announce.header.number);
180 current_state_ = ProductionState::CATCHING_UP;
181 block_executor_->requestBlocks(
182 announce.header, [self_weak{weak_from_this()}] {
183 if (auto self = self_weak.lock()) {
184 self->log_->info("Catching up is done, getting slot time");
185 // all blocks were successfully applied, now we need to get
186 // slot time
187 self->current_state_ = ProductionState::NEED_SLOT_TIME;
188 }
189 });
190 break;
191 case ProductionState::NEED_SLOT_TIME:
192 // if block is new add it to the storage and sync missing blocks. Then
193 // calculate slot time and execute production
194 block_executor_->processNextBlock(
195 announce.header,
196 [this](const auto &header) { synchronizeSlots(header); });
197 break;
198 case ProductionState::CATCHING_UP:
199 case ProductionState::SYNCHRONIZED:
200 block_executor_->processNextBlock(announce.header, [](auto &) {});
201 break;
202 }
203 }

212 void ProductionImpl::runSlot() {
213 bool rewind_slots;
214 do {
215 if (current_slot_ != 0
216 && current_slot_ % genesis_configuration_->epoch_length == 0) {
217 // end of the epoch
218 finishEpoch();
219 }
220 log_->info("starting a slot {} in epoch {}",
221 current_slot_,
222 current_epoch_.epoch_index);
223
224 // check that we are really in the middle of the slot, as expected; we can
225 // cooperate with a relatively little (kMaxLatency) latency, as our node
226 // will be able to retrieve
227 auto now = clock_->now();
228
229 rewind_slots = now > next_slot_finish_time_
230 && (now - next_slot_finish_time_)
231 > genesis_configuration_->slot_duration;
232
233 if (rewind_slots) {
234 // we are too far behind; after skipping some slots (but not epochs)
235 // control will be returned to this method
236
237 current_slot_++;
238 next_slot_finish_time_ += genesis_configuration_->slot_duration;
239 }
240 } while (rewind_slots);
241
242 // everything is OK: wait for the end of the slot
243 timer_->expiresAt(next_slot_finish_time_);
244 timer_->asyncWait([this](auto &&ec) {
245 if (ec) {
246 log_->error("error happened while waiting on the timer: {}",
247 ec.message());
248 return;
249 }
250 finishSlot();
251 });
252 }

254 void ProductionImpl::finishSlot() {
255 auto slot_leadership =
256 slots_leadership_[current_slot_ % current_epoch_.epoch_duration];
257 if (slot_leadership) {
258 log_->debug("Peer {} is leader (vrfOutput: {}, proof: {})",
259 keypair_.public_key.toHex(),
260 base::Buffer(slot_leadership->output).toHex(),
261 base::Buffer(slot_leadership->proof).toHex());
262
263 processSlotLeadership(*slot_leadership);
264 }
265
266 ++current_slot_;
267 next_slot_finish_time_ += genesis_configuration_->slot_duration;
268 log_->debug("Slot {} in epoch {} has finished",
269 current_slot_,
270 current_epoch_.epoch_index);
271 runSlot();
272 }

289 primitives::Seal ProductionImpl::sealBlock(const primitives::Block &block) const {
290 auto pre_seal_encoded_block = scale::encode(block.header).value();
291
292 auto pre_seal_hash = hasher_->blake2b_256(pre_seal_encoded_block);
293
294 Seal seal{};
295 seal.signature.fill(0);
296 sr25519_sign(seal.signature.data(),
297 keypair_.public_key.data(),
298 keypair_.secret_key.data(),
299 pre_seal_hash.data(),
300 decltype(pre_seal_hash)::size());
301 auto encoded_seal = base::Buffer(scale::encode(seal).value());
302 return primitives::Seal{{primitives::kProductionEngineId, encoded_seal}};

305 void ProductionImpl::processSlotLeadership(const crypto::VRFOutput &output) {
306 // build a block to be announced
307 log_->info("Obtained slot leadership");
308
309 primitives::InherentData inherent_data;
310 auto now = std::chrono::duration_cast<std::chrono::milliseconds>(
311 clock_->now().time_since_epoch())
312 .count();
313 // identifiers are guaranteed to be correct, so use .value() directly
314 auto put_res = inherent_data.putData<uint64_t>(kTimestampId, now);
315 if (!put_res) {
316 return log_->error("cannot put an inherent data: {}",
317 put_res.error().message());
318 }
319 put_res = inherent_data.putData(kProdSlotId, current_slot_);
320 if (!put_res) {
321 return log_->error("cannot put an inherent data: {}",
322 put_res.error().message());
323 }
324
325 auto best_block_info = block_tree_->deepestLeaf();
326 auto &[best_block_number, best_block_hash] = best_block_info;
327 log_->info("Production builds block on top of block with number {} and hash {}",
328 best_block_number,
329 best_block_hash);
330
331 auto authority_index_res =
332 getAuthorityIndex(current_epoch_.authorities, keypair_.public_key);
333 BOOST_ASSERT_MSG(authority_index_res.has_value(), "Authority is not known");
334 // calculate production_pre_digest
335 auto production_pre_digest_res =
336 productionPreDigest(output, authority_index_res.value());
337 if (! production_pre_digest_res) {
338 return log_->error("cannot propose a block: {}",
339 production_pre_digest_res.error().message());
340 }
341 auto production_pre_digest = production_pre_digest_res.value();
342
343 // create new block
344 auto pre_seal_block_res =
345 proposer_->propose(best_block_hash, inherent_data, {production_pre_digest});
346 if (!pre_seal_block_res) {
347 return log_->error("cannot propose a block: {}",
348 pre_seal_block_res.error().message());
349 }
350
351 auto block = pre_seal_block_res.value();
352
353 if (auto next_epoch_digest_res = getNextEpochDigest(block.header);
354 next_epoch_digest_res) {
355 log_->info("Obtained next epoch digest");
356 if (! epoch_storage_->addEpochDescriptor(
357 current_epoch_.epoch_index + 2, next_epoch_digest_res.value())) {
358 log_->error("Could not add nex epoch digest to epoch storage");
359 }
360 }
361
362 // seal the block
363 auto seal = sealBlock(block);
364
365 // add seal digest item
366 block.header.digest.emplace_back(seal);
367
368 // check that we are still in the middle of the
369 auto current_time = clock_->now();
370 auto time_diff = current_time - next_slot_finish_time_;
371 log_->warn(
372 "=============================================. time diff {} , genesis_configuration {}",
373 time_diff.count(), genesis_configuration_->slot_duration.count());
374 if (current_time
375 > next_slot_finish_time_ + genesis_configuration_->slot_duration) {
376 log_->warn(
377 "Block was not built in time. Slot has finished. If you are "
378 "executing in debug mode, consider to rebuild in release. time diff {} , genesis_configuration {}",
379 time_diff.count(), genesis_configuration_->slot_duration.count());
380 // return;
381 }
382
383 // observe possible changes of authorities
384 for (auto &digest_item : block.header.digest) {
385 visit_in_place(
386 digest_item,
387 [&](const primitives::Verification &verification_message) {
388 [[maybe_unused]] auto res = authority_update_observer_->onVerification(
389 verification_message.verification_engine_id,
390 best_block_info,
391 verification_message);
392 },
393 [](const auto &) {});
394 }
395
396 // add block to the block tree
397 if (auto add_res = block_tree_->addBlock(block); ! add_res) {
398 log_->error("Could not add block: {}", add_res.error().message());
399 return;
400 }
401
402 auto next_epoch_digest_res = getNextEpochDigest(block.header);
403 if (next_epoch_digest_res) {
404 log_->info("Got next epoch digest for epoch: {}",
405 current_epoch_.epoch_index + 2);
406 if (auto add_epoch_res = epoch_storage_->addEpochDescriptor(
407 current_epoch_.epoch_index + 2, next_epoch_digest_res.value());
408 ! add_epoch_res) {
409 log_->error("Could not add next epoch digest. Reason: {}",
410 add_epoch_res.error().message());
411 }
412 }
413
414 // finally, broadcast the sealed block
415 gossiper_->blockAnnounce(network::BlockAnnounce{block.header});
416 log_->debug("Announced block number {} in slot {} with timestamp {}",
417 block.header.number,
418 current_slot_,
419 now);
420 }

433 void ProductionImpl::finishEpoch() {
434 // compute new randomness
435 auto next_epoch_digest_res =
436 epoch_storage_->getEpochDescriptor(++current_epoch_.epoch_index);
437 if (! next_epoch_digest_res) {
438 log_->error("Next epoch digest does not exist for epoch {}",
439 current_epoch_.epoch_index);
440 // TODO (kamilsa): uncomment `return;` and remove next line when
441 // PRE-364 is done. For now assume no changes after epoch
442 //
443 // return.
444 next_epoch_digest_res =
445 NextEpochDescriptor{/*.authorities =*/ current_epoch_.authorities,
446 /*.randomness =*/ current_epoch_.randomness};
447 }
448
449 current_epoch_.start_slot = current_slot_;
450 current_epoch_.authorities = next_epoch_digest_res.value().authorities;
451 current_epoch_.randomness = next_epoch_digest_res.value().randomness;
452 slots_leadership_ = getEpochLeadership(current_epoch_);
453
454 log_->debug("Epoch {} has finished", current_epoch_.epoch_index);

457 void ProductionImpl::synchronizeSlots(const primitives::BlockHeader &new_header) {
458 static boost::optional<ProductionSlotNumber> first_production_slot = boost::none;
459
460 auto production_digests_res = getProductionDigests(new_header);
461 if (! production_digests_res) {
462 log_->error("Could not get digests: {}",
463 production_digests_res.error().message());
464 }
465
466 auto [_, production_header] = production_digests_res.value();
467 auto observed_slot = production_header.slot_number;
468
469 if (! first_production_slot) {
470 first_production_slot = observed_slot + kSlotTail;
471 log_->info("Peer will start produce blocks at slot number: {}",
472 *first_production_slot);
473 }
474
475 // get the difference between observed slot and the one that we are trying
476 // to launch
477 auto diff = *first_production_slot - observed_slot;
478
479 first_slot_times_.emplace_back(
480 clock_->now() + diff * genesis_configuration_->slot_duration);
481 if (observed_slot >= first_production_slot.value()) {
482 current_state_ = ProductionState::SYNCHRONIZED;
483 log_->info("Slot time obtained. Peer is synchronized");
484
485 // get median as here:
486 // https://en.cppreference.com/w/cpp/algorithm/nth_element
487 std::nth_element(first_slot_times_.begin(),
488 first_slot_times_.begin() + first_slot_times_.size() / 2,
489 first_slot_times_.end());
490 ProductionTimePoint first_slot_ending_time =
491 first_slot_times_[first_slot_times_.size() / 2];
492
493 Epoch epoch;
494 epoch.epoch_index =
495 *first_production_slot / genesis_configuration_->epoch_length;
496 epoch.start_slot = *first_production_slot;
497 epoch.epoch_duration = genesis_configuration_->epoch_length;
498 auto next_epoch_digest_res =
499 epoch_storage_->getEpochDescriptor(epoch.epoch_index);
500 if (! next_epoch_digest_res) {
501 log_->error("Could not fetch epoch descriptor for epoch {}. Reason: {}",
502 epoch.epoch_index,
503 next_epoch_digest_res.error().message());
504 return;
505 }
506 epoch.randomness = next_epoch_digest_res.value().randomness;
507 epoch.authorities = next_epoch_digest_res.value().authorities;
508
509 runEpoch(epoch, first_slot_ending_time);
510 }
