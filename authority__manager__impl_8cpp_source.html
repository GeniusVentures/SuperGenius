29 bool AuthorityManagerImpl::prepare() {
30 auto encoded_root_res = storage_->get(SCHEDULER_TREE);
31 if (!encoded_root_res.has_value()) {
32 // Get initial authorities from genesis
33 root_ = ScheduleNode::createAsRoot({});
34 root_->actual_authorities = std::make_shared<primitives::AuthorityList>(
35 genesis_configuration_->genesis_authorities);
36 return true;
37 }
38
39 auto root_res =
40 scale::decode<std::shared_ptr<ScheduleNode>>(encoded_root_res.value());
41 if (!root_res.has_value()) {
42 log_->critical("Can't decode stored state");
43 return false;
44 }
45
46 root_ = std::move(root_res.value());
47 return true;
48 }

69 outcome::result<std::shared_ptr<const primitives::AuthorityList>>
70 AuthorityManagerImpl::authorities(const primitives::BlockInfo &block) {
71 auto node = getAppropriateAncestor(block);
72
73 if (! node) {
74 return AuthorityManagerError::ORPHAN_BLOCK_OR_ALREADY_FINALISED;
75 }
76
77 auto adjusted_node =
78 (node->block == block) ? node : node->makeDescendant(block);
79
80 if (adjusted_node->enabled) {
81 // Original authorities
82 return adjusted_node->actual_authorities;
83 }
84
85 // Zero-weighted authorities
86 auto authorities = std::make_shared<primitives::AuthorityList>(
87 *adjusted_node->actual_authorities);
88 std::for_each(authorities->begin(),
89 authorities->end(),
90 [](auto &authority) { authority.weight = 0; });
91 return std::move(authorities);
92 }

94 outcome::result<void> AuthorityManagerImpl::applyScheduledChange(
95 const primitives::BlockInfo &block,
96 const primitives::AuthorityList &authorities,
97 primitives::BlockNumber activate_at) {
98 auto node = getAppropriateAncestor(block);
99
100 auto new_node = node->makeDescendant(block);
101
102 BOOST_OUTCOME_TRYV2(auto &&, new_node->ensureReadyToSchedule());
103
104 // Schedule change
105 new_node->scheduled_authorities =
106 std::make_shared<primitives::AuthorityList>(authorities);
107 new_node->scheduled_after = activate_at;
108
109 // Reorganize ancestry
110 for (auto &descendant : std::move(node->descendants)) {
111 auto &ancestor =
112 directChainExists(block, descendant->block) ? new_node : node;
113
114 if (descendant->block.block_number >= ancestor->forced_for) {
115 descendant->actual_authorities = ancestor->forced_authorities;
116 descendant->forced_authorities.reset();
117 descendant->forced_for = ScheduleNode::INACTIVE;
118 }
119
120 ancestor->descendants.emplace_back(std::move(descendant));
121 }
122 node->descendants.emplace_back(std::move(new_node));
123
124 return outcome::success();
125 }

127 outcome::result<void> AuthorityManagerImpl::applyForcedChange(
128 const primitives::BlockInfo &block,

132
133 auto new_node = node->makeDescendant(block);
134
135 BOOST_OUTCOME_TRYV2(auto &&, new_node->ensureReadyToSchedule());
136
137 // Force changes
138 if (new_node->block.block_number >= activate_at) {
139 new_node->actual_authorities =
140 std::make_shared<primitives::AuthorityList>(authorities);
141 } else {
142 new_node->forced_authorities =
143 std::make_shared<primitives::AuthorityList>(authorities);
144 new_node->forced_for = activate_at;
145 }
146
147 // Reorganize ancestry
148 for (auto &descendant : std::move(node->descendants)) {
149 auto &ancestor =
150 directChainExists(block, descendant->block) ? new_node : node;
151
152 // Apply forced changes if dalay will be passed for descendant
153 if (descendant->block.block_number >= ancestor->forced_for) {
154 descendant->actual_authorities = ancestor->forced_authorities;
155 descendant->forced_authorities.reset();
156 descendant->forced_for = ScheduleNode::INACTIVE;

159 descendant->enabled = true;

161 }
162
163 ancestor->descendants.emplace_back(std::move(descendant));
164 }

170 outcome::result<void> AuthorityManagerImpl::applyOnDisabled(
171 const primitives::BlockInfo &block, uint64_t authority_index) {
172 auto node = getAppropriateAncestor(block);
173
174 auto new_node = node->makeDescendant(block);
175
176 // Check if index not out of bound
177 if (authority_index >= node->actual_authorities->size()) {
178 return AuthorityUpdateObserverError::WRONG_AUTHORITY_INDEX;
179 }
180
181 // Make changed authorities
182 auto authorities = std::make_shared<primitives::AuthorityList>(
183 *new_node->actual_authorities);
184 (*authorities)[authority_index].weight = 0;
185 new_node->actual_authorities = std::move(authorities);
186
187 // Reorganize ancestry
188 for (auto &descendant : std::move(node->descendants)) {
189 if (directChainExists(block, descendant->block)) {
190 // Propogate change to descendants
191 if (descendant->actual_authorities == node->actual_authorities) {
192 descendant->actual_authorities = new_node->actual_authorities;
193 }
194 new_node->descendants.emplace_back(std::move(descendant));
195 } else {
196 node->descendants.emplace_back(std::move(descendant));
197 }
198 }
199 node->descendants.emplace_back(std::move(new_node));
200
201 return outcome::success();
202 }

258 outcome::result<void> AuthorityManagerImpl::onVerification(
259 const primitives::VerificationEngineId &engine_id,
260 const primitives::BlockInfo &block,
261 const primitives::Verification &message) {
262 if (std::find(known_engines.begin(), known_engines.end(), engine_id)
263 == known_engines.end()) {
264 return AuthorityManagerError::UNKNOWN_ENGINE_ID;
265 }
266
267 return visit_in_place(
268 message.payload,
269 [this, &block](
270 const primitives::ScheduledChange &msg) -> outcome::result<void> {
271 return applyScheduledChange(
272 block, msg.authorities, block.block_number + msg.subchain_lenght);
273 },
274 [this, &block](const primitives::ForcedChange &msg) {
275 return applyForcedChange(
276 block, msg.authorities, block.block_number + msg.subchain_lenght);
277 },
278 [this, &block](const primitives::OnDisabled &msg) {
279 return applyOnDisabled(block, msg.authority_index);
280 },
281 [this, &block](const primitives::Pause &msg) {
282 return applyPause(block, block.block_number + msg.subchain_lenght);
283 },
284 [this, &block](const primitives::Resume &msg) {
285 return applyResume(block, block.block_number + msg.subchain_lenght);
286 },
287 [](auto &) {
288 return AuthorityUpdateObserverError::UNSUPPORTED_MESSAGE_TYPE;
289 });
290 }

292 outcome::result<void> AuthorityManagerImpl::onFinalize(
293 const primitives::BlockInfo &block) {
294 if (block == root_->block) {
295 return outcome::success();
296 }
297
298 if (block.block_number < root_->block.block_number) {
299 return AuthorityManagerError::WRONG_FINALISATION_ORDER;
300 }
301
302 auto node = getAppropriateAncestor(block);
303
304 // Create new node
305 auto new_node = node->makeDescendant(block, true);
306
307 // Reorganize ancestry
308 for (auto &descendant : std::move(node->descendants)) {
309 if (directChainExists(block, descendant->block)) {
310 new_node->descendants.emplace_back(std::move(descendant));
311 }
312 }
313
314 root_ = std::move(new_node);
315
316 OUTCOME_TRY((auto &&, encoded_state), scale::encode(root_));
317 BOOST_OUTCOME_TRYV2(auto &&, storage_->put(SCHEDULER_TREE,
318 base::Buffer(std::move(encoded_state))));
319
320 return outcome::success();
321 }

323 std::shared_ptr<ScheduleNode> AuthorityManagerImpl::getAppropriateAncestor(
324 const primitives::BlockInfo &block) {
325 BOOST_ASSERT(root_ != nullptr);
326 std::shared_ptr<ScheduleNode> ancestor;
327 // Target block is not descendant of the current root
328 if (root_->block.block_number > block.block_number
329 || (root_->block != block
330 && ! directChainExists(root_->block, block))) {
331 return ancestor;
332 }
333 ancestor = root_;
334 while (ancestor->block != block) {
335 bool goto_next_generation = false;
336 for (const auto &node : ancestor->descendants) {
337 if (node->block == block || directChainExists(node->block, block)) {
338 ancestor = node;
339 goto_next_generation = true;
340 break;
341 }
342 }
343 if (! goto_next_generation) {
344 break;
345 }
346 }
347 return ancestor;
348 }

350 bool AuthorityManagerImpl::directChainExists(
351 const primitives::BlockInfo &ancestor,
352 const primitives::BlockInfo &descendant) {
353 // Any block is descendant of genesis
354 if (ancestor.block_number <= 1
355 && ancestor.block_number < descendant.block_number) {
356 return true;
357 }
358 auto result = ancestor.block_number < descendant.block_number
359 && block_tree_->hasDirectChain(ancestor.block_hash,
360 descendant.block_hash);
361 return result;
362 }
