30 // First do a pass with a read to see if any writing needs doing, this saves needing to open a write lock (and potentially blocking)
31 auto is_initialized (false);

33 auto transaction (store.tx_begin_read ());
34 is_initialized = (store.latest_begin (transaction) != store.latest_end ());
35 }
36
37 sgns::genesis genesis;
38 if (!is_initialized)
39 {
40 release_assert (!flags.read_only);
41 auto transaction (store.tx_begin_write ({ tables::accounts, tables::cached_counts, tables::confirmation_height, tables::frontiers, tables::open_blocks }));
42 // Store was empty meaning we just created it, add the genesis block

173{
174 // Faster account calculation
175 account_a = block_a->account ();
176 if (account_a.is_zero ())
177 {
178 account_a = block_a->sideband ().account;
179 }
180 // Faster amount calculation
181 auto previous (block_a->previous ());
182 auto previous_balance (ledger.balance (transaction_a, previous));
183 auto block_balance (store.block_balance_calculated (block_a));
184 if (hash_a != ledger.network_params.ledger.genesis_account)
185 {
186 amount_a = block_balance > previous_balance ? block_balance - previous_balance : previous_balance - block_balance;
187 }
188 else
189 {
190 amount_a = ledger.network_params.ledger.genesis_amount;
191 }
192 if (auto state = dynamic_cast<sgns::state_block *> (block_a.get ()))
193 {
194 if (state->hashables.balance < previous_balance)
195 {
196 is_state_send_a = true;
197 }
198 pending_account_a = state->hashables.link;
199 }
200 if (auto send = dynamic_cast<sgns::send_block *> (block_a.get ()))
201 {
202 pending_account_a = send->hashables.destination;
