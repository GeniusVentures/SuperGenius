223bool sgns::logging::upgrade_json (unsigned version_a, sgns::jsonconfig & json)
224{
225 json.put ("version", json_version ());
226 switch (version_a)
227 {
228 case 1:
229 json.put ("vote", vote_logging_value);
230 case 2:
231 json.put ("rotation_size", rotation_size);
232 json.put ("flush", true);
233 case 3:
234 json.put ("network_node_id_handshake", false);
235 case 4:
236 json.put ("upnp_details", "false");
237 json.put ("timing", "false");
238 case 5:
239 uintmax_t config_max_size;
240 json.get<uintmax_t> ("max_size", config_max_size);
241 max_size = std::max (max_size, config_max_size);
242 json.put ("max_size", max_size);
243 json.put ("log_ipc", true);
244 case 6:
245 json.put ("min_time_between_output", min_time_between_log_output.count ());
246 json.put ("network_timeout", network_timeout_logging_value);
247 json.erase ("log_rpc");
248 break;
249 case 7:
250 json.put ("single_line_record", single_line_record_value);
251 case 8:
252 break;
253 default:
254 throw std::runtime_error ("Unknown logging_config version");
255 break;
256 }
257 return version_a < json_version ();
258}
