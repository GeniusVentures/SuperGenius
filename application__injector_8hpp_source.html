246 template <typename Injector>
247 sptr<blockchain::BlockStorage> get_block_storage(const Injector &injector) {
248 static auto initialized =
249 boost::optional<sptr<blockchain::BlockStorage>>(boost::none);
250
251 if (initialized) {
252 return initialized.value();
253 }
254 auto &&hasher = injector.template create<sptr<crypto::Hasher>>();
255
256 const auto &db = injector.template create<sptr<storage::BufferStorage>>();
257
258 const auto &trie_storage =
259 injector.template create<sptr<storage::trie::TrieStorage>>();
260
261 auto storage = blockchain::KeyValueBlockStorage::create(
262 trie_storage->getRootHash(),
263 db,
264 hasher,
265 [&db, &injector](const primitives::Block &genesis_block) {
266 // handle genesis initialization, which happens when there is not
267 // authorities and last completed round in the storage
268 if (! db->get(storage::kAuthoritySetKey)) {
269 // insert authorities
270 auto finality_api =
271 injector.template create<sptr<runtime::FinalityApi>>();
272 const auto &weighted_authorities_res = finality_api->authorities(
273 primitives::BlockId(primitives::BlockNumber{0}));
274 BOOST_ASSERT_MSG(weighted_authorities_res,
275 "finality_api_->authorities failed");
276 const auto &weighted_authorities = weighted_authorities_res.value();
277
278 for (const auto authority : weighted_authorities) {
279 spdlog::info("Finality authority: {}", authority.id.id.toHex());
280 }
281
282 verification::finality::VoterSet voters{0};
283 for (const auto &weighted_authority : weighted_authorities) {
284 voters.insert(weighted_authority.id.id,
285 weighted_authority.weight);
286 spdlog::debug("Added to finality authorities: {}, weight: {}",
287 weighted_authority.id.id.toHex(),
288 weighted_authority.weight);
289 }
290 BOOST_ASSERT_MSG(voters.size() != 0, "Finality voters are empty");
291 auto authorities_put_res =
292 db->put(storage::kAuthoritySetKey,
293 base::Buffer(scale::encode(voters).value()));
294 if (! authorities_put_res) {
295 BOOST_ASSERT_MSG(false, "Could not insert authorities");
296 std::exit(1);
297 }
298 }
299 });
300 if (storage.has_error()) {
301 base::raise(storage.error());
302 }
303 initialized = storage.value();
304 return initialized.value();
305 }

523 res->clients.emplace_back(
524 std::make_shared<network::RemoteSyncProtocolClient>(
525 *host, std::move(peer_info)));
526 } else {
527 res->clients.emplace_back(
528 std::make_shared<network::DummySyncProtocolClient>());
529 }
530 }
531 std::reverse(res->clients.begin(), res->clients.end());
532 initialized = res;
