97 outcome::result<rocksdb::QueryResult> rocksdb::query(const Buffer& keyPrefix) const
98 {
99 ReadOptions read_options = ro_;
100 read_options.auto_prefix_mode = true; //Adaptive Prefix Mode
101
102 auto strKeyPrefix = std::string(keyPrefix.toString());
103
104 Buffer key;
105 Buffer value;
106
107 QueryResult results;
108 auto iter = db_->NewIterator(read_options);
109 auto slicePrefix = make_slice(keyPrefix);
110 for (iter->Seek(slicePrefix); iter->Valid() && iter->key().starts_with(slicePrefix); iter->Next())
111 {
112 key.clear();
113 key.put(iter->key().ToString());
114
115 value.clear();
116 value.put(iter->value().ToString());
117 results[key] = value;
118 }
119 delete iter;
120 return results;
121 }
